135 double QGis::fromUnitToUnitFactor( QGis::UnitType fromUnit, QGis::UnitType toUnit )
136 {
137 #define DEGREE_TO_METER 111319.49079327358
138 #define FEET_TO_METER 0.3048
139 #define NMILE_TO_METER 1852.0
140 
141  // Unify degree units
142  if ( fromUnit == QGis::DecimalDegrees || fromUnit == QGis::DegreesMinutesSeconds || fromUnit == QGis::DegreesDecimalMinutes )
143  fromUnit = QGis::Degrees;
144  if ( toUnit == QGis::DecimalDegrees || toUnit == QGis::DegreesMinutesSeconds || toUnit == QGis::DegreesDecimalMinutes )
145  toUnit = QGis::Degrees;
146 
147  // Calculate the conversion factor between the specified units
148  if ( fromUnit != toUnit && fromUnit != QGis::UnknownUnit && toUnit != QGis::UnknownUnit )
149  {
150  switch ( fromUnit )
151  {
152  case QGis::Meters:
153  {
154  if ( toUnit == QGis::Feet ) return 1.0 / FEET_TO_METER;
155  if ( toUnit == QGis::Degrees ) return 1.0 / DEGREE_TO_METER;
156  if ( toUnit == QGis::NauticalMiles ) return 1.0 / NMILE_TO_METER;
157  break;
158  }
159  case QGis::Feet:
160  {
161  if ( toUnit == QGis::Meters ) return FEET_TO_METER;
162  if ( toUnit == QGis::Degrees ) return FEET_TO_METER / DEGREE_TO_METER;
163  if ( toUnit == QGis::NauticalMiles ) return FEET_TO_METER / NMILE_TO_METER;
164  break;
165  }
166  case QGis::Degrees:
167  {
168  if ( toUnit == QGis::Meters ) return DEGREE_TO_METER;
169  if ( toUnit == QGis::Feet ) return DEGREE_TO_METER / FEET_TO_METER;
170  if ( toUnit == QGis::NauticalMiles ) return DEGREE_TO_METER / NMILE_TO_METER;
171  break;
172  }
173  case QGis::NauticalMiles:
174  {
175  if ( toUnit == QGis::Meters ) return NMILE_TO_METER;
176  if ( toUnit == QGis::Feet ) return NMILE_TO_METER / FEET_TO_METER;
177  if ( toUnit == QGis::Degrees ) return NMILE_TO_METER / DEGREE_TO_METER;
178  break;
179  }
180  case QGis::UnknownUnit:
181  break;
182  }
183  }
184  return 1.0;
185 }

222 bool qgsVariantLessThan( const QVariant& lhs, const QVariant& rhs )
223 {
224  switch ( lhs.type() )
225  {
226  case QVariant::Int:
227  return lhs.toInt() < rhs.toInt();
228  case QVariant::UInt:
229  return lhs.toUInt() < rhs.toUInt();
230  case QVariant::LongLong:
231  return lhs.toLongLong() < rhs.toLongLong();
232  case QVariant::ULongLong:
233  return lhs.toULongLong() < rhs.toULongLong();
234  case QVariant::Double:
235  return lhs.toDouble() < rhs.toDouble();
236  case QVariant::Char:
237  return lhs.toChar() < rhs.toChar();
238  case QVariant::Date:
239  return lhs.toDate() < rhs.toDate();
240  case QVariant::Time:
241  return lhs.toTime() < rhs.toTime();
242  case QVariant::DateTime:
243  return lhs.toDateTime() < rhs.toDateTime();
244  default:
245  return QString::localeAwareCompare( lhs.toString(), rhs.toString() ) < 0;
246  }
247 }
