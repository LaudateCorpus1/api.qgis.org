180 QVariant QgsVectorLayerUtils::createUniqueValue( const QgsVectorLayer *layer, int fieldIndex, const QVariant &seed )
181 {
182  if ( !layer )
183  return QVariant();
184 
185  QgsFields fields = layer->fields();
186 
187  if ( fieldIndex < 0 || fieldIndex >= fields.count() )
188  return QVariant();
189 
190  QgsField field = fields.at( fieldIndex );
191 
192  if ( field.isNumeric() )
193  {
194  QVariant maxVal = layer->maximumValue( fieldIndex );
195  QVariant newVar( maxVal.toLongLong() + 1 );
196  if ( field.convertCompatible( newVar ) )
197  return newVar;
198  else
199  return QVariant();
200  }
201  else
202  {
203  switch ( field.type() )
204  {
205  case QVariant::String:
206  {
207  QString base;
208  if ( seed.isValid() )
209  base = seed.toString();
210 
211  if ( !base.isEmpty() )
212  {
213  // strip any existing _1, _2 from the seed
214  QRegularExpression rx( QStringLiteral( "(.*)_\\d+" ) );
215  QRegularExpressionMatch match = rx.match( base );
216  if ( match.hasMatch() )
217  {
218  base = match.captured( 1 );
219  }
220  }
221  else
222  {
223  // no base seed - fetch first value from layer
224  QgsFeatureRequest req;
225  req.setLimit( 1 );
226  req.setSubsetOfAttributes( QgsAttributeList() << fieldIndex );
227  req.setFlags( QgsFeatureRequest::NoGeometry );
228  QgsFeature f;
229  layer->getFeatures( req ).nextFeature( f );
230  base = f.attribute( fieldIndex ).toString();
231  }
232 
233  // try variants like base_1, base_2, etc until a new value found
234  QStringList vals = layer->uniqueStringsMatching( fieldIndex, base );
235 
236  // might already be unique
237  if ( !base.isEmpty() && !vals.contains( base ) )
238  return base;
239 
240  for ( int i = 1; i < 10000; ++i )
241  {
242  QString testVal = base + '_' + QString::number( i );
243  if ( !vals.contains( testVal ) )
244  return testVal;
245  }
246 
247  // failed
248  return QVariant();
249  }
250 
251  default:
252  // todo other types - dates? times?
253  break;
254  }
255  }
256 
257  return QVariant();
258 }

260 bool QgsVectorLayerUtils::validateAttribute( const QgsVectorLayer *layer, const QgsFeature &feature, int attributeIndex, QStringList &errors,
261  QgsFieldConstraints::ConstraintStrength strength, QgsFieldConstraints::ConstraintOrigin origin )
262 {
263  if ( !layer )
264  return false;
265 
266  if ( attributeIndex < 0 || attributeIndex >= layer->fields().count() )
267  return false;
268 
269  QgsFields fields = layer->fields();
270  QgsField field = fields.at( attributeIndex );
271  QVariant value = feature.attribute( attributeIndex );
272  bool valid = true;
273  errors.clear();
274 
275  QgsFieldConstraints constraints = field.constraints();
276 
277  if ( constraints.constraints() & QgsFieldConstraints::ConstraintExpression && !constraints.constraintExpression().isEmpty()
278  && ( strength == QgsFieldConstraints::ConstraintStrengthNotSet || strength == constraints.constraintStrength( QgsFieldConstraints::ConstraintExpression ) )
279  && ( origin == QgsFieldConstraints::ConstraintOriginNotSet || origin == constraints.constraintOrigin( QgsFieldConstraints::ConstraintExpression ) ) )
280  {
281  QgsExpressionContext context = layer->createExpressionContext();
282  context.setFeature( feature );
283 
284  QgsExpression expr( constraints.constraintExpression() );
285 
286  valid = expr.evaluate( &context ).toBool();
287 
288  if ( expr.hasParserError() )
289  {
290  errors << QObject::tr( "parser error: %1" ).arg( expr.parserErrorString() );
291  }
292  else if ( expr.hasEvalError() )
293  {
294  errors << QObject::tr( "evaluation error: %1" ).arg( expr.evalErrorString() );
295  }
296  else if ( !valid )
297  {
298  errors << QObject::tr( "%1 check failed" ).arg( constraints.constraintDescription() );
299  }
300  }
301 
302  if ( constraints.constraints() & QgsFieldConstraints::ConstraintNotNull
303  && ( strength == QgsFieldConstraints::ConstraintStrengthNotSet || strength == constraints.constraintStrength( QgsFieldConstraints::ConstraintNotNull ) )
304  && ( origin == QgsFieldConstraints::ConstraintOriginNotSet || origin == constraints.constraintOrigin( QgsFieldConstraints::ConstraintNotNull ) ) )
305  {
306  bool exempt = false;
307  if ( fields.fieldOrigin( attributeIndex ) == QgsFields::OriginProvider
308  && constraints.constraintOrigin( QgsFieldConstraints::ConstraintNotNull ) == QgsFieldConstraints::ConstraintOriginProvider )
309  {
310  int providerIdx = fields.fieldOriginIndex( attributeIndex );
311  exempt = layer->dataProvider()->skipConstraintCheck( providerIdx, QgsFieldConstraints::ConstraintNotNull, value );
312  }
313 
314  if ( !exempt )
315  {
316  valid = valid && !value.isNull();
317 
318  if ( value.isNull() )
319  {
320  errors << QObject::tr( "value is NULL" );
321  }
322  }
323  }
324 
325  if ( constraints.constraints() & QgsFieldConstraints::ConstraintUnique
326  && ( strength == QgsFieldConstraints::ConstraintStrengthNotSet || strength == constraints.constraintStrength( QgsFieldConstraints::ConstraintUnique ) )
327  && ( origin == QgsFieldConstraints::ConstraintOriginNotSet || origin == constraints.constraintOrigin( QgsFieldConstraints::ConstraintUnique ) ) )
328  {
329  bool exempt = false;
330  if ( fields.fieldOrigin( attributeIndex ) == QgsFields::OriginProvider
331  && constraints.constraintOrigin( QgsFieldConstraints::ConstraintNotNull ) == QgsFieldConstraints::ConstraintOriginProvider )
332  {
333  int providerIdx = fields.fieldOriginIndex( attributeIndex );
334  exempt = layer->dataProvider()->skipConstraintCheck( providerIdx, QgsFieldConstraints::ConstraintUnique, value );
335  }
336 
337  if ( !exempt )
338  {
339  bool alreadyExists = QgsVectorLayerUtils::valueExists( layer, attributeIndex, value, QgsFeatureIds() << feature.id() );
340  valid = valid && !alreadyExists;
341 
342  if ( alreadyExists )
343  {
344  errors << QObject::tr( "value is not unique" );
345  }
346  }
347  }
348 
349  return valid;
350 }

352 QgsFeature QgsVectorLayerUtils::createFeature( const QgsVectorLayer *layer, const QgsGeometry &geometry,
353  const QgsAttributeMap &attributes, QgsExpressionContext *context )
354 {
355  if ( !layer )
356  {
357  return QgsFeature();
358  }
359 
360  QgsExpressionContext *evalContext = context;
361  std::unique_ptr< QgsExpressionContext > tempContext;
362  if ( !evalContext )
363  {
364  // no context passed, so we create a default one
365  tempContext.reset( new QgsExpressionContext( QgsExpressionContextUtils::globalProjectLayerScopes( layer ) ) );
366  evalContext = tempContext.get();
367  }
368 
369  QgsFields fields = layer->fields();
370 
371  QgsFeature newFeature( fields );
372  newFeature.setValid( true );
373  newFeature.setGeometry( geometry );
374 
375  // initialize attributes
376  newFeature.initAttributes( fields.count() );
377  for ( int idx = 0; idx < fields.count(); ++idx )
378  {
379  QVariant v;
380  bool checkUnique = true;
381 
382  // in order of priority:
383  // 1. passed attribute value and if field does not have a unique constraint like primary key
384  if ( attributes.contains( idx ) )
385  {
386  v = attributes.value( idx );
387  }
388 
389  // 2. client side default expression
390  // note - deliberately not using else if!
391  if ( ( !v.isValid() || ( fields.at( idx ).constraints().constraints() & QgsFieldConstraints::ConstraintUnique && QgsVectorLayerUtils::valueExists( layer, idx, v ) ) )
392  && layer->defaultValueDefinition( idx ).isValid() )
393  {
394  // client side default expression set - takes precedence over all. Why? Well, this is the only default
395  // which QGIS users have control over, so we assume that they're deliberately overriding any
396  // provider defaults for some good reason and we should respect that
397  v = layer->defaultValue( idx, newFeature, evalContext );
398  }
399 
400  // 3. provider side default value clause
401  // note - not an else if deliberately. Users may return null from a default value expression to fallback to provider defaults
402  if ( ( !v.isValid() || ( fields.at( idx ).constraints().constraints() & QgsFieldConstraints::ConstraintUnique && QgsVectorLayerUtils::valueExists( layer, idx, v ) ) )
403  && fields.fieldOrigin( idx ) == QgsFields::OriginProvider )
404  {
405  int providerIndex = fields.fieldOriginIndex( idx );
406  QString providerDefault = layer->dataProvider()->defaultValueClause( providerIndex );
407  if ( !providerDefault.isEmpty() )
408  {
409  v = providerDefault;
410  checkUnique = false;
411  }
412  }
413 
414  // 4. provider side default literal
415  // note - deliberately not using else if!
416  if ( ( !v.isValid() || ( checkUnique && fields.at( idx ).constraints().constraints() & QgsFieldConstraints::ConstraintUnique && QgsVectorLayerUtils::valueExists( layer, idx, v ) ) )
417  && fields.fieldOrigin( idx ) == QgsFields::OriginProvider )
418  {
419  int providerIndex = fields.fieldOriginIndex( idx );
420  v = layer->dataProvider()->defaultValue( providerIndex );
421  if ( v.isValid() )
422  {
423  //trust that the provider default has been sensibly set not to violate any constraints
424  checkUnique = false;
425  }
426  }
427 
428  // 5. passed attribute value
429  // note - deliberately not using else if!
430  if ( !v.isValid() && attributes.contains( idx ) )
431  {
432  v = attributes.value( idx );
433  }
434 
435  // last of all... check that unique constraints are respected
436  // we can't handle not null or expression constraints here, since there's no way to pick a sensible
437  // value if the constraint is violated
438  if ( checkUnique && fields.at( idx ).constraints().constraints() & QgsFieldConstraints::ConstraintUnique )
439  {
440  if ( QgsVectorLayerUtils::valueExists( layer, idx, v ) )
441  {
442  // unique constraint violated
443  QVariant uniqueValue = QgsVectorLayerUtils::createUniqueValue( layer, idx, v );
444  if ( uniqueValue.isValid() )
445  v = uniqueValue;
446  }
447  }
448 
449  newFeature.setAttribute( idx, v );
450  }
451 
452  return newFeature;
453 }

455 QgsFeature QgsVectorLayerUtils::duplicateFeature( QgsVectorLayer *layer, const QgsFeature &feature, QgsProject *project, int depth, QgsDuplicateFeatureContext &duplicateFeatureContext )
456 {
457  if ( !layer )
458  return QgsFeature();
459 
460  if ( !layer->isEditable() )
461  return QgsFeature();
462 
463  //get context from layer
464  QgsExpressionContext context = layer->createExpressionContext();
465  context.setFeature( feature );
466 
467  QgsFeature newFeature = createFeature( layer, feature.geometry(), feature.attributes().toMap(), &context );
468 
469  const QList<QgsRelation> relations = project->relationManager()->referencedRelations( layer );
470 
471  for ( const QgsRelation &relation : relations )
472  {
473  //check if composition (and not association)
474  if ( relation.strength() == QgsRelation::Composition && depth < 1 )
475  {
476  depth++;
477  //get features connected over this relation
478  QgsFeatureIterator relatedFeaturesIt = relation.getRelatedFeatures( feature );
479  QgsFeatureIds childFeatureIds;
480  QgsFeature childFeature;
481  while ( relatedFeaturesIt.nextFeature( childFeature ) )
482  {
483  //set childlayer editable
484  relation.referencingLayer()->startEditing();
485  //change the fk of the child to the id of the new parent
486  const auto pairs = relation.fieldPairs();
487  for ( const QgsRelation::FieldPair &fieldPair : pairs )
488  {
489  childFeature.setAttribute( fieldPair.first, newFeature.attribute( fieldPair.second ) );
490  }
491  //call the function for the child
492  childFeatureIds.insert( duplicateFeature( relation.referencingLayer(), childFeature, project, depth, duplicateFeatureContext ).id() );
493  }
494 
495  //store for feedback
496  duplicateFeatureContext.setDuplicatedFeatures( relation.referencingLayer(), childFeatureIds );
497  }
498  }
499 
500  layer->addFeature( newFeature );
501 
502  return newFeature;
503 }

529 void QgsVectorLayerUtils::matchAttributesToFields( QgsFeature &feature, const QgsFields &fields )
530 {
531  if ( !feature.fields().isEmpty() )
532  {
533  QgsAttributes attributes;
534  attributes.reserve( fields.size() );
535  // feature has a field mapping, so we can match attributes to field names
536  for ( const QgsField &field : fields )
537  {
538  int index = feature.fields().lookupField( field.name() );
539  attributes.append( index >= 0 ? feature.attribute( index ) : QVariant( field.type() ) );
540  }
541  feature.setAttributes( attributes );
542  }
543  else
544  {
545  // no field name mapping in feature, just use order
546  const int lengthDiff = feature.attributes().count() - fields.count();
547  if ( lengthDiff > 0 )
548  {
549  // truncate extra attributes
550  QgsAttributes attributes = feature.attributes().mid( 0, fields.count() );
551  feature.setAttributes( attributes );
552  }
553  else if ( lengthDiff < 0 )
554  {
555  // add missing null attributes
556  QgsAttributes attributes = feature.attributes();
557  attributes.reserve( fields.count() );
558  for ( int i = feature.attributes().count(); i < fields.count(); ++i )
559  {
560  attributes.append( QVariant( fields.at( i ).type() ) );
561  }
562  feature.setAttributes( attributes );
563  }

567 QgsFeatureList QgsVectorLayerUtils::makeFeatureCompatible( const QgsFeature &feature, const QgsVectorLayer *layer )
568 {
569  QgsWkbTypes::Type inputWkbType( layer->wkbType( ) );
570  QgsFeatureList resultFeatures;
571  QgsFeature newF( feature );
572  // Fix attributes
573  QgsVectorLayerUtils::matchAttributesToFields( newF, layer->fields( ) );
574  // Does geometry need transformations?
575  QgsWkbTypes::GeometryType newFGeomType( QgsWkbTypes::geometryType( newF.geometry().wkbType() ) );
576  bool newFHasGeom = newFGeomType !=
577  QgsWkbTypes::GeometryType::UnknownGeometry &&
578  newFGeomType != QgsWkbTypes::GeometryType::NullGeometry;
579  bool layerHasGeom = inputWkbType !=
580  QgsWkbTypes::Type::NoGeometry &&
581  inputWkbType != QgsWkbTypes::Type::Unknown;
582  // Drop geometry if layer is geometry-less
583  if ( newFHasGeom && ! layerHasGeom )
584  {
585  QgsFeature _f = QgsFeature( layer->fields() );
586  _f.setAttributes( newF.attributes() );
587  resultFeatures.append( _f );
588  }
589  else
590  {
591  // Geometry need fixing
592  if ( newFHasGeom && layerHasGeom && newF.geometry().wkbType() != inputWkbType )
593  {
594  // Single -> multi
595  if ( QgsWkbTypes::isMultiType( inputWkbType ) && ! newF.geometry().isMultipart( ) )
596  {
597  QgsGeometry newGeom( newF.geometry( ) );
598  newGeom.convertToMultiType();
599  newF.setGeometry( newGeom );
600  }
601  // Drop Z/M
602  if ( newF.geometry().constGet()->is3D() && ! QgsWkbTypes::hasZ( inputWkbType ) )
603  {
604  QgsGeometry newGeom( newF.geometry( ) );
605  newGeom.get()->dropZValue();
606  newF.setGeometry( newGeom );
607  }
608  if ( newF.geometry().constGet()->isMeasure() && ! QgsWkbTypes::hasM( inputWkbType ) )
609  {
610  QgsGeometry newGeom( newF.geometry( ) );
611  newGeom.get()->dropMValue();
612  newF.setGeometry( newGeom );
613  }
614  // Add Z/M back, set to 0
615  if ( ! newF.geometry().constGet()->is3D() && QgsWkbTypes::hasZ( inputWkbType ) )
616  {
617  QgsGeometry newGeom( newF.geometry( ) );
618  newGeom.get()->addZValue( 0.0 );
619  newF.setGeometry( newGeom );
620  }
621  if ( ! newF.geometry().constGet()->isMeasure() && QgsWkbTypes::hasM( inputWkbType ) )
622  {
623  QgsGeometry newGeom( newF.geometry( ) );
624  newGeom.get()->addMValue( 0.0 );
625  newF.setGeometry( newGeom );
626  }
627  // Multi -> single
628  if ( ! QgsWkbTypes::isMultiType( inputWkbType ) && newF.geometry().isMultipart( ) )
629  {
630  QgsGeometry newGeom( newF.geometry( ) );
631  const QgsGeometryCollection *parts( static_cast< const QgsGeometryCollection * >( newGeom.constGet() ) );
632  QgsAttributeMap attrMap;
633  for ( int j = 0; j < newF.fields().count(); j++ )
634  {
635  attrMap[j] = newF.attribute( j );
636  }
637  for ( int i = 0; i < parts->partCount( ); i++ )
638  {
639  QgsGeometry g( parts->geometryN( i )->clone() );
640  QgsFeature _f( createFeature( layer, g, attrMap ) );
641  resultFeatures.append( _f );
642  }
643  }
644  else
645  {
646  resultFeatures.append( newF );
647  }
648  }
649  else
650  {
651  resultFeatures.append( newF );
652  }
653  }
654  return resultFeatures;
655 }
