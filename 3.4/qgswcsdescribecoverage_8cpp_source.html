38  {

70  QDomDocument createDescribeCoverageDocument( QgsServerInterface *serverIface, const QgsProject *project, const QString &version,
71  const QgsServerRequest &request )
72  {
73  Q_UNUSED( version );
74 
75  QDomDocument doc;
76 
77  QgsServerRequest::Parameters parameters = request.parameters();
78 
79 #ifdef HAVE_SERVER_PYTHON_PLUGINS
80  QgsAccessControl *accessControl = serverIface->accessControls();
81 #endif
82 
83  //wcs:WCS_Capabilities element
84  QDomElement coveDescElement = doc.createElement( QStringLiteral( "CoverageDescription" )/*wcs:CoverageDescription*/ );
85  coveDescElement.setAttribute( QStringLiteral( "xmlns" ), WCS_NAMESPACE );
86  coveDescElement.setAttribute( QStringLiteral( "xmlns:xsi" ), QStringLiteral( "http://www.w3.org/2001/XMLSchema-instance" ) );
87  coveDescElement.setAttribute( QStringLiteral( "xsi:schemaLocation" ), WCS_NAMESPACE + " http://schemas.opengis.net/wcs/1.0.0/describeCoverage.xsd" );
88  coveDescElement.setAttribute( QStringLiteral( "xmlns:gml" ), GML_NAMESPACE );
89  coveDescElement.setAttribute( QStringLiteral( "xmlns:xlink" ), QStringLiteral( "http://www.w3.org/1999/xlink" ) );
90  coveDescElement.setAttribute( QStringLiteral( "version" ), QStringLiteral( "1.0.0" ) );
91  coveDescElement.setAttribute( QStringLiteral( "updateSequence" ), QStringLiteral( "0" ) );
92  doc.appendChild( coveDescElement );
93 
94  //defining coverage name
95  QString coveNames;
96  //read COVERAGE
97  QMap<QString, QString>::const_iterator cove_name_it = parameters.constFind( QStringLiteral( "COVERAGE" ) );
98  if ( cove_name_it != parameters.constEnd() )
99  {
100  coveNames = cove_name_it.value();
101  }
102  if ( coveNames.isEmpty() )
103  {
104  QMap<QString, QString>::const_iterator cove_name_it = parameters.constFind( QStringLiteral( "IDENTIFIER" ) );
105  if ( cove_name_it != parameters.constEnd() )
106  {
107  coveNames = cove_name_it.value();
108  }
109  }
110 
111  QStringList coveNameList;
112  if ( !coveNames.isEmpty() )
113  {
114  coveNameList = coveNames.split( ',' );
115  for ( int i = 0; i < coveNameList.size(); ++i )
116  {
117  coveNameList.replace( i, coveNameList.at( i ).trimmed() );
118  }
119  }
120 
121  QStringList wcsLayersId = QgsServerProjectUtils::wcsLayerIds( *project );
122  for ( int i = 0; i < wcsLayersId.size(); ++i )
123  {
124  QgsMapLayer *layer = project->mapLayer( wcsLayersId.at( i ) );
125  if ( !layer )
126  {
127  continue;
128  }
129  if ( layer->type() != QgsMapLayer::LayerType::RasterLayer )
130  {
131  continue;
132  }
133 #ifdef HAVE_SERVER_PYTHON_PLUGINS
134  if ( !accessControl->layerReadPermission( layer ) )
135  {
136  continue;
137  }
138 #endif
139  QString name = layer->name();
140  if ( !layer->shortName().isEmpty() )
141  name = layer->shortName();
142  name = name.replace( ' ', '_' );
143 
144  if ( coveNameList.size() == 0 || coveNameList.contains( name ) )
145  {
146  QgsRasterLayer *rLayer = qobject_cast<QgsRasterLayer *>( layer );
147  coveDescElement.appendChild( getCoverageOffering( doc, const_cast<QgsRasterLayer *>( rLayer ), project ) );
148  }
149  }
150  return doc;
151  }
