191  }
192  else
193  {
194  addTransactionResult( resp, respElem, QStringLiteral( "ERROR" ), locator, message );
195  }
196  }

686  QgsFeatureList featuresFromGML( QDomNodeList featureNodeList, QgsVectorDataProvider *provider )
687  {
688  // Store the inserted features
689  QgsFeatureList featList;
690 
691  // Get Layer Field Information
692  QgsFields fields = provider->fields();
693  QMap<QString, int> fieldMap = provider->fieldNameMap();
694  QMap<QString, int>::const_iterator fieldMapIt;
695 
696  for ( int i = 0; i < featureNodeList.count(); i++ )
697  {
698  QgsFeature feat( fields );
699 
700  QDomElement featureElem = featureNodeList.at( i ).toElement();
701  QDomNode currentAttributeChild = featureElem.firstChild();
702  bool conversionSuccess = true;
703

705  {
706  QDomElement currentAttributeElement = currentAttributeChild.toElement();
707  QString attrName = currentAttributeElement.localName();
708 
709  if ( attrName != QLatin1String( "boundedBy" ) )
710  {
711  if ( attrName != QLatin1String( "geometry" ) ) //a normal attribute
