127 void QgsDataSourceManagerDialog::addProviderDialog( QgsAbstractDataSourceWidget *dlg, const QString &providerKey, const QString &providerName, const QIcon &icon, const QString &toolTip )
128 {
129  mPageNames.append( providerKey );
130  ui->mOptionsStackedWidget->addWidget( dlg );
131  QListWidgetItem *layerItem = new QListWidgetItem( providerName, ui->mOptionsListWidget );
132  layerItem->setToolTip( toolTip.isEmpty() ? tr( "Add %1 layer" ).arg( providerName ) : toolTip );
133  layerItem->setIcon( icon );
134  // Set crs and extent from canvas
135  if ( mMapCanvas )
136  {
137  dlg->setMapCanvas( mMapCanvas );
138  }
139  connect( dlg, &QgsAbstractDataSourceWidget::rejected, this, &QgsDataSourceManagerDialog::reject );
140  connect( dlg, &QgsAbstractDataSourceWidget::accepted, this, &QgsDataSourceManagerDialog::accept );
141  makeConnections( dlg, providerKey );
142 }

144 void QgsDataSourceManagerDialog::makeConnections( QgsAbstractDataSourceWidget *dlg, const QString &providerKey )
145 {
146  // DB
147  connect( dlg, SIGNAL( addDatabaseLayers( QStringList const &, QString const & ) ),
148  this, SIGNAL( addDatabaseLayers( QStringList const &, QString const & ) ) );
149  connect( dlg, SIGNAL( progressMessage( QString ) ),
150  this, SIGNAL( showStatusMessage( QString ) ) );
151  // Vector
152  connect( dlg, &QgsAbstractDataSourceWidget::addVectorLayer, this, [ = ]( const QString & vectorLayerPath, const QString & baseName, const QString & specifiedProvider )
153  {
154  QString key = specifiedProvider.isEmpty() ? providerKey : specifiedProvider;
155  this->vectorLayerAdded( vectorLayerPath, baseName, key );
156  }
157  );
158  connect( dlg, &QgsAbstractDataSourceWidget::addVectorLayers,
159  this, &QgsDataSourceManagerDialog::vectorLayersAdded );
160  connect( dlg, SIGNAL( connectionsChanged() ), this, SIGNAL( connectionsChanged() ) );
161  // Raster
162  connect( dlg, SIGNAL( addRasterLayer( QString const &, QString const &, QString const & ) ),
163  this, SIGNAL( addRasterLayer( QString const &, QString const &, QString const & ) ) );
164  // Mesh
165  connect( dlg, &QgsAbstractDataSourceWidget::addMeshLayer, this, &QgsDataSourceManagerDialog::addMeshLayer );
166 
167  // Virtual
168  connect( dlg, SIGNAL( replaceVectorLayer( QString, QString, QString, QString ) ),
169  this, SIGNAL( replaceSelectedVectorLayer( QString, QString, QString, QString ) ) );
170  // Common
171  connect( dlg, SIGNAL( connectionsChanged() ), this, SIGNAL( connectionsChanged() ) );
172  connect( this, SIGNAL( providerDialogsRefreshRequested() ), dlg, SLOT( refresh() ) );
173 }
