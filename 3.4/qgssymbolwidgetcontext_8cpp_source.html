77 QList<QgsExpressionContextScope *> QgsSymbolWidgetContext::globalProjectAtlasMapLayerScopes( const QgsMapLayer *layer ) const
78 {
79  QList<QgsExpressionContextScope *> scopes;
80  scopes << QgsExpressionContextUtils::globalScope()
81  << QgsExpressionContextUtils::projectScope( QgsProject::instance() )
82  << QgsExpressionContextUtils::atlasScope( nullptr );
83  if ( mMapCanvas )
84  {
85  scopes << QgsExpressionContextUtils::mapSettingsScope( mMapCanvas->mapSettings() )
86  << mMapCanvas->defaultExpressionContextScope()
87  << new QgsExpressionContextScope( mMapCanvas->expressionContextScope() );
88  }
89  else
90  {
91  scopes << QgsExpressionContextUtils::mapSettingsScope( QgsMapSettings() );
92  }
93  if ( layer )
94  scopes << QgsExpressionContextUtils::layerScope( layer );
95  return scopes;
