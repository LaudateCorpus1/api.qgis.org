147 bool QgsMapToolCapture::tracingMouseMove( QgsMapMouseEvent *e )
148 {
149  if ( !e->isSnapped() )
150  return false;
151 
152  QgsPointXY pt0 = tracingStartPoint();
153  if ( pt0 == QgsPointXY() )
154  return false;
155 
156  QgsMapCanvasTracer *tracer = QgsMapCanvasTracer::tracerForCanvas( mCanvas );
157  if ( !tracer )
158  return false; // this should not happen!
159 
160  mTempRubberBand->reset( mCaptureMode == CapturePolygon ? QgsWkbTypes::PolygonGeometry : QgsWkbTypes::LineGeometry );
161 
162  QgsTracer::PathError err;
163  QVector<QgsPointXY> points = tracer->findShortestPath( pt0, e->mapPoint(), &err );
164  if ( points.isEmpty() )
165  {
166  tracer->reportError( err, false );
167  return false;
168  }
169 
170  if ( mCaptureMode == CapturePolygon )
171  mTempRubberBand->addPoint( *mRubberBand->getPoint( 0, 0 ), false );
172 
173  // if there is offset, we need to fix the rubber bands to make sure they are aligned correctly.
174  // There are two cases we need to sort out:
175  // 1. the last point of mRubberBand may need to be moved off the traced curve to respect the offset
176  // 2. extra first point of mTempRubberBand may be needed if there is gap between where mRubberBand ends and trace starts
177  if ( mRubberBand->numberOfVertices() != 0 )
178  {
179  QgsPointXY lastPoint = *mRubberBand->getPoint( 0, mRubberBand->numberOfVertices() - 1 );
180  if ( lastPoint == pt0 && points[0] != lastPoint )
181  {
182  // if rubber band had just one point, for some strange reason it contains the point twice
183  // we only want to move the last point if there are multiple points already
184  if ( mRubberBand->numberOfVertices() > 2 || ( mRubberBand->numberOfVertices() == 2 && *mRubberBand->getPoint( 0, 0 ) != *mRubberBand->getPoint( 0, 1 ) ) )
185  mRubberBand->movePoint( points[0] );
186  }
187  else
188  {
189  mTempRubberBand->addPoint( lastPoint, false );
190  }
191  }
192 
193  // update rubberband
194  for ( int i = 0; i < points.count(); ++i )
195  mTempRubberBand->addPoint( points.at( i ), i == points.count() - 1 );
196 
197  tracer->reportError( QgsTracer::ErrNone, false ); // clear messagebar if there was any error
198  return true;
199 }

202 bool QgsMapToolCapture::tracingAddVertex( const QgsPointXY &point )
203 {
204  QgsMapCanvasTracer *tracer = QgsMapCanvasTracer::tracerForCanvas( mCanvas );
205  if ( !tracer )
206  return false; // this should not happen!
207 
208  if ( mCaptureCurve.numPoints() == 0 )
209  {
210  if ( !tracer->init() )
211  {
212  tracer->reportError( QgsTracer::ErrTooManyFeatures, true );
213  return false;
214  }
215 
216  // only accept first point if it is snapped to the graph (to vertex or edge)
217  bool res = tracer->isPointSnapped( point );
218  if ( res )
219  {
220  QgsPoint layerPoint;
221  nextPoint( QgsPoint( point ), layerPoint ); // assuming the transform went fine earlier
222 
223  mRubberBand->addPoint( point );
224  mCaptureCurve.addVertex( layerPoint );
225  mSnappingMatches.append( QgsPointLocator::Match() );
226  }
227  return res;
228  }
229 
230  QgsPointXY pt0 = tracingStartPoint();
231  if ( pt0 == QgsPointXY() )
232  return false;
233 
234  QgsTracer::PathError err;
235  QVector<QgsPointXY> points = tracer->findShortestPath( pt0, point, &err );
236  if ( points.isEmpty() )
237  return false; // ignore the vertex - can't find path to the end point!
238 
239  if ( !mCaptureCurve.isEmpty() )
240  {
241  QgsPoint lp; // in layer coords
242  if ( nextPoint( QgsPoint( pt0 ), lp ) != 0 )
243  return false;
244  QgsPoint last;
245  QgsVertexId::VertexType type;
246  mCaptureCurve.pointAt( mCaptureCurve.numPoints() - 1, last, type );
247  if ( last == lp )
248  {
249  // remove the last point in the curve if it is the same as our first point
250  if ( mCaptureCurve.numPoints() != 2 )
251  mCaptureCurve.deleteVertex( QgsVertexId( 0, 0, mCaptureCurve.numPoints() - 1 ) );
252  else
253  {
254  // there is a strange behavior in deleteVertex() that with just two points
255  // the whole curve is cleared - so we need to do this little dance to work it around
256  QgsPoint first = mCaptureCurve.startPoint();
257  mCaptureCurve.clear();
258  mCaptureCurve.addVertex( first );
259  }
260  // for unknown reasons, rubber band has 2 points even if only one point has been added - handle that case
261  if ( mRubberBand->numberOfVertices() == 2 && *mRubberBand->getPoint( 0, 0 ) == *mRubberBand->getPoint( 0, 1 ) )
262  mRubberBand->removeLastPoint();
263  mRubberBand->removeLastPoint();
264  mSnappingMatches.removeLast();
265  }
266  }
267 
268  // transform points
269  QgsPointSequence layerPoints;
270  QgsPoint lp; // in layer coords
271  for ( int i = 0; i < points.count(); ++i )
272  {
273  if ( nextPoint( QgsPoint( points[i] ), lp ) != 0 )
274  return false;
275  layerPoints << lp;
276  }
277 
278  for ( int i = 0; i < points.count(); ++i )
279  {
280  if ( i == 0 && !mCaptureCurve.isEmpty() && mCaptureCurve.endPoint() == layerPoints[0] )
281  continue; // avoid duplicate of the first vertex
282  if ( i > 0 && points[i] == points[i - 1] )
283  continue; // avoid duplicate vertices if there are any
284  mRubberBand->addPoint( points[i], i == points.count() - 1 );
285  mCaptureCurve.addVertex( layerPoints[i] );
286  mSnappingMatches.append( QgsPointLocator::Match() );
287  }
288 
289  tracer->reportError( QgsTracer::ErrNone, true ); // clear messagebar if there was any error
290  return true;
291 }

342 int QgsMapToolCapture::nextPoint( const QgsPoint &mapPoint, QgsPoint &layerPoint )
343 {
344  QgsVectorLayer *vlayer = qobject_cast<QgsVectorLayer *>( mCanvas->currentLayer() );
345  if ( !vlayer )
346  {
347  QgsDebugMsg( QStringLiteral( "no vector layer" ) );
348  return 1;
349  }
350  try
351  {
352  QgsPointXY mapP( mapPoint.x(), mapPoint.y() ); //#spellok
353  layerPoint = QgsPoint( toLayerCoordinates( vlayer, mapP ) ); //transform snapped point back to layer crs //#spellok
354  if ( QgsWkbTypes::hasZ( vlayer->wkbType() ) )
355  layerPoint.addZValue( defaultZValue() );
356  if ( QgsWkbTypes::hasM( vlayer->wkbType() ) )
357  layerPoint.addMValue( 0.0 );
358  }
359  catch ( QgsCsException &cse )
360  {
361  Q_UNUSED( cse );
362  QgsDebugMsg( QStringLiteral( "transformation to layer coordinate failed" ) );
363  return 2;
364  }
365 
366  return 0;
367 }

375 int QgsMapToolCapture::fetchLayerPoint( const QgsPointLocator::Match &match, QgsPoint &layerPoint )
376 {
377  QgsVectorLayer *vlayer = qobject_cast<QgsVectorLayer *>( mCanvas->currentLayer() );
378  QgsVectorLayer *sourceLayer = match.layer();
379  if ( match.isValid() && match.hasVertex() && sourceLayer &&
380  ( sourceLayer->crs() == vlayer->crs() ) )
381  {
382  QgsFeature f;
383  QgsFeatureRequest request;
384  request.setFilterFid( match.featureId() );
385  bool fetched = match.layer()->getFeatures( request ).nextFeature( f );
386  if ( fetched )
387  {
388  QgsVertexId vId;
389  if ( !f.geometry().vertexIdFromVertexNr( match.vertexIndex(), vId ) )
390  return 2;
391 
392  layerPoint = f.geometry().constGet()->vertexAt( vId );
393 
394  // ZM support depends on the target layer
395  if ( !QgsWkbTypes::hasZ( vlayer->wkbType() ) )
396  {
397  layerPoint.dropZValue();
398  }
399 
400  if ( !QgsWkbTypes::hasM( vlayer->wkbType() ) )
401  {
402  layerPoint.dropMValue();
403  }
404 
405  return 0;
406  }
407  else
408  {
409  return 2;
410  }
411  }
412  else
413  {
414  return 1;
415  }
416 }

423 int QgsMapToolCapture::addVertex( const QgsPointXY &point, const QgsPointLocator::Match &match )
424 {
425  if ( mode() == CaptureNone )
426  {
427  QgsDebugMsg( QStringLiteral( "invalid capture mode" ) );
428  return 2;
429  }
430 
431  int res;
432  QgsPoint layerPoint;
433  res = fetchLayerPoint( match, layerPoint );
434  if ( res != 0 )
435  {
436  res = nextPoint( QgsPoint( point ), layerPoint );
437  if ( res != 0 )
438  {
439  return res;
440  }
441  }
442 
443  if ( !mRubberBand )
444  {
445  mRubberBand = createRubberBand( mCaptureMode == CapturePolygon ? QgsWkbTypes::PolygonGeometry : QgsWkbTypes::LineGeometry );
446  }
447 
448  if ( !mTempRubberBand )
449  {
450  mTempRubberBand = createRubberBand( mCaptureMode == CapturePolygon ? QgsWkbTypes::PolygonGeometry : QgsWkbTypes::LineGeometry, true );
451  }
452  else
453  {
454  mTempRubberBand->reset( mCaptureMode == CapturePolygon ? QgsWkbTypes::PolygonGeometry : QgsWkbTypes::LineGeometry );
455  }
456 
457  bool traceCreated = false;
458  if ( tracingEnabled() )
459  {
460  traceCreated = tracingAddVertex( point );
461  }
462 
463  // keep new tracing start point if we created a trace. This is useful when tracing with
464  // offset so that the user stays "snapped"
465  mTracingStartPoint = traceCreated ? point : QgsPointXY();
466 
467  if ( !traceCreated )
468  {
469  // ordinary digitizing
470  mRubberBand->addPoint( point );
471  mCaptureCurve.addVertex( layerPoint );
472  mSnappingMatches.append( match );
473  }
474 
475  if ( mCaptureMode == CaptureLine )
476  {
477  mTempRubberBand->addPoint( point );
478  }
479  else if ( mCaptureMode == CapturePolygon )
480  {
481  const QgsPointXY *firstPoint = mRubberBand->getPoint( 0, 0 );
482  mTempRubberBand->addPoint( *firstPoint );
483  mTempRubberBand->movePoint( point );
484  mTempRubberBand->addPoint( point );
485  }
486 
487  validateGeometry();
488 
489  return 0;
490 }

492 int QgsMapToolCapture::addCurve( QgsCurve *c )
493 {
494  if ( !c )
495  {
496  return 1;
497  }
498 
499  if ( !mRubberBand )
500  {
501  mRubberBand = createRubberBand( mCaptureMode == CapturePolygon ? QgsWkbTypes::PolygonGeometry : QgsWkbTypes::LineGeometry );
502  }
503 
504  QgsLineString *lineString = c->curveToLine();
505  QgsPointSequence linePoints;
506  lineString->points( linePoints );
507  delete lineString;
508  QgsPointSequence::const_iterator ptIt = linePoints.constBegin();
509  for ( ; ptIt != linePoints.constEnd(); ++ptIt )
510  {
511  mRubberBand->addPoint( QgsPointXY( ptIt->x(), ptIt->y() ) );
512  }
513 
514  if ( !mTempRubberBand )
515  {
516  mTempRubberBand = createRubberBand( mCaptureMode == CapturePolygon ? QgsWkbTypes::PolygonGeometry : QgsWkbTypes::LineGeometry, true );
517  }
518  else
519  {
520  mTempRubberBand->reset();
521  }
522  QgsPoint endPt = c->endPoint();
523  mTempRubberBand->addPoint( QgsPointXY( endPt.x(), endPt.y() ) ); //add last point of c
524 
525  //transform back to layer CRS in case map CRS and layer CRS are different
526  QgsVectorLayer *vlayer = qobject_cast<QgsVectorLayer *>( mCanvas->currentLayer() );
527  QgsCoordinateTransform ct = mCanvas->mapSettings().layerTransform( vlayer );
528  if ( ct.isValid() )
529  {
530  c->transform( ct, QgsCoordinateTransform::ReverseTransform );
531  }
532  mCaptureCurve.addCurve( c );
533  for ( int i = 0; i < c->length(); ++i )
534  mSnappingMatches.append( QgsPointLocator::Match() );
535 
536  return 0;
537 }

550 void QgsMapToolCapture::undo()
551 {
552  mTracingStartPoint = QgsPointXY();
553 
554  if ( mRubberBand )
555  {
556  int rubberBandSize = mRubberBand->numberOfVertices();
557  int tempRubberBandSize = mTempRubberBand->numberOfVertices();
558  int captureListSize = size();
559 
560  if ( rubberBandSize < 1 || captureListSize < 1 )
561  {
562  return;
563  }
564 
565  mRubberBand->removePoint( -1 );
566 
567  if ( rubberBandSize > 1 )
568  {
569  if ( tempRubberBandSize > 1 )
570  {
571  const QgsPointXY *point = mRubberBand->getPoint( 0, rubberBandSize - 2 );
572  mTempRubberBand->movePoint( tempRubberBandSize - 2, *point );
573  }
574  }
575  else
576  {
577  mTempRubberBand->reset( mCaptureMode == CapturePolygon ? QgsWkbTypes::PolygonGeometry : QgsWkbTypes::LineGeometry );
578  }
579 
580  QgsVertexId vertexToRemove;
581  vertexToRemove.part = 0;
582  vertexToRemove.ring = 0;
583  vertexToRemove.vertex = size() - 1;
584  mCaptureCurve.deleteVertex( vertexToRemove );
585  mSnappingMatches.removeAt( vertexToRemove.vertex );
586 
587  validateGeometry();
588  }

669 void QgsMapToolCapture::validateGeometry()
670 {
671  QgsSettings settings;
672  if ( settings.value( QStringLiteral( "qgis/digitizing/validate_geometries" ), 1 ).toInt() == 0 )
673  return;
674 
675  if ( mValidator )
676  {
677  mValidator->deleteLater();
678  mValidator = nullptr;
679  }
680 
681  mGeomErrors.clear();
682  while ( !mGeomErrorMarkers.isEmpty() )
683  {
684  delete mGeomErrorMarkers.takeFirst();
685  }
686 
687  QgsGeometry geom;
688 
689  switch ( mCaptureMode )
690  {
691  case CaptureNone:
692  case CapturePoint:
693  return;
694  case CaptureLine:
695  if ( size() < 2 )
696  return;
697  geom = QgsGeometry( mCaptureCurve.curveToLine() );
698  break;
699  case CapturePolygon:
700  if ( size() < 3 )
701  return;
702  QgsLineString *exteriorRing = mCaptureCurve.curveToLine();
703  exteriorRing->close();
704  QgsPolygon *polygon = new QgsPolygon();
705  polygon->setExteriorRing( exteriorRing );
706  geom = QgsGeometry( polygon );
707  break;
708  }
709 
710  if ( geom.isNull() )
711  return;
712 
713  QgsGeometry::ValidationMethod method = QgsGeometry::ValidatorQgisInternal;
714  if ( settings.value( QStringLiteral( "qgis/digitizing/validate_geometries" ), 1 ).toInt() == 2 )
715  method = QgsGeometry::ValidatorGeos;

766 QgsPoint QgsMapToolCapture::mapPoint( const QgsPointXY &point ) const
767 {
768  QgsPoint newPoint( QgsWkbTypes::Point, point.x(), point.y() );
769 
770  // get current layer
771  QgsVectorLayer *vlayer = qobject_cast<QgsVectorLayer *>( mCanvas->currentLayer() );
772  if ( !vlayer )
773  {
774  return newPoint;
775  }
776 
777  // convert to the corresponding type for a full ZM support
778  const QgsWkbTypes::Type type = vlayer->wkbType();
779  if ( QgsWkbTypes::hasZ( type ) && !QgsWkbTypes::hasM( type ) )
780  {
781  newPoint.convertTo( QgsWkbTypes::PointZ );
782  }
783  else if ( !QgsWkbTypes::hasZ( type ) && QgsWkbTypes::hasM( type ) )
784  {
785  newPoint.convertTo( QgsWkbTypes::PointM );
786  }
787  else if ( QgsWkbTypes::hasZ( type ) && QgsWkbTypes::hasM( type ) )
788  {
789  newPoint.convertTo( QgsWkbTypes::PointZM );
790  }
791 
792  // set z value if necessary
793  if ( QgsWkbTypes::hasZ( newPoint.wkbType() ) )
794  {
795  newPoint.setZ( defaultZValue() );
796  }
797 
798  return newPoint;
799 }

801 QgsPoint QgsMapToolCapture::mapPoint( const QgsMapMouseEvent &e ) const
802 {
803  QgsPoint newPoint = mapPoint( e.mapPoint() );
804 
805  // set z value from snapped point if necessary
806  if ( QgsWkbTypes::hasZ( newPoint.wkbType() ) )
807  {
808  // if snapped, z dimension is taken from the corresponding snapped
809  // point.
810  if ( e.isSnapped() )
811  {
812  const QgsPointLocator::Match match = e.mapPointMatch();
813  const QgsWkbTypes::Type snappedType = match.layer()->wkbType();
814 
815  if ( QgsWkbTypes::hasZ( snappedType ) )
816  {
817  const QgsFeature ft = match.layer()->getFeature( match.featureId() );
818  newPoint.setZ( ft.geometry().vertexAt( match.vertexIndex() ).z() );
819  }
820  }
821  }
822 
823  return newPoint;
824 }
