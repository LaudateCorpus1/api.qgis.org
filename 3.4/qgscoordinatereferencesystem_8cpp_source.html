102 {
103  QList<long> results;
104  // check both standard & user defined projection databases
105  QStringList dbs = QStringList() << QgsApplication::srsDatabaseFilePath() << QgsApplication::qgisUserDatabaseFilePath();
106 
107  Q_FOREACH ( const QString &db, dbs )
108  {
109  QFileInfo myInfo( db );
110  if ( !myInfo.exists() )
111  {
112  QgsDebugMsg( "failed : " + db + " does not exist!" );
113  continue;
114  }
115 
116  sqlite3_database_unique_ptr database;
117  sqlite3_statement_unique_ptr statement;
118 
119  //check the db is available
120  int result = openDatabase( db, database );
121  if ( result != SQLITE_OK )
122  {
123  QgsDebugMsg( "failed : " + db + " could not be opened!" );
124  continue;
125  }
126 
127  QString sql = QStringLiteral( "select srs_id from tbl_srs" );
128  int rc;
129  statement = database.prepare( sql, rc );
130  while ( true )
131  {
132  // this one is an infinitive loop, intended to fetch any row
133  int ret = statement.step();
134 
135  if ( ret == SQLITE_DONE )
136  {
137  // there are no more rows to fetch - we can stop looping
138  break;
139  }
140 
141  if ( ret == SQLITE_ROW )
142  {
143  results.append( statement.columnAsInt64( 0 ) );
144  }
145  else
146  {
147  QgsMessageLog::logMessage( QObject::tr( "SQLite error: %2\nSQL: %1" ).arg( sql, sqlite3_errmsg( database.get() ) ), QObject::tr( "SpatiaLite" ) );
148  break;
149  }
150  }
151  }
152  std::sort( results.begin(), results.end() );
153  return results;

214 bool QgsCoordinateReferenceSystem::createFromString( const QString &definition )
215 {
216  sCrsStringLock.lockForRead();
217  QHash< QString, QgsCoordinateReferenceSystem >::const_iterator crsIt = sStringCache.constFind( definition );
218  if ( crsIt != sStringCache.constEnd() )
219  {
220  // found a match in the cache
221  *this = crsIt.value();
222  sCrsStringLock.unlock();
223  return true;
224  }
225  sCrsStringLock.unlock();
226 
227  bool result = false;
228  QRegularExpression reCrsId( "^(epsg|postgis|internal|user)\\:(\\d+)$", QRegularExpression::CaseInsensitiveOption );
229  QRegularExpressionMatch match = reCrsId.match( definition );
230  if ( match.capturedStart() == 0 )
231  {
232  QString authName = match.captured( 1 ).toLower();
233  CrsType type = InternalCrsId;
234  if ( authName == QLatin1String( "epsg" ) )
235  type = EpsgCrsId;
236  if ( authName == QLatin1String( "postgis" ) )
237  type = PostgisCrsId;
238  long id = match.captured( 2 ).toLong();
239  result = createFromId( id, type );
240  }
241  else
242  {
243  QRegularExpression reCrsStr( "^(?:(wkt|proj4)\\:)?(.+)$", QRegularExpression::CaseInsensitiveOption );
244  match = reCrsStr.match( definition );
245  if ( match.capturedStart() == 0 )
246  {
247  if ( match.captured( 1 ).compare( QLatin1String( "proj4" ), Qt::CaseInsensitive ) == 0 )
248  {
249  result = createFromProj4( match.captured( 2 ) );
250  //TODO: createFromProj4 used to save to the user database any new CRS
251  // this behavior was changed in order to separate creation and saving.
252  // Not sure if it necessary to save it here, should be checked by someone
253  // familiar with the code (should also give a more descriptive name to the generated CRS)
254  if ( srsid() == 0 )
255  {
256  QString myName = QStringLiteral( " * %1 (%2)" )
257  .arg( QObject::tr( "Generated CRS", "A CRS automatically generated from layer info get this prefix for description" ),
258  toProj4() );
259  saveAsUserCrs( myName );
260  }
261  }
262  else
263  {
264  result = createFromWkt( match.captured( 2 ) );
265  }
266  }
267  }
268 
269  sCrsStringLock.lockForWrite();
270  sStringCache.insert( definition, *this );
271  sCrsStringLock.unlock();
272  return result;

322 {
323  sOgcLock.lockForRead();
324  QHash< QString, QgsCoordinateReferenceSystem >::const_iterator crsIt = sOgcCache.constFind( crs );
325  if ( crsIt != sOgcCache.constEnd() )
326  {
327  // found a match in the cache
328  *this = crsIt.value();
329  sOgcLock.unlock();
330  return true;
331  }
332  sOgcLock.unlock();
333 
334  QString wmsCrs = crs;
335 
336  QRegExp re( "urn:ogc:def:crs:([^:]+).+([^:]+)", Qt::CaseInsensitive );
337  if ( re.exactMatch( wmsCrs ) )
338  {
339  wmsCrs = re.cap( 1 ) + ':' + re.cap( 2 );
340  }
341  else
342  {
343  re.setPattern( QStringLiteral( "(user|custom|qgis):(\\d+)" ) );
344  if ( re.exactMatch( wmsCrs ) && createFromSrsId( re.cap( 2 ).toInt() ) )
345  {
346  sOgcLock.lockForWrite();
347  sOgcCache.insert( crs, *this );
348  sOgcLock.unlock();
349  return true;
350  }
351  }
352 
353  if ( loadFromDatabase( QgsApplication::srsDatabaseFilePath(), QStringLiteral( "lower(auth_name||':'||auth_id)" ), wmsCrs.toLower() ) )
354  {
355  sOgcLock.lockForWrite();
356  sOgcCache.insert( crs, *this );
357  sOgcLock.unlock();
358  return true;
359  }
360 
361  // NAD27
362  if ( wmsCrs.compare( QLatin1String( "CRS:27" ), Qt::CaseInsensitive ) == 0 ||
363  wmsCrs.compare( QLatin1String( "OGC:CRS27" ), Qt::CaseInsensitive ) == 0 )
364  {
365  // TODO: verify same axis orientation
366  return createFromOgcWmsCrs( QStringLiteral( "EPSG:4267" ) );
367  }
368 
369  // NAD83

372  {

379  wmsCrs.compare( QLatin1String( "OGC:CRS84" ), Qt::CaseInsensitive ) == 0 )
380  {
381  createFromOgcWmsCrs( QStringLiteral( "EPSG:4326" ) );
382 
383  d.detach();
384  d->mAxisInverted = false;
385  d->mAxisInvertedDirty = false;
386 
387  sOgcLock.lockForWrite();
388  sOgcCache.insert( crs, *this );
389  sOgcLock.unlock();
390 
391  return d->mIsValid;

676 {

681  if ( crsIt != sProj4Cache.constEnd() )

691  // Examples:

693  // +ellps=clrk80 +towgs84=-255,-15,71,0,0,0,0 +units=m +no_defs
694  //
695  // +proj=lcc +lat_1=46.8 +lat_0=46.8 +lon_0=2.337229166666664 +k_0=0.99987742
696  // +x_0=600000 +y_0=2200000 +a=6378249.2 +b=6356515.000000472 +units=m +no_defs
697  //

699  d->mIsValid = false;
700  d->mWkt.clear();

703  int myStart = myProjRegExp.indexIn( myProj4String );
704  if ( myStart == -1 )
705  {
706  sProj4CacheLock.lockForWrite();
707  sProj4Cache.insert( proj4String, *this );
708  sProj4CacheLock.unlock();
709 
710  return d->mIsValid;
711  }
712 
713  d->mProjectionAcronym = myProjRegExp.cap( 1 );
714 
715  QRegExp myEllipseRegExp( "\\+ellps=(\\S+)" );
716  myStart = myEllipseRegExp.indexIn( myProj4String );
717  if ( myStart == -1 )
718  {
719  d->mEllipsoidAcronym.clear();
720  }
721  else
722  {
723  d->mEllipsoidAcronym = myEllipseRegExp.cap( 1 );
724  }
725 
726  QRegExp myAxisRegExp( "\\+a=(\\S+)" );
727  myStart = myAxisRegExp.indexIn( myProj4String );

730  QgsCoordinateReferenceSystem::RecordMap myRecord;

733  * We try to match the proj string to and srsid using the following logic:
734  * - perform a whole text search on proj4 string (if not null)
735  */
736  myRecord = getRecord( "select * from tbl_srs where parameters=" + QgsSqliteUtils::quotedString( myProj4String ) + " order by deprecated" );
737  if ( myRecord.empty() )
738  {
739  // Ticket #722 - aaronr
740  // Check if we can swap the lat_1 and lat_2 params (if they exist) to see if we match...
741  // First we check for lat_1 and lat_2
742  QRegExp myLat1RegExp( "\\+lat_1=\\S+" );
743  QRegExp myLat2RegExp( "\\+lat_2=\\S+" );
744  int myStart1 = 0;
745  int myLength1 = 0;
746  int myStart2 = 0;
747  int myLength2 = 0;
748  QString lat1Str;
749  QString lat2Str;
750  myStart1 = myLat1RegExp.indexIn( myProj4String, myStart1 );
751  myStart2 = myLat2RegExp.indexIn( myProj4String, myStart2 );
752  if ( myStart1 != -1 && myStart2 != -1 )
753  {
754  myLength1 = myLat1RegExp.matchedLength();
755  myLength2 = myLat2RegExp.matchedLength();
756  lat1Str = myProj4String.mid( myStart1 + LAT_PREFIX_LEN, myLength1 - LAT_PREFIX_LEN );
757  lat2Str = myProj4String.mid( myStart2 + LAT_PREFIX_LEN, myLength2 - LAT_PREFIX_LEN );
758  }
759  // If we found the lat_1 and lat_2 we need to swap and check to see if we can find it...
760  if ( !lat1Str.isEmpty() && !lat2Str.isEmpty() )
761  {
762  // Make our new string to check...
763  QString proj4StringModified = myProj4String;
764  // First just swap in the lat_2 value for lat_1 value
765  proj4StringModified.replace( myStart1 + LAT_PREFIX_LEN, myLength1 - LAT_PREFIX_LEN, lat2Str );
766  // Now we have to find the lat_2 location again since it has potentially moved...
767  myStart2 = 0;
768  myStart2 = myLat2RegExp.indexIn( proj4String, myStart2 );
769  proj4StringModified.replace( myStart2 + LAT_PREFIX_LEN, myLength2 - LAT_PREFIX_LEN, lat1Str );
770  QgsDebugMsgLevel( QStringLiteral( "trying proj4string match with swapped lat_1,lat_2" ), 4 );
771  myRecord = getRecord( "select * from tbl_srs where parameters=" + QgsSqliteUtils::quotedString( proj4StringModified.trimmed() ) + " order by deprecated" );
772  }
773  }
774 
775  if ( myRecord.empty() )
776  {
777  // match all parameters individually:
778  // - order of parameters doesn't matter
779  // - found definition may have more parameters (like +towgs84 in GDAL)
780  // - retry without datum, if no match is found (looks like +datum<>WGS84 was dropped in GDAL)
781 
782  QString sql = QStringLiteral( "SELECT * FROM tbl_srs WHERE " );
783  QString delim;
784  QString datum;

787  // also with parameters containing spaces (e.g. +nadgrids)
788  // make sure result is trimmed (#5598)
789  QStringList myParams;

791  {
792  QString arg = QStringLiteral( "' '||parameters||' ' LIKE %1" ).arg( QgsSqliteUtils::quotedString( QStringLiteral( "% %1 %" ).arg( param.trimmed() ) ) );
793  if ( param.startsWith( QLatin1String( "+datum=" ) ) )
794  {
795  datum = arg;
796  }
797  else
798  {
799  sql += delim + arg;
800  delim = QStringLiteral( " AND " );
801  myParams << param.trimmed();
802  }
803  }
804 
805  if ( !datum.isEmpty() )
806  {
807  myRecord = getRecord( sql + delim + datum + " order by deprecated" );
808  }
809 
810  if ( myRecord.empty() )
811  {
812  // datum might have disappeared in definition - retry without it
813  myRecord = getRecord( sql + " order by deprecated" );
814  }
815 
816  if ( !myRecord.empty() )
817  {
818  // Bugfix 8487 : test param lists are equal, except for +datum
819  QStringList foundParams;
820  Q_FOREACH ( const QString &param, myRecord["parameters"].split( QRegExp( "\\s+(?=\\+)" ), QString::SkipEmptyParts ) )
821  {
822  if ( !param.startsWith( QLatin1String( "+datum=" ) ) )
823  foundParams << param.trimmed();
824  }
825 
826  myParams.sort();
827  foundParams.sort();
828 
829  if ( myParams != foundParams )
830  {
831  myRecord.clear();
832  }
833  }
834  }
835 
836  if ( !myRecord.empty() )
837  {
838  mySrsId = myRecord[QStringLiteral( "srs_id" )].toLong();
839  if ( mySrsId > 0 )
840  {
841  createFromSrsId( mySrsId );
842  }
843  }
844  else
845  {
846  // Last ditch attempt to piece together what we know of the projection to find a match...
847  setProj4String( myProj4String );
848  mySrsId = findMatchingProj();
849  if ( mySrsId > 0 )
850  {
851  createFromSrsId( mySrsId );
852  }
853  else
854  {
855  d->mIsValid = false;
856  }
857  }
858 
859  // if we failed to look up the projection in database, don't worry. we can still use it :)
860  if ( !d->mIsValid )
861  {
862  QgsDebugMsgLevel( QStringLiteral( "Projection is not found in databases." ), 4 );
863  //setProj4String will set mIsValidFlag to true if there is no issue
864  setProj4String( myProj4String );
865  }
866 
867  sProj4CacheLock.lockForWrite();
868  sProj4Cache.insert( proj4String, *this );
869  sProj4CacheLock.unlock();
870 
871  return d->mIsValid;
