96

99  << QStringLiteral( "current_feature" )

101 
102  QgsExpressionBuilderDialog dlg( vl, mFilterExpression->toPlainText(), this, QStringLiteral( "generic" ), context );
103  dlg.setWindowTitle( tr( "Edit Filter Expression" ) );
104 
105  if ( dlg.exec() == QDialog::Accepted )
