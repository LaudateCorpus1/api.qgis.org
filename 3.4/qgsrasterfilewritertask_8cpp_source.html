41 bool QgsRasterFileWriterTask::run()
42 {
43  if ( !mPipe )
44  return false;
45 
46  connect( mFeedback.get(), &QgsRasterBlockFeedback::progressChanged, this, &QgsRasterFileWriterTask::setProgress );
47 
48  mError = mWriter.writeRaster( mPipe.get(), mColumns, mRows, mExtent, mCrs, mFeedback.get() );
49 
50  return mError == QgsRasterFileWriter::NoError;
