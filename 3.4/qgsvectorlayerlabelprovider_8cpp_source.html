102 bool QgsVectorLayerLabelProvider::prepare( const QgsRenderContext &context, QSet<QString> &attributeNames )
103 {
104  QgsPalLayerSettings &lyr = mSettings;
105  const QgsMapSettings &mapSettings = mEngine->mapSettings();
106 
107  QgsDebugMsgLevel( "PREPARE LAYER " + mLayerId, 4 );
108 
109  if ( lyr.drawLabels )
110  {
111  if ( lyr.fieldName.isEmpty() )
112  {
113  return false;
114  }
115 
116  if ( lyr.isExpression )
117  {
118  QgsExpression exp( lyr.fieldName );
119  if ( exp.hasEvalError() )
120  {
121  QgsDebugMsgLevel( "Prepare error:" + exp.evalErrorString(), 4 );
122  return false;
123  }
124  }
125  else
126  {
127  // If we aren't an expression, we check to see if we can find the column.
128  if ( mFields.lookupField( lyr.fieldName ) == -1 )
129  {
130  return false;
131  }
132  }
133  }
134 
135  lyr.mCurFields = mFields;
136 
137  if ( lyr.drawLabels || lyr.obstacle )
138  {
139  if ( lyr.drawLabels )
140  {
141  // add field indices for label's text, from expression or field
142  if ( lyr.isExpression )
143  {
144  // prepare expression for use in QgsPalLayerSettings::registerFeature()
145  QgsExpression *exp = lyr.getLabelExpression();
146  exp->prepare( &context.expressionContext() );
147  if ( exp->hasEvalError() )
148  {
149  QgsDebugMsgLevel( "Prepare error:" + exp->evalErrorString(), 4 );
150  }
151  Q_FOREACH ( const QString &name, exp->referencedColumns() )
152  {
153  QgsDebugMsgLevel( "REFERENCED COLUMN = " + name, 4 );
154  attributeNames.insert( name );
155  }
156  }
157  else
158  {
159  attributeNames.insert( lyr.fieldName );
160  }
161  }
162 
163  lyr.dataDefinedProperties().prepare( context.expressionContext() );
164  // add field indices of data defined expression or field
165  attributeNames.unite( lyr.dataDefinedProperties().referencedFields( context.expressionContext() ) );
166  }
167 
168  // NOW INITIALIZE QgsPalLayerSettings
169 
170  // TODO: ideally these (non-configuration) members should get out of QgsPalLayerSettings to here
171  // (together with registerFeature() & related methods) and QgsPalLayerSettings just stores config
172 
173  // save the pal layer to our layer context (with some additional info)
174  lyr.fieldIndex = mFields.lookupField( lyr.fieldName );
175 
176  lyr.xform = &mapSettings.mapToPixel();
177  lyr.ct = QgsCoordinateTransform();
178  if ( context.coordinateTransform().isValid() )
179  // this is context for layer rendering
180  lyr.ct = context.coordinateTransform();
181  else
182  {
183  // otherwise fall back to creating our own CT
184  lyr.ct = QgsCoordinateTransform( mCrs, mapSettings.destinationCrs(), mapSettings.transformContext() );
185  }
186  lyr.ptZero = lyr.xform->toMapCoordinates( 0, 0 );
187  lyr.ptOne = lyr.xform->toMapCoordinates( 1, 0 );
188 
189  // rect for clipping
190  lyr.extentGeom = QgsGeometry::fromRect( mapSettings.visibleExtent() );
191  if ( !qgsDoubleNear( mapSettings.rotation(), 0.0 ) )
192  {
193  //PAL features are prerotated, so extent also needs to be unrotated
194  lyr.extentGeom.rotate( -mapSettings.rotation(), mapSettings.visibleExtent().center() );
195  }
196 
197  lyr.mFeatsSendingToPal = 0;
198 
199  return true;
200 }

204 QList<QgsLabelFeature *> QgsVectorLayerLabelProvider::labelFeatures( QgsRenderContext &ctx )
205 {
206  if ( !mSource )
207  {
208  // we have created the provider with "own feature loop" == false
209  // so it is assumed that prepare() has been already called followed by registerFeature() calls
210  return mLabels;
211  }
212 
213  QSet<QString> attrNames;
214  if ( !prepare( ctx, attrNames ) )
215  return QList<QgsLabelFeature *>();
216 
217  if ( mRenderer )
218  mRenderer->startRender( ctx, mFields );
219 
220  QgsRectangle layerExtent = ctx.extent();
221  if ( mSettings.ct.isValid() && !mSettings.ct.isShortCircuited() )
222  layerExtent = mSettings.ct.transformBoundingBox( ctx.extent(), QgsCoordinateTransform::ReverseTransform );
223 
224  QgsFeatureRequest request;
225  request.setFilterRect( layerExtent );
226  request.setSubsetOfAttributes( attrNames, mFields );
227  QgsFeatureIterator fit = mSource->getFeatures( request );
228 
229  QgsExpressionContextScope *symbolScope = new QgsExpressionContextScope();
230  ctx.expressionContext().appendScope( symbolScope );
231  QgsFeature fet;
232  while ( fit.nextFeature( fet ) )
233  {
234  QgsGeometry obstacleGeometry;
235  if ( mRenderer )
236  {
237  QgsSymbolList symbols = mRenderer->originalSymbolsForFeature( fet, ctx );
238  if ( !symbols.isEmpty() && fet.geometry().type() == QgsWkbTypes::PointGeometry )
239  {
240  //point feature, use symbol bounds as obstacle
241  obstacleGeometry = QgsVectorLayerLabelProvider::getPointObstacleGeometry( fet, ctx, symbols );
242  }
243  if ( !symbols.isEmpty() )
244  {
245  symbolScope = QgsExpressionContextUtils::updateSymbolScope( symbols.at( 0 ), symbolScope );
246  }
247  }
248  ctx.expressionContext().setFeature( fet );
249  registerFeature( fet, ctx, obstacleGeometry );
250  }
251 
252  if ( ctx.expressionContext().lastScope() == symbolScope )
253  delete ctx.expressionContext().popScope();
254 
255  if ( mRenderer )
256  mRenderer->stopRender( ctx );
257 
258  return mLabels;
259 }

269 QgsGeometry QgsVectorLayerLabelProvider::getPointObstacleGeometry( QgsFeature &fet, QgsRenderContext &context, const QgsSymbolList &symbols )
270 {
271  if ( !fet.hasGeometry() || fet.geometry().type() != QgsWkbTypes::PointGeometry )
272  return QgsGeometry();
273 
274  bool isMultiPoint = fet.geometry().constGet()->nCoordinates() > 1;
275  std::unique_ptr< QgsAbstractGeometry > obstacleGeom;
276  if ( isMultiPoint )
277  obstacleGeom = qgis::make_unique< QgsMultiPolygon >();
278 
279  // for each point
280  for ( int i = 0; i < fet.geometry().constGet()->nCoordinates(); ++i )
281  {
282  QRectF bounds;
283  QgsPoint p = fet.geometry().constGet()->vertexAt( QgsVertexId( i, 0, 0 ) );
284  double x = p.x();
285  double y = p.y();
286  double z = 0; // dummy variable for coordinate transforms
287 
288  //transform point to pixels
289  if ( context.coordinateTransform().isValid() )
290  {
291  try
292  {
293  context.coordinateTransform().transformInPlace( x, y, z );
294  }
295  catch ( QgsCsException & )
296  {
297  return QgsGeometry();
298  }
299  }
300  context.mapToPixel().transformInPlace( x, y );
301 
302  QPointF pt( x, y );
303  Q_FOREACH ( QgsSymbol *symbol, symbols )
304  {
305  if ( symbol->type() == QgsSymbol::Marker )
306  {
307  if ( bounds.isValid() )
308  bounds = bounds.united( static_cast< QgsMarkerSymbol * >( symbol )->bounds( pt, context, fet ) );
309  else
310  bounds = static_cast< QgsMarkerSymbol * >( symbol )->bounds( pt, context, fet );
311  }
312  }
313 
314  //convert bounds to a geometry
315  QVector< double > bX;
316  bX << bounds.left() << bounds.right() << bounds.right() << bounds.left();
317  QVector< double > bY;
318  bY << bounds.top() << bounds.top() << bounds.bottom() << bounds.bottom();
319  std::unique_ptr< QgsLineString > boundLineString = qgis::make_unique< QgsLineString >( bX, bY );
320 
321  //then transform back to map units
322  //TODO - remove when labeling is refactored to use screen units
323  for ( int i = 0; i < boundLineString->numPoints(); ++i )
324  {
325  QgsPointXY point = context.mapToPixel().toMapCoordinates( static_cast<int>( boundLineString->xAt( i ) ),
326  static_cast<int>( boundLineString->yAt( i ) ) );
327  boundLineString->setXAt( i, point.x() );
328  boundLineString->setYAt( i, point.y() );
329  }
330  if ( context.coordinateTransform().isValid() )
331  {
332  try
333  {
334  boundLineString->transform( context.coordinateTransform(), QgsCoordinateTransform::ReverseTransform );
335  }
336  catch ( QgsCsException & )
337  {
338  return QgsGeometry();
339  }
340  }
341  boundLineString->close();
342 
343  if ( context.coordinateTransform().isValid() )
344  {
345  // coordinate transforms may have resulted in nan coordinates - if so, strip these out
346  boundLineString->filterVertices( []( const QgsPoint & point )->bool
347  {
348  return std::isfinite( point.x() ) && std::isfinite( point.y() );
349  } );
350  if ( !boundLineString->isRing() )
351  return QgsGeometry();
352  }
353 
354  std::unique_ptr< QgsPolygon > obstaclePolygon = qgis::make_unique< QgsPolygon >();
355  obstaclePolygon->setExteriorRing( boundLineString.release() );
356 
357  if ( isMultiPoint )
358  {
359  static_cast<QgsMultiPolygon *>( obstacleGeom.get() )->addGeometry( obstaclePolygon.release() );
360  }
361  else
362  {
363  obstacleGeom = std::move( obstaclePolygon );
364  }
365  }
366 
367  return QgsGeometry( std::move( obstacleGeom ) );
368 }

370 void QgsVectorLayerLabelProvider::drawLabel( QgsRenderContext &context, pal::LabelPosition *label ) const
371 {
372  if ( !mSettings.drawLabels )
373  return;
374 
375  QgsTextLabelFeature *lf = dynamic_cast<QgsTextLabelFeature *>( label->getFeaturePart()->feature() );
376 
377  // Copy to temp, editable layer settings
378  // these settings will be changed by any data defined values, then used for rendering label components
379  // settings may be adjusted during rendering of components
380  QgsPalLayerSettings tmpLyr( mSettings );
381 
382  // apply any previously applied data defined settings for the label
383  const QMap< QgsPalLayerSettings::Property, QVariant > &ddValues = lf->dataDefinedValues();
384 
385  //font
386  QFont dFont = lf->definedFont();
387  QgsDebugMsgLevel( QStringLiteral( "PAL font tmpLyr: %1, Style: %2" ).arg( tmpLyr.format().font().toString(), tmpLyr.format().font().styleName() ), 4 );
388  QgsDebugMsgLevel( QStringLiteral( "PAL font definedFont: %1, Style: %2" ).arg( dFont.toString(), dFont.styleName() ), 4 );
389 
390  QgsTextFormat format = tmpLyr.format();
391  format.setFont( dFont );
392 
393  // size has already been calculated and stored in the defined font - this calculated size
394  // is in pixels
395  format.setSize( dFont.pixelSize() );
396  format.setSizeUnit( QgsUnitTypes::RenderPixels );
397  tmpLyr.setFormat( format );
398 
399  if ( tmpLyr.multilineAlign == QgsPalLayerSettings::MultiFollowPlacement )
400  {
401  //calculate font alignment based on label quadrant
402  switch ( label->getQuadrant() )
403  {
404  case LabelPosition::QuadrantAboveLeft:
405  case LabelPosition::QuadrantLeft:
406  case LabelPosition::QuadrantBelowLeft:
407  tmpLyr.multilineAlign = QgsPalLayerSettings::MultiRight;
408  break;
409  case LabelPosition::QuadrantAbove:
410  case LabelPosition::QuadrantOver:
411  case LabelPosition::QuadrantBelow:
412  tmpLyr.multilineAlign = QgsPalLayerSettings::MultiCenter;
413  break;
414  case LabelPosition::QuadrantAboveRight:
415  case LabelPosition::QuadrantRight:
416  case LabelPosition::QuadrantBelowRight:
417  tmpLyr.multilineAlign = QgsPalLayerSettings::MultiLeft;
418  break;
419  }
420  }
421 
422  // update tmpLyr with any data defined text style values
423  QgsPalLabeling::dataDefinedTextStyle( tmpLyr, ddValues );
424 
425  // update tmpLyr with any data defined text buffer values
426  QgsPalLabeling::dataDefinedTextBuffer( tmpLyr, ddValues );
427 
428  // update tmpLyr with any data defined text formatting values
429  QgsPalLabeling::dataDefinedTextFormatting( tmpLyr, ddValues );
430 
431  // update tmpLyr with any data defined shape background values
432  QgsPalLabeling::dataDefinedShapeBackground( tmpLyr, ddValues );
433 
434  // update tmpLyr with any data defined drop shadow values
435  QgsPalLabeling::dataDefinedDropShadow( tmpLyr, ddValues );
436 
437  // Render the components of a label in reverse order
438  // (backgrounds -> text)
439 
440  if ( tmpLyr.format().shadow().enabled() && tmpLyr.format().shadow().shadowPlacement() == QgsTextShadowSettings::ShadowLowest )
441  {
442  QgsTextFormat format = tmpLyr.format();
443 
444  if ( tmpLyr.format().background().enabled() )
445  {
446  format.shadow().setShadowPlacement( QgsTextShadowSettings::ShadowShape );
447  }
448  else if ( tmpLyr.format().buffer().enabled() )
449  {
450  format.shadow().setShadowPlacement( QgsTextShadowSettings::ShadowBuffer );
451  }
452  else
453  {
454  format.shadow().setShadowPlacement( QgsTextShadowSettings::ShadowText );
455  }
456 
457  tmpLyr.setFormat( format );
458  }
459 
460  if ( tmpLyr.format().background().enabled() )
461  {
462  drawLabelPrivate( label, context, tmpLyr, QgsTextRenderer::Background );
463  }
464 
465  if ( tmpLyr.format().buffer().enabled() )
466  {
467  drawLabelPrivate( label, context, tmpLyr, QgsTextRenderer::Buffer );
468  }
469 
470  drawLabelPrivate( label, context, tmpLyr, QgsTextRenderer::Text );
471 
472  // add to the results
473  QString labeltext = label->getFeaturePart()->feature()->labelText();
474  mEngine->results()->mLabelSearchTree->insertLabel( label, label->getFeaturePart()->featureId(), mLayerId, labeltext, dFont, false, lf->hasFixedPosition(), mProviderId );
475 }

478 void QgsVectorLayerLabelProvider::drawLabelPrivate( pal::LabelPosition *label, QgsRenderContext &context, QgsPalLayerSettings &tmpLyr, QgsTextRenderer::TextPart drawType, double dpiRatio ) const
479 {
480  // NOTE: this is repeatedly called for multi-part labels
481  QPainter *painter = context.painter();
482 
483  // features are pre-rotated but not scaled/translated,
484  // so we only disable rotation here. Ideally, they'd be
485  // also pre-scaled/translated, as suggested here:
486  // https://issues.qgis.org/issues/11856
487  QgsMapToPixel xform = context.mapToPixel();
488  xform.setMapRotation( 0, 0, 0 );
489 
490  QPointF outPt = xform.transform( label->getX(), label->getY() ).toQPointF();
491 
492  if ( mEngine->engineSettings().testFlag( QgsLabelingEngineSettings::DrawLabelRectOnly ) ) // TODO: this should get directly to labeling engine
493  {
494  //debugging rect
495  if ( drawType != QgsTextRenderer::Text )
496  return;
497 
498  QgsPointXY outPt2 = xform.transform( label->getX() + label->getWidth(), label->getY() + label->getHeight() );
499  QRectF rect( 0, 0, outPt2.x() - outPt.x(), outPt2.y() - outPt.y() );
500  painter->save();
501  painter->setRenderHint( QPainter::Antialiasing, false );
502  painter->translate( QPointF( outPt.x(), outPt.y() ) );
503  painter->rotate( -label->getAlpha() * 180 / M_PI );
504 
505  if ( label->conflictsWithObstacle() )
506  {
507  painter->setBrush( QColor( 255, 0, 0, 100 ) );
508  painter->setPen( QColor( 255, 0, 0, 150 ) );
509  }
510  else
511  {
512  painter->setBrush( QColor( 0, 255, 0, 100 ) );
513  painter->setPen( QColor( 0, 255, 0, 150 ) );
514  }
515 
516  painter->drawRect( rect );
517  painter->restore();
518 
519  if ( label->getNextPart() )
520  drawLabelPrivate( label->getNextPart(), context, tmpLyr, drawType, dpiRatio );
521 
522  return;
523  }
524 
525  QgsTextRenderer::Component component;
526  component.dpiRatio = dpiRatio;
527  component.origin = outPt;
528  component.rotation = label->getAlpha();
529 
530 
531 
532  if ( drawType == QgsTextRenderer::Background )
533  {
534  // get rotated label's center point
535  QPointF centerPt( outPt );
536  QgsPointXY outPt2 = xform.transform( label->getX() + label->getWidth() / 2,
537  label->getY() + label->getHeight() / 2 );
538 
539  double xc = outPt2.x() - outPt.x();
540  double yc = outPt2.y() - outPt.y();
541 
542  double angle = -component.rotation;
543  double xd = xc * std::cos( angle ) - yc * std::sin( angle );
544  double yd = xc * std::sin( angle ) + yc * std::cos( angle );
545 
546  centerPt.setX( centerPt.x() + xd );
547  centerPt.setY( centerPt.y() + yd );
548 
549  component.center = centerPt;
550 
551  // convert label size to render units
552  double labelWidthPx = context.convertToPainterUnits( label->getWidth(), QgsUnitTypes::RenderMapUnits, QgsMapUnitScale() );
553  double labelHeightPx = context.convertToPainterUnits( label->getHeight(), QgsUnitTypes::RenderMapUnits, QgsMapUnitScale() );
554 
555  component.size = QSizeF( labelWidthPx, labelHeightPx );
556 
557  QgsTextRenderer::drawBackground( context, component, tmpLyr.format(), QStringList(), QgsTextRenderer::Label );
558  }
559 
560  else if ( drawType == QgsTextRenderer::Buffer
561  || drawType == QgsTextRenderer::Text )
562  {
563 
564  // TODO: optimize access :)
565  QgsTextLabelFeature *lf = static_cast<QgsTextLabelFeature *>( label->getFeaturePart()->feature() );
566  QString txt = lf->text( label->getPartId() );
567  QFontMetricsF *labelfm = lf->labelFontMetrics();
568 
569  //add the direction symbol if needed
570  if ( !txt.isEmpty() && tmpLyr.placement == QgsPalLayerSettings::Line &&
571  tmpLyr.addDirectionSymbol )
572  {
573  bool prependSymb = false;
574  QString symb = tmpLyr.rightDirectionSymbol;
575 
576  if ( label->getReversed() )
577  {
578  prependSymb = true;
579  symb = tmpLyr.leftDirectionSymbol;
580  }
581 
582  if ( tmpLyr.reverseDirectionSymbol )
583  {
584  if ( symb == tmpLyr.rightDirectionSymbol )
585  {
586  prependSymb = true;
587  symb = tmpLyr.leftDirectionSymbol;
588  }
589  else
590  {
591  prependSymb = false;
592  symb = tmpLyr.rightDirectionSymbol;
593  }
594  }
595 
596  if ( tmpLyr.placeDirectionSymbol == QgsPalLayerSettings::SymbolAbove )
597  {
598  prependSymb = true;
599  symb = symb + QStringLiteral( "\n" );
600  }
601  else if ( tmpLyr.placeDirectionSymbol == QgsPalLayerSettings::SymbolBelow )
602  {
603  prependSymb = false;
604  symb = QStringLiteral( "\n" ) + symb;
605  }
606 
607  if ( prependSymb )
608  {
609  txt.prepend( symb );
610  }
611  else
612  {
613  txt.append( symb );
614  }
615  }
616 
617  //QgsDebugMsgLevel( "drawLabel " + txt, 4 );
618  QStringList multiLineList = QgsPalLabeling::splitToLines( txt, tmpLyr.wrapChar, tmpLyr.autoWrapLength, tmpLyr.useMaxLineLengthForAutoWrap );
619 
620  QgsTextRenderer::HAlignment hAlign = QgsTextRenderer::AlignLeft;
621  if ( tmpLyr.multilineAlign == QgsPalLayerSettings::MultiCenter )
622  hAlign = QgsTextRenderer::AlignCenter;
623  else if ( tmpLyr.multilineAlign == QgsPalLayerSettings::MultiRight )
624  hAlign = QgsTextRenderer::AlignRight;
625 
626  QgsTextRenderer::Component component;
627  component.origin = outPt;
628  component.rotation = label->getAlpha();
629 
630  QgsTextRenderer::drawTextInternal( drawType, context, tmpLyr.format(), component, multiLineList, labelfm,
631  hAlign, QgsTextRenderer::Label );
632 
633  }
634 
635  // NOTE: this used to be within above multi-line loop block, at end. (a mistake since 2010? [LS])
636  if ( label->getNextPart() )
637  drawLabelPrivate( label->getNextPart(), context, tmpLyr, drawType, dpiRatio );
638 }
