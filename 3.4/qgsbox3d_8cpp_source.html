93 bool QgsBox3d::intersects( const QgsBox3d &other ) const
94 {
95  if ( !mBounds2d.intersects( other.mBounds2d ) )
96  return false;
97 
98  double z1 = ( mZmin > other.mZmin ? mZmin : other.mZmin );
99  double z2 = ( mZmax < other.mZmax ? mZmax : other.mZmax );
100  return z1 <= z2;
101 }
