61 QgsRasterCalculator::Result QgsRasterCalculator::processCalculation( QgsFeedback *feedback )
62 {
63  mLastError.clear();
64 
65  //prepare search string / tree
66  std::unique_ptr< QgsRasterCalcNode > calcNode( QgsRasterCalcNode::parseRasterCalcString( mFormulaString, mLastError ) );
67  if ( !calcNode )
68  {
69  //error
70  return ParserError;
71  }
72 
73  // Check input layers and bands
74  for ( const auto &entry : qgis::as_const( mRasterEntries ) )
75  {
76  if ( !entry.raster ) // no raster layer in entry
77  {
78  mLastError = QObject::tr( "No raster layer for entry %1" ).arg( entry.ref );
79  return InputLayerError;
80  }
81  if ( entry.bandNumber <= 0 || entry.bandNumber > entry.raster->bandCount() )
82  {
83  mLastError = QObject::tr( "Band number %1 is not valid for entry %2" ).arg( entry.bandNumber ).arg( entry.ref );
84  return BandError;
85  }
86  }
87 
88  //open output dataset for writing
89  GDALDriverH outputDriver = openOutputDriver();
90  if ( !outputDriver )
91  {
92  mLastError = QObject::tr( "Could not obtain driver for %1" ).arg( mOutputFormat );
93  return CreateOutputError;
94  }
95 
96  gdal::dataset_unique_ptr outputDataset( openOutputFile( outputDriver ) );
97  if ( !outputDataset )
98  {
99  mLastError = QObject::tr( "Could not create output %1" ).arg( mOutputFile );
100  return CreateOutputError;
101  }
102 
103  GDALSetProjection( outputDataset.get(), mOutputCrs.toWkt().toLocal8Bit().data() );
104  GDALRasterBandH outputRasterBand = GDALGetRasterBand( outputDataset.get(), 1 );
105 
106  float outputNodataValue = -FLT_MAX;
107  GDALSetRasterNoDataValue( outputRasterBand, outputNodataValue );
108 
109  // Check if we need to read the raster as a whole (which is memory inefficient
110  // and not interruptable by the user) by checking if any raster matrix nodes are
111  // in the expression
112  bool requiresMatrix = ! calcNode->findNodes( QgsRasterCalcNode::Type::tMatrix ).isEmpty();
113 
114  // Take the fast route (process one line at a time) if we can
115  if ( ! requiresMatrix )
116  {
117  // Map of raster names -> blocks
118  std::map<QString, std::unique_ptr<QgsRasterBlock>> inputBlocks;
119  std::map<QString, QgsRasterCalculatorEntry> uniqueRasterEntries;
120  for ( const auto &r : calcNode->findNodes( QgsRasterCalcNode::Type::tRasterRef ) )
121  {
122  QString layerRef( r->toString().remove( 0, 1 ) );
123  layerRef.chop( 1 );
124  if ( ! inputBlocks.count( layerRef ) )
125  {
126  for ( const auto &ref : mRasterEntries )
127  {
128  if ( ref.ref == layerRef )
129  {
130  uniqueRasterEntries[layerRef] = ref;
131  inputBlocks[layerRef ] = qgis::make_unique<QgsRasterBlock>();
132  }
133  }
134  }
135  }
136 
137  //read / write line by line
138  QMap<QString, QgsRasterBlock * > _rasterData;
139  // Cast to float
140  std::vector<float> castedResult;
141  castedResult.reserve( static_cast<size_t>( mNumOutputColumns ) );
142  auto rowHeight = mOutputRectangle.height() / mNumOutputRows;
143  for ( size_t row = 0; row < static_cast<size_t>( mNumOutputRows ); ++row )
144  {
145  if ( feedback )
146  {
147  feedback->setProgress( 100.0 * static_cast< double >( row ) / mNumOutputRows );
148  }
149 
150  if ( feedback && feedback->isCanceled() )
151  {
152  break;
153  }
154 
155  // Calculates the rect for a single row read
156  QgsRectangle rect( mOutputRectangle );
157  rect.setYMaximum( rect.yMaximum() - rowHeight * row );
158  rect.setYMinimum( rect.yMaximum() - rowHeight );
159 
160  // Read rows into input blocks
161  for ( auto &layerRef : inputBlocks )
162  {
163  QgsRasterCalculatorEntry ref = uniqueRasterEntries[layerRef.first];
164  if ( uniqueRasterEntries[layerRef.first].raster->crs() != mOutputCrs )
165  {
166  QgsRasterProjector proj;
167  proj.setCrs( ref.raster->crs(), mOutputCrs );
168  proj.setInput( ref.raster->dataProvider() );
169  proj.setPrecision( QgsRasterProjector::Exact );
170  layerRef.second.reset( proj.block( ref.bandNumber, rect, mNumOutputColumns, 1 ) );
171  }
172  else
173  {
174  inputBlocks[layerRef.first].reset( ref.raster->dataProvider()->block( ref.bandNumber, rect, mNumOutputColumns, 1 ) );
175  }
176  }
177 
178  QgsRasterMatrix resultMatrix;
179  resultMatrix.setNodataValue( outputNodataValue );
180 
181  _rasterData.clear();
182  for ( const auto &layerRef : inputBlocks )
183  {
184  _rasterData.insert( layerRef.first, inputBlocks[layerRef.first].get() );
185  }
186 
187  if ( calcNode->calculate( _rasterData, resultMatrix, 0 ) )
188  {
189  // write scanline to the dataset
190  for ( size_t i = 0; i < static_cast<size_t>( mNumOutputColumns ); i++ )
191  {
192  castedResult[i] = static_cast<float>( resultMatrix.data()[i] );
193  }
194  if ( GDALRasterIO( outputRasterBand, GF_Write, 0, row, mNumOutputColumns, 1, castedResult.data(), mNumOutputColumns, 1, GDT_Float32, 0, 0 ) != CE_None )
195  {
196  QgsDebugMsg( QStringLiteral( "RasterIO error!" ) );
197  }
198  }
199  }
200 
201  if ( feedback )
202  {
203  feedback->setProgress( 100.0 );
204  }
205  }
206  else // Original code (memory inefficient route)
207  {
208  QMap< QString, QgsRasterBlock * > inputBlocks;
209  QVector<QgsRasterCalculatorEntry>::const_iterator it = mRasterEntries.constBegin();
210  for ( ; it != mRasterEntries.constEnd(); ++it )
211  {
212 
213  std::unique_ptr< QgsRasterBlock > block;
214  // if crs transform needed
215  if ( it->raster->crs() != mOutputCrs )
216  {
217  QgsRasterProjector proj;
218  proj.setCrs( it->raster->crs(), mOutputCrs );
219  proj.setInput( it->raster->dataProvider() );
220  proj.setPrecision( QgsRasterProjector::Exact );
221 
222  QgsRasterBlockFeedback *rasterBlockFeedback = new QgsRasterBlockFeedback();
223  QObject::connect( feedback, &QgsFeedback::canceled, rasterBlockFeedback, &QgsRasterBlockFeedback::cancel );
224  block.reset( proj.block( it->bandNumber, mOutputRectangle, mNumOutputColumns, mNumOutputRows, rasterBlockFeedback ) );
225  if ( rasterBlockFeedback->isCanceled() )
226  {
227  qDeleteAll( inputBlocks );
228  return Canceled;
229  }
230  }
231  else
232  {
233  block.reset( it->raster->dataProvider()->block( it->bandNumber, mOutputRectangle, mNumOutputColumns, mNumOutputRows ) );
234  }
235  if ( block->isEmpty() )
236  {
237  mLastError = QObject::tr( "Could not allocate required memory for %1" ).arg( it->ref );
238  qDeleteAll( inputBlocks );
239  return MemoryError;
240  }
241  inputBlocks.insert( it->ref, block.release() );
242  }
243 
244  QgsRasterMatrix resultMatrix;
245  resultMatrix.setNodataValue( outputNodataValue );
246 
247  //read / write line by line
248  for ( int i = 0; i < mNumOutputRows; ++i )
249  {
250  if ( feedback )
251  {
252  feedback->setProgress( 100.0 * static_cast< double >( i ) / mNumOutputRows );
253  }
254 
255  if ( feedback && feedback->isCanceled() )
256  {
257  break;
258  }
259 
260  if ( calcNode->calculate( inputBlocks, resultMatrix, i ) )
261  {
262  bool resultIsNumber = resultMatrix.isNumber();
263  float *calcData = new float[mNumOutputColumns];
264 
265  for ( int j = 0; j < mNumOutputColumns; ++j )
266  {
267  calcData[j] = ( float )( resultIsNumber ? resultMatrix.number() : resultMatrix.data()[j] );
268  }
269 
270  //write scanline to the dataset
271  if ( GDALRasterIO( outputRasterBand, GF_Write, 0, i, mNumOutputColumns, 1, calcData, mNumOutputColumns, 1, GDT_Float32, 0, 0 ) != CE_None )
272  {
273  QgsDebugMsg( QStringLiteral( "RasterIO error!" ) );
274  }
275 
276  delete[] calcData;
277  }
278 
279  }
280 
281  if ( feedback )
282  {
283  feedback->setProgress( 100.0 );
284  }
285 
286  //close datasets and release memory
287  calcNode.reset();
288  qDeleteAll( inputBlocks );
289  inputBlocks.clear();
290 
291  }
292 
293  if ( feedback && feedback->isCanceled() )
294  {
295  //delete the dataset without closing (because it is faster)
296  gdal::fast_delete_and_close( outputDataset, outputDriver, mOutputFile );
297  return Canceled;
298  }
299  return Success;
300 }

302 GDALDriverH QgsRasterCalculator::openOutputDriver()
303 {
304  //open driver
305  GDALDriverH outputDriver = GDALGetDriverByName( mOutputFormat.toLocal8Bit().data() );
306 
307  if ( !outputDriver )
308  {
309  return outputDriver; //return nullptr, driver does not exist
310  }
311 
312  if ( !QgsGdalUtils::supportsRasterCreate( outputDriver ) )
313  {
314  return nullptr; //driver exist, but it does not support the create operation
315  }
316 
317  return outputDriver;
318 }

320 gdal::dataset_unique_ptr QgsRasterCalculator::openOutputFile( GDALDriverH outputDriver )
321 {
322  //open output file
323  char **papszOptions = nullptr;
324  gdal::dataset_unique_ptr outputDataset( GDALCreate( outputDriver, mOutputFile.toUtf8().constData(), mNumOutputColumns, mNumOutputRows, 1, GDT_Float32, papszOptions ) );
325  if ( !outputDataset )
326  {
327  return nullptr;
328  }
329 
330  //assign georef information

332  outputGeoTransform( geotransform );
333  GDALSetGeoTransform( outputDataset.get(), geotransform );
334 
335  return outputDataset;

338 void QgsRasterCalculator::outputGeoTransform( double *transform ) const
339 {
340  transform[0] = mOutputRectangle.xMinimum();
341  transform[1] = mOutputRectangle.width() / mNumOutputColumns;
342  transform[2] = 0;
343  transform[3] = mOutputRectangle.yMaximum();
344  transform[4] = 0;
345  transform[5] = -mOutputRectangle.height() / mNumOutputRows;
