65 void QgsExpressionFieldBuffer::readXml( const QDomNode &layerNode )
66 {
67  mExpressions.clear();
68 
69  const QDomElement expressionFieldsElem = layerNode.firstChildElement( QStringLiteral( "expressionfields" ) );
70 
71  if ( !expressionFieldsElem.isNull() )
72  {
73  QDomNodeList fields = expressionFieldsElem.elementsByTagName( QStringLiteral( "field" ) );
74 
75  for ( int i = 0; i < fields.size(); ++i )
76  {
77  QDomElement field = fields.at( i ).toElement();
78  QString exp = field.attribute( QStringLiteral( "expression" ) );
79  QString name = field.attribute( QStringLiteral( "name" ) );
80  QString comment = field.attribute( QStringLiteral( "comment" ) );
81  int precision = field.attribute( QStringLiteral( "precision" ) ).toInt();
82  int length = field.attribute( QStringLiteral( "length" ) ).toInt();
83  QVariant::Type type = static_cast< QVariant::Type >( field.attribute( QStringLiteral( "type" ) ).toInt() );
84  QVariant::Type subType = static_cast< QVariant::Type >( field.attribute( QStringLiteral( "subType" ), QStringLiteral( "0" ) ).toInt() );
85  QString typeName = field.attribute( QStringLiteral( "typeName" ) );
86 
87  mExpressions.append( ExpressionField( exp, QgsField( name, type, typeName, length, precision, comment, subType ) ) );
88  }
89  }
