420 QWidget *QgsProcessingNumericWidgetWrapper::createWidget()
421 {
422  const QgsProcessingParameterNumber *numberDef = static_cast< const QgsProcessingParameterNumber * >( parameterDefinition() );
423  const QVariantMap metadata = numberDef->metadata();
424  const int decimals = metadata.value( QStringLiteral( "widget_wrapper" ) ).toMap().value( QStringLiteral( "decimals" ), 6 ).toInt();
425  switch ( type() )
426  {
427  case QgsProcessingGui::Standard:
428  case QgsProcessingGui::Modeler:
429  case QgsProcessingGui::Batch:
430  {
431  // lots of duplicate code here -- but there's no common interface between QSpinBox/QDoubleSpinBox which would allow us to avoid this
432  QAbstractSpinBox *spinBox = nullptr;
433  switch ( numberDef->dataType() )
434  {
435  case QgsProcessingParameterNumber::Double:
436  mDoubleSpinBox = new QgsDoubleSpinBox();
437  mDoubleSpinBox->setExpressionsEnabled( true );
438  mDoubleSpinBox->setDecimals( decimals );
439 
440  // guess reasonable step value for double spin boxes
441  if ( !qgsDoubleNear( numberDef->maximum(), std::numeric_limits<double>::max() ) &&
442  !qgsDoubleNear( numberDef->minimum(), std::numeric_limits<double>::lowest() + 1 ) )
443  {
444  double singleStep = calculateStep( numberDef->minimum(), numberDef->maximum() );
445  singleStep = std::max( singleStep, std::pow( 10, -decimals ) );
446  mDoubleSpinBox->setSingleStep( singleStep );
447  }
448 
449  spinBox = mDoubleSpinBox;
450  break;
451 
452  case QgsProcessingParameterNumber::Integer:
453  mSpinBox = new QgsSpinBox();
454  mSpinBox->setExpressionsEnabled( true );
455  spinBox = mSpinBox;
456  break;
457  }
458  spinBox->setToolTip( parameterDefinition()->toolTip() );
459 
460  double max = 999999999;
461  if ( !qgsDoubleNear( numberDef->maximum(), std::numeric_limits<double>::max() ) )
462  {
463  max = numberDef->maximum();
464  }
465  double min = -999999999;
466  if ( !qgsDoubleNear( numberDef->minimum(), std::numeric_limits<double>::lowest() ) )
467  {
468  min = numberDef->minimum();
469  }
470  if ( mDoubleSpinBox )
471  {
472  mDoubleSpinBox->setMinimum( min );
473  mDoubleSpinBox->setMaximum( max );
474  }
475  else
476  {
477  mSpinBox->setMinimum( static_cast< int >( min ) );
478  mSpinBox->setMaximum( static_cast< int >( max ) );
479  }
480 
481  if ( numberDef->flags() & QgsProcessingParameterDefinition::FlagOptional )
482  {
483  mAllowingNull = true;
484  if ( mDoubleSpinBox )
485  {
486  mDoubleSpinBox->setShowClearButton( true );
487  const double min = mDoubleSpinBox->minimum() - 1;
488  mDoubleSpinBox->setMinimum( min );
489  mDoubleSpinBox->setValue( min );
490  }
491  else
492  {
493  mSpinBox->setShowClearButton( true );
494  const int min = mSpinBox->minimum() - 1;
495  mSpinBox->setMinimum( min );
496  mSpinBox->setValue( min );
497  }
498  spinBox->setSpecialValueText( tr( "Not set" ) );
499  }
500  else
501  {
502  if ( numberDef->defaultValue().isValid() )
503  {
504  // if default value for parameter, we clear to that
505  bool ok = false;
506  if ( mDoubleSpinBox )
507  {
508  double defaultVal = numberDef->defaultValue().toDouble( &ok );
509  if ( ok )
510  mDoubleSpinBox->setClearValue( defaultVal );
511  }
512  else
513  {
514  int intVal = numberDef->defaultValue().toInt( &ok );
515  if ( ok )
516  mSpinBox->setClearValue( intVal );
517  }
518  }
519  else if ( !qgsDoubleNear( numberDef->minimum(), std::numeric_limits<double>::lowest() ) )
520  {
521  // otherwise we clear to the minimum, if it's set
522  if ( mDoubleSpinBox )
523  mDoubleSpinBox->setClearValue( numberDef->minimum() );
524  else
525  mSpinBox->setClearValue( static_cast< int >( numberDef->minimum() ) );
526  }
527  else
528  {
529  // last resort, we clear to 0
530  if ( mDoubleSpinBox )
531  {
532  mDoubleSpinBox->setValue( 0 );
533  mDoubleSpinBox->setClearValue( 0 );
534  }
535  else
536  {
537  mSpinBox->setValue( 0 );
538  mSpinBox->setClearValue( 0 );
539  }
540  }
541  }
542 
543  if ( mDoubleSpinBox )
544  connect( mDoubleSpinBox, qgis::overload<double>::of( &QgsDoubleSpinBox::valueChanged ), this, [ = ] { emit widgetValueHasChanged( this ); } );
545  else if ( mSpinBox )
546  connect( mSpinBox, qgis::overload<int>::of( &QgsSpinBox::valueChanged ), this, [ = ] { emit widgetValueHasChanged( this ); } );
547 
548  return spinBox;
549  };
550  }
551  return nullptr;
552 }

618 {

620  if ( valueRange <= 1.0 )
621  {

624  return qgsRound( step, -std::floor( std::log( step ) ) );
625  }
626  else
627  {
628  return 1.0;
