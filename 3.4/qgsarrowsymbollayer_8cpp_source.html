188 inline qreal euclidian_distance( QPointF po, QPointF pd )
189 {
190  return std::sqrt( ( po.x() - pd.x() ) * ( po.x() - pd.x() ) + ( po.y() - pd.y() ) * ( po.y() - pd.y() ) );
191 }

193 QPolygonF straightArrow( QPointF po, QPointF pd,
194  qreal startWidth, qreal width,
195  qreal headWidth, qreal headHeight,
196  QgsArrowSymbolLayer::HeadType headType, QgsArrowSymbolLayer::ArrowType arrowType,
197  qreal offset )
198 {
199  QPolygonF polygon; // implicitly shared
200  // vector length
201  qreal length = euclidian_distance( po, pd );
202 
203  // shift points if there is not enough room for the head(s)
204  if ( ( headType == QgsArrowSymbolLayer::HeadSingle ) && ( length < headWidth ) )
205  {
206  po = pd - ( pd - po ) / length * headWidth;
207  length = headWidth;
208  }
209  else if ( ( headType == QgsArrowSymbolLayer::HeadReversed ) && ( length < headWidth ) )
210  {
211  pd = po + ( pd - po ) / length * headWidth;
212  length = headWidth;
213  }
214  else if ( ( headType == QgsArrowSymbolLayer::HeadDouble ) && ( length < 2 * headWidth ) )
215  {
216  QPointF v = ( pd - po ) / length * headWidth;
217  QPointF npo = ( po + pd ) / 2.0 - v;
218  QPointF npd = ( po + pd ) / 2.0 + v;
219  po = npo;
220  pd = npd;
221  length = 2 * headWidth;
222  }
223 
224  qreal bodyLength = length - headWidth;
225 
226  // unit vector
227  QPointF unitVec = ( pd - po ) / length;
228  // perpendicular vector
229  QPointF perpVec( -unitVec.y(), unitVec.x() );
230 
231  // set offset
232  po += perpVec * offset;
233  pd += perpVec * offset;
234 
235  if ( headType == QgsArrowSymbolLayer::HeadDouble )
236  {
237  // first head
238  polygon << po;
239  if ( arrowType == QgsArrowSymbolLayer::ArrowPlain || arrowType == QgsArrowSymbolLayer::ArrowRightHalf )
240  {
241  polygon << po + unitVec *headWidth + perpVec *headHeight;
242  polygon << po + unitVec *headWidth + perpVec * ( width * 0.5 );
243 
244  polygon << po + unitVec *bodyLength + perpVec * ( width * 0.5 );
245 
246  // second head
247  polygon << po + unitVec *bodyLength + perpVec *headHeight;
248  }
249  polygon << pd;
250 
251  if ( arrowType == QgsArrowSymbolLayer::ArrowPlain || arrowType == QgsArrowSymbolLayer::ArrowLeftHalf )
252  {
253  polygon << po + unitVec *bodyLength - perpVec *headHeight;
254  polygon << po + unitVec *bodyLength - perpVec * ( width * 0.5 );
255 
256  // end of the first head
257  polygon << po + unitVec *headWidth - perpVec * ( width * 0.5 );
258  polygon << po + unitVec *headWidth - perpVec *headHeight;
259  }
260  }
261  else if ( headType == QgsArrowSymbolLayer::HeadSingle )
262  {
263  if ( arrowType == QgsArrowSymbolLayer::ArrowPlain || arrowType == QgsArrowSymbolLayer::ArrowRightHalf )
264  {
265  polygon << po + perpVec * ( startWidth * 0.5 );
266  polygon << po + unitVec *bodyLength + perpVec * ( width * 0.5 );
267  polygon << po + unitVec *bodyLength + perpVec *headHeight;
268  }
269  else
270  {
271  polygon << po;
272  }
273  polygon << pd;
274  if ( arrowType == QgsArrowSymbolLayer::ArrowPlain || arrowType == QgsArrowSymbolLayer::ArrowLeftHalf )
275  {
276  polygon << po + unitVec *bodyLength - perpVec *headHeight;
277  polygon << po + unitVec *bodyLength - perpVec * ( width * 0.5 );
278  polygon << po - perpVec * ( startWidth * 0.5 );
279  }
280  else
281  {
282  polygon << po;
283  }
284  }
285  else if ( headType == QgsArrowSymbolLayer::HeadReversed )
286  {
287  polygon << po;
288  if ( arrowType == QgsArrowSymbolLayer::ArrowPlain || arrowType == QgsArrowSymbolLayer::ArrowRightHalf )
289  {
290  polygon << po + unitVec *headWidth + perpVec *headHeight;
291  polygon << po + unitVec *headWidth + perpVec * ( width * 0.5 );
292 
293  polygon << pd + perpVec * ( startWidth * 0.5 );
294  }
295  else
296  {
297  polygon << pd;
298  }
299  if ( arrowType == QgsArrowSymbolLayer::ArrowPlain || arrowType == QgsArrowSymbolLayer::ArrowLeftHalf )
300  {
301  polygon << pd - perpVec * ( startWidth * 0.5 );
302 
303  polygon << po + unitVec *headWidth - perpVec * ( width * 0.5 );
304  polygon << po + unitVec *headWidth - perpVec *headHeight;
305  }
306  else
307  {
308  polygon << pd;
309  }
310  }
311  // close the polygon
312  polygon << polygon.first();
313 
314  return polygon;
315 }

331 bool pointsToCircle( QPointF a, QPointF b, QPointF c, QPointF &center, qreal &radius )
332 {
333  qreal cx, cy;
334 
335  // AB and BC vectors
336  QPointF ab = b - a;
337  QPointF bc = c - b;
338 
339  // AB and BC middles
340  QPointF ab2 = ( a + b ) / 2.0;
341  QPointF bc2 = ( b + c ) / 2.0;
342 
343  // Aligned points
344  if ( std::fabs( ab.x() * bc.y() - ab.y() * bc.x() ) < 0.001 ) // Empirical threshold for nearly aligned points
345  return false;
346 
347  // in case AB is horizontal
348  if ( ab.y() == 0 )
349  {
350  cx = ab2.x();
351  cy = bc2.y() - ( cx - bc2.x() ) * bc.x() / bc.y();
352  }
353  //# BC horizontal
354  else if ( bc.y() == 0 )
355  {
356  cx = bc2.x();
357  cy = ab2.y() - ( cx - ab2.x() ) * ab.x() / ab.y();
358  }
359  // Otherwise
360  else
361  {
362  cx = ( bc2.y() - ab2.y() + bc.x() * bc2.x() / bc.y() - ab.x() * ab2.x() / ab.y() ) / ( bc.x() / bc.y() - ab.x() / ab.y() );
363  cy = bc2.y() - ( cx - bc2.x() ) * bc.x() / bc.y();
364  }
365  // Radius
366  radius = std::sqrt( ( a.x() - cx ) * ( a.x() - cx ) + ( a.y() - cy ) * ( a.y() - cy ) );
367  // Center
368  center.setX( cx );
369  center.setY( cy );
370  return true;
371 }

379 void pathArcTo( QPainterPath &path, QPointF circleCenter, qreal circleRadius, qreal angle_o, qreal angle_d, int direction )
380 {
381  QRectF circleRect( circleCenter - QPointF( circleRadius, circleRadius ), circleCenter + QPointF( circleRadius, circleRadius ) );
382  if ( direction == 1 )
383  {
384  if ( angle_o < angle_d )
385  path.arcTo( circleRect, angle_o / M_PI * 180.0, ( angle_d - angle_o ) / M_PI * 180.0 );
386  else
387  path.arcTo( circleRect, angle_o / M_PI * 180.0, 360.0 - ( angle_o - angle_d ) / M_PI * 180.0 );
388  }
389  else
390  {
391  if ( angle_o < angle_d )
392  path.arcTo( circleRect, angle_o / M_PI * 180.0, - ( 360.0 - ( angle_d - angle_o ) / M_PI * 180.0 ) );
393  else
394  path.arcTo( circleRect, angle_o / M_PI * 180.0, ( angle_d - angle_o ) / M_PI * 180.0 );
395  }
396 }

398 // Draw a "spiral" arc defined by circle arcs around a center, a start and an end radius
399 void spiralArcTo( QPainterPath &path, QPointF center, qreal startAngle, qreal startRadius, qreal endAngle, qreal endRadius, int direction )
400 {
401  // start point
402  QPointF A = circlePoint( center, startRadius, startAngle );
403  // end point
404  QPointF B = circlePoint( center, endRadius, endAngle );
405  // middle points
406  qreal deltaAngle;
407 
408  deltaAngle = endAngle - startAngle;
409  if ( direction * deltaAngle < 0.0 )
410  deltaAngle = deltaAngle + direction * 2 * M_PI;
411 
412  QPointF I1 = circlePoint( center, 0.75 * startRadius + 0.25 * endRadius, startAngle + 0.25 * deltaAngle );
413  QPointF I2 = circlePoint( center, 0.50 * startRadius + 0.50 * endRadius, startAngle + 0.50 * deltaAngle );
414  QPointF I3 = circlePoint( center, 0.25 * startRadius + 0.75 * endRadius, startAngle + 0.75 * deltaAngle );
415 
416  qreal cRadius;
417  QPointF cCenter;
418  // first circle arc
419  if ( ! pointsToCircle( A, I1, I2, cCenter, cRadius ) )
420  {
421  // aligned points => draw a straight line
422  path.lineTo( I2 );
423  }
424  else
425  {
426  // angles in the new circle
427  qreal a1 = std::atan2( cCenter.y() - A.y(), A.x() - cCenter.x() );
428  qreal a2 = std::atan2( cCenter.y() - I2.y(), I2.x() - cCenter.x() );
429  pathArcTo( path, cCenter, cRadius, a1, a2, direction );
430  }
431 
432  // second circle arc
433  if ( ! pointsToCircle( I2, I3, B, cCenter, cRadius ) )
434  {
435  // aligned points => draw a straight line
436  path.lineTo( B );
437  }
438  else
439  {
440  // angles in the new circle
441  qreal a1 = std::atan2( cCenter.y() - I2.y(), I2.x() - cCenter.x() );
442  qreal a2 = std::atan2( cCenter.y() - B.y(), B.x() - cCenter.x() );
443  pathArcTo( path, cCenter, cRadius, a1, a2, direction );
444  }
445 }

447 QPolygonF curvedArrow( QPointF po, QPointF pm, QPointF pd,
448  qreal startWidth, qreal width,
449  qreal headWidth, qreal headHeight,
450  QgsArrowSymbolLayer::HeadType headType, QgsArrowSymbolLayer::ArrowType arrowType,
451  qreal offset )
452 {
453  qreal circleRadius;
454  QPointF circleCenter;
455  if ( ! pointsToCircle( po, pm, pd, circleCenter, circleRadius ) )
456  {
457  // aligned points => draw a straight arrow
458  return straightArrow( po, pd, startWidth, width, headWidth, headHeight, headType, arrowType, offset );
459  }
460 
461  // angles of each point
462  qreal angle_o = clampAngle( std::atan2( circleCenter.y() - po.y(), po.x() - circleCenter.x() ) );
463  qreal angle_m = clampAngle( std::atan2( circleCenter.y() - pm.y(), pm.x() - circleCenter.x() ) );
464  qreal angle_d = clampAngle( std::atan2( circleCenter.y() - pd.y(), pd.x() - circleCenter.x() ) );
465 
466  // arc direction : 1 = counter-clockwise, -1 = clockwise
467  int direction = clampAngle( angle_m - angle_o ) < clampAngle( angle_m - angle_d ) ? 1 : -1;
468 
469  // arrow type, independent of the direction
470  int aType = 0;
471  if ( arrowType == QgsArrowSymbolLayer::ArrowRightHalf )
472  aType = direction;
473  else if ( arrowType == QgsArrowSymbolLayer::ArrowLeftHalf )
474  aType = -direction;
475 
476  qreal deltaAngle = angle_d - angle_o;
477  if ( direction * deltaAngle < 0.0 )
478  deltaAngle = deltaAngle + direction * 2 * M_PI;
479 
480  qreal length = euclidian_distance( po, pd );
481  // for close points and deltaAngle < 180, draw a straight line
482  if ( std::fabs( deltaAngle ) < M_PI && ( ( ( headType == QgsArrowSymbolLayer::HeadSingle ) && ( length < headWidth ) ) ||
483  ( ( headType == QgsArrowSymbolLayer::HeadReversed ) && ( length < headWidth ) ) ||
484  ( ( headType == QgsArrowSymbolLayer::HeadDouble ) && ( length < 2 * headWidth ) ) ) )
485  {
486  return straightArrow( po, pd, startWidth, width, headWidth, headHeight, headType, arrowType, offset );
487  }
488 
489  // ajust coordinates to include offset
490  circleRadius += offset;
491  po = circlePoint( circleCenter, circleRadius, angle_o );
492  pm = circlePoint( circleCenter, circleRadius, angle_m );
493  pd = circlePoint( circleCenter, circleRadius, angle_d );
494 
495  qreal headAngle = direction * std::atan( headWidth / circleRadius );
496 
497  QPainterPath path;
498 
499  if ( headType == QgsArrowSymbolLayer::HeadDouble )
500  {
501  // the first head
502  path.moveTo( po );
503  if ( aType <= 0 )
504  {
505  path.lineTo( circlePoint( circleCenter, circleRadius + direction * headHeight, angle_o + headAngle ) );
506 
507  pathArcTo( path, circleCenter, circleRadius + direction * width / 2, angle_o + headAngle, angle_d - headAngle, direction );
508 
509  // the second head
510  path.lineTo( circlePoint( circleCenter, circleRadius + direction * headHeight, angle_d - headAngle ) );
511  path.lineTo( pd );
512  }
513  else
514  {
515  pathArcTo( path, circleCenter, circleRadius, angle_o, angle_d, direction );
516  }
517  if ( aType >= 0 )
518  {
519  path.lineTo( circlePoint( circleCenter, circleRadius - direction * headHeight, angle_d - headAngle ) );
520 
521  pathArcTo( path, circleCenter, circleRadius - direction * width / 2, angle_d - headAngle, angle_o + headAngle, -direction );
522 
523  // the end of the first head
524  path.lineTo( circlePoint( circleCenter, circleRadius - direction * headHeight, angle_o + headAngle ) );
525  path.lineTo( po );
526  }
527  else
528  {
529  pathArcTo( path, circleCenter, circleRadius, angle_d, angle_o, -direction );
530  }
531  }
532  else if ( headType == QgsArrowSymbolLayer::HeadSingle )
533  {
534  if ( aType <= 0 )
535  {
536  path.moveTo( circlePoint( circleCenter, circleRadius + direction * startWidth / 2, angle_o ) );
537 
538  spiralArcTo( path, circleCenter, angle_o, circleRadius + direction * startWidth / 2, angle_d - headAngle, circleRadius + direction * width / 2, direction );
539 
540  // the arrow head
541  path.lineTo( circlePoint( circleCenter, circleRadius + direction * headHeight, angle_d - headAngle ) );
542  path.lineTo( pd );
543  }
544  else
545  {
546  path.moveTo( po );
547  pathArcTo( path, circleCenter, circleRadius, angle_o, angle_d, direction );
548  }
549  if ( aType >= 0 )
550  {
551  path.lineTo( circlePoint( circleCenter, circleRadius - direction * headHeight, angle_d - headAngle ) );
552 
553  spiralArcTo( path, circleCenter, angle_d - headAngle, circleRadius - direction * width / 2, angle_o, circleRadius - direction * startWidth / 2, -direction );
554

676 void QgsArrowSymbolLayer::renderPolyline( const QPolygonF &points, QgsSymbolRenderContext &context )
677 {
678  Q_UNUSED( points );
679 
680  if ( !context.renderContext().painter() )
681  {
682  return;
683  }
684 
685  context.renderContext().expressionContext().appendScope( mExpressionScope.get() );
686  mExpressionScope->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_POINT_COUNT, points.size() + 1, true ) );
687  mExpressionScope->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_POINT_NUM, 1, true ) );
688  if ( isCurved() )
689  {
690  _resolveDataDefined( context );
691 
692  if ( ! isRepeated() )
693  {
694  if ( points.size() >= 3 )
695  {
696  // origin point
697  QPointF po( points.at( 0 ) );
698  // middle point
699  QPointF pm( points.at( points.size() / 2 ) );
700  // destination point
701  QPointF pd( points.back() );
702 
703  QPolygonF poly = curvedArrow( po, pm, pd, mScaledArrowStartWidth, mScaledArrowWidth, mScaledHeadLength, mScaledHeadThickness, mComputedHeadType, mComputedArrowType, mScaledOffset );
704  mSymbol->renderPolygon( poly, /* rings */ nullptr, context.feature(), context.renderContext(), -1, context.selected() );
705  }
706  // straight arrow
707  else if ( points.size() == 2 )
708  {
709  // origin point
710  QPointF po( points.at( 0 ) );
711  // destination point
712  QPointF pd( points.at( 1 ) );
713 
714  QPolygonF poly = straightArrow( po, pd, mScaledArrowStartWidth, mScaledArrowWidth, mScaledHeadLength, mScaledHeadThickness, mComputedHeadType, mComputedArrowType, mScaledOffset );
715  mSymbol->renderPolygon( poly, /* rings */ nullptr, context.feature(), context.renderContext(), -1, context.selected() );
716  }
717  }
718  else
719  {
720  for ( int pIdx = 0; pIdx < points.size() - 1; pIdx += 2 )
721  {
722  mExpressionScope->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_POINT_NUM, pIdx + 1, true ) );
723  _resolveDataDefined( context );
724 
725  if ( points.size() - pIdx >= 3 )
726  {
727  // origin point
728  QPointF po( points.at( pIdx ) );
729  // middle point
730  QPointF pm( points.at( pIdx + 1 ) );
731  // destination point
732  QPointF pd( points.at( pIdx + 2 ) );
733 
734  QPolygonF poly = curvedArrow( po, pm, pd, mScaledArrowStartWidth, mScaledArrowWidth, mScaledHeadLength, mScaledHeadThickness, mComputedHeadType, mComputedArrowType, mScaledOffset );
735  mSymbol->renderPolygon( poly, /* rings */ nullptr, context.feature(), context.renderContext(), -1, context.selected() );
736  }
737  // straight arrow
738  else if ( points.size() - pIdx == 2 )
739  {
740  // origin point
741  QPointF po( points.at( pIdx ) );
742  // destination point
743  QPointF pd( points.at( pIdx + 1 ) );
744 
745  QPolygonF poly = straightArrow( po, pd, mScaledArrowStartWidth, mScaledArrowWidth, mScaledHeadLength, mScaledHeadThickness, mComputedHeadType, mComputedArrowType, mScaledOffset );
746  mSymbol->renderPolygon( poly, /* rings */ nullptr, context.feature(), context.renderContext(), -1, context.selected() );
747  }
748  }
749  }
750  }
751  else
752  {
753  if ( !isRepeated() )
754  {
755  _resolveDataDefined( context );
756 
757  if ( !points.isEmpty() )
758  {
759  // origin point
760  QPointF po( points.at( 0 ) );
761  // destination point
762  QPointF pd( points.back() );
763 
764  QPolygonF poly = straightArrow( po, pd, mScaledArrowStartWidth, mScaledArrowWidth, mScaledHeadLength, mScaledHeadThickness, mComputedHeadType, mComputedArrowType, mScaledOffset );
765  mSymbol->renderPolygon( poly, /* rings */ nullptr, context.feature(), context.renderContext(), -1, context.selected() );
766  }
767  }
768  else
769  {
770  // only straight arrows
771  for ( int pIdx = 0; pIdx < points.size() - 1; pIdx++ )
772  {
773  mExpressionScope->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_POINT_NUM, pIdx + 1, true ) );
774  _resolveDataDefined( context );
775 
776  // origin point
777  QPointF po( points.at( pIdx ) );
778  // destination point
779  QPointF pd( points.at( pIdx + 1 ) );
780 
781  QPolygonF poly = straightArrow( po, pd, mScaledArrowStartWidth, mScaledArrowWidth, mScaledHeadLength, mScaledHeadThickness, mComputedHeadType, mComputedArrowType, mScaledOffset );
782  mSymbol->renderPolygon( poly, /* rings */ nullptr, context.feature(), context.renderContext(), -1, context.selected() );
783  }
784  }
785  }
786  context.renderContext().expressionContext().popScope();
787 }
