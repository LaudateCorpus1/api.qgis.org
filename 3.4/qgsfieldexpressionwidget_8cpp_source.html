175 void QgsFieldExpressionWidget::setField( const QString &fieldName )
176 {
177  if ( fieldName.isEmpty() )
178  {
179  setRow( -1 );
180  return;
181  }
182 
183  QModelIndex idx = mFieldProxyModel->sourceFieldModel()->indexFromName( fieldName );
184  if ( !idx.isValid() )
185  {
186  // try to remove quotes and white spaces
187  QString simpleFieldName = fieldName.trimmed();
188  if ( simpleFieldName.startsWith( '"' ) && simpleFieldName.endsWith( '"' ) )
189  {
190  simpleFieldName.remove( 0, 1 ).chop( 1 );
191  idx = mFieldProxyModel->sourceFieldModel()->indexFromName( simpleFieldName );
192  }
193 
194  if ( !idx.isValid() )
195  {
196  // new expression
197  mFieldProxyModel->sourceFieldModel()->setExpression( fieldName );
198  idx = mFieldProxyModel->sourceFieldModel()->indexFromName( fieldName );
199  }
200  }
201  QModelIndex proxyIndex = mFieldProxyModel->mapFromSource( idx );
202  mCombo->setCurrentIndex( proxyIndex.row() );
203  currentFieldChanged();
204 }

217 
218  QgsExpressionBuilderDialog dlg( vl, currentExpression, this, QStringLiteral( "generic" ), context );
219  if ( mDa )

221  dlg.setGeomCalculator( *mDa );

224  dlg.setAllowEvalErrors( mAllowEvalErrors );
225
