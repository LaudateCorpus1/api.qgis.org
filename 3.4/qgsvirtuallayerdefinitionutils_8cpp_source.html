24 QgsVirtualLayerDefinition QgsVirtualLayerDefinitionUtils::fromJoinedLayer( QgsVectorLayer *layer )
25 {
26  QgsVirtualLayerDefinition def;
27 
28  QStringList leftJoins;
29  QStringList columns;
30 
31  // add the geometry column if the layer is spatial
32  if ( layer->isSpatial() )
33  columns << "t.geometry";
34 
35  // look for the uid
36  QgsFields fields = layer->dataProvider()->fields();
37  {
38  QgsAttributeList pk = layer->dataProvider()->pkAttributeIndexes();
39  if ( pk.size() == 1 )
40  {
41  def.setUid( fields.field( pk[0] ).name() );
42  }
43  else
44  {
45  // find an uid name
46  QString uid = QStringLiteral( "uid" );
47  while ( fields.lookupField( uid ) != -1 )
48  uid += QLatin1String( "_" ); // add "_" each time this name already exists
49 
50  // add a column
51  columns << "t.rowid AS " + uid;
52  def.setUid( uid );
53  }
54  }
55  const QgsFields providerFields = layer->dataProvider()->fields();
56  for ( const auto &f : providerFields )
57  {
58  columns << "t.\"" + f.name() + "\"";
59  }
60 
61  int joinIdx = 0;
62  Q_FOREACH ( const QgsVectorLayerJoinInfo &join, layer->vectorJoins() )
63  {
64  QString joinName = QStringLiteral( "j%1" ).arg( ++joinIdx );
65  QgsVectorLayer *joinedLayer = join.joinLayer();
66  if ( !joinedLayer )
67  continue;
68  QString prefix = join.prefix().isEmpty() ? joinedLayer->name() + "_" : join.prefix();
69 
70  leftJoins << QStringLiteral( "LEFT JOIN \"%1\" AS %2 ON t.\"%5\"=%2.\"%3\"" ).arg( joinedLayer->id(), joinName, join.joinFieldName(), join.targetFieldName() );
71  if ( join.joinFieldNamesSubset() )
72  {
73  Q_FOREACH ( const QString &f, *join.joinFieldNamesSubset() )
74  {
75  columns << joinName + ".\"" + f + "\" AS \"" + prefix + f + "\"";
76  }
77  }
78  else
79  {
80  const QgsFields joinFields = joinedLayer->fields();
81  for ( const QgsField &f : joinFields )
82  {
83  if ( f.name() == join.joinFieldName() )
84  continue;
85  columns << joinName + ".\"" + f.name() + "\" AS \"" + prefix + f.name() + "\"";
86  }
87  }
88  }
89 
90  QString query = "SELECT " + columns.join( QStringLiteral( ", " ) ) + " FROM \"" + layer->id() + "\" AS t " + leftJoins.join( QStringLiteral( " " ) );
91  def.setQuery( query );
92 
93  return def;
94 }
