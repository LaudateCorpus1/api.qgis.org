134 QStringList QgsSymbolLayerRegistry::symbolLayersForType( QgsSymbol::SymbolType type )
135 {
136  QStringList lst;
137  QMap<QString, QgsSymbolLayerAbstractMetadata *>::ConstIterator it = mMetadata.constBegin();
138  for ( ; it != mMetadata.constEnd(); ++it )
139  {
140  if ( it.value()->type() == type || it.value()->type() == QgsSymbol::Hybrid )
141  lst.append( it.key() );
142  }
143  return lst;
144 }
