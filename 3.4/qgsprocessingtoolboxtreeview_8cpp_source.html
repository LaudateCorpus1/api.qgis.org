101 QModelIndex QgsProcessingToolboxTreeView::findFirstVisibleAlgorithm( const QModelIndex &parent )
102 {
103  for ( int r = 0; r < mModel->rowCount( parent ); ++r )
104  {
105  QModelIndex proxyIndex = mModel->index( r, 0, parent );
106  QModelIndex sourceIndex = mModel->mapToSource( proxyIndex );
107  if ( mToolboxModel->isAlgorithm( sourceIndex ) )
108  return proxyIndex;
109 
110  QModelIndex index = findFirstVisibleAlgorithm( proxyIndex );
111  if ( index.isValid() )
112  return index;
113  }
114  return QModelIndex();
115 }
