00430 
00431     QDomElement ruleChildElem = ruleElem.firstChildElement();
00432     while ( !ruleChildElem.isNull() )
00433     {
00434       // rule has filter or min/max scale denominator, use the RuleRenderer

00440         needRuleRenderer = true;
00441         break;
