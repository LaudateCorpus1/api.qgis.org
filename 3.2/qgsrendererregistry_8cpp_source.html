97 bool QgsRendererRegistry::addRenderer( QgsRendererAbstractMetadata *metadata )
98 {
99  if ( !metadata || mRenderers.contains( metadata->name() ) )
100  return false;
101 
102  mRenderers[metadata->name()] = metadata;
103  mRenderersOrder << metadata->name();
104  return true;
105 }

123 QStringList QgsRendererRegistry::renderersList( QgsRendererAbstractMetadata::LayerTypes layerTypes ) const
124 {
125  QStringList renderers;
126  Q_FOREACH ( const QString &renderer, mRenderersOrder )
127  {
128  QgsRendererAbstractMetadata *r = mRenderers.value( renderer );
129  if ( r && r->compatibleLayerTypes() & layerTypes )
130  renderers << renderer;
131  }
132  return renderers;
133 }

135 QStringList QgsRendererRegistry::renderersList( const QgsVectorLayer *layer ) const
136 {
137  QgsRendererAbstractMetadata::LayerType layerType = QgsRendererAbstractMetadata::All;
138 
139  switch ( layer->geometryType() )
140  {
141  case QgsWkbTypes::PointGeometry:
142  layerType = QgsRendererAbstractMetadata::PointLayer;
143  break;
144 
145  case QgsWkbTypes::LineGeometry:
146  layerType = QgsRendererAbstractMetadata::LineLayer;
147  break;
148 
149  case QgsWkbTypes::PolygonGeometry:
150  layerType = QgsRendererAbstractMetadata::PolygonLayer;
151  break;
152 
153  case QgsWkbTypes::UnknownGeometry:
154  case QgsWkbTypes::NullGeometry:
155  break;
156  }
157 
158  return renderersList( layerType );
159 }
