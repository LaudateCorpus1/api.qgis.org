34 QgsRectangle QgsRectangle::fromWkt( const QString &wkt )
35 {
36  QgsGeometry geom = QgsGeometry::fromWkt( wkt );
37  if ( geom.isMultipart() )
38  return QgsRectangle();
39 
40  QgsPolygonXY poly = geom.asPolygon();
41 
42  if ( poly.size() != 1 )
43  return QgsRectangle();
44 
45  QgsPolylineXY polyline = geom.asPolygon().at( 0 );
46  if ( polyline.size() == 5 && polyline.at( 0 ) == polyline.at( 4 ) && geom.isGeosValid() )
47  return QgsRectangle( polyline.at( 0 ).x(), polyline.at( 0 ).y(), polyline.at( 2 ).x(), polyline.at( 2 ).y() );
48  else
49  return QgsRectangle();
50 }

120 QString QgsRectangle::toString( int precision ) const
121 {
122  QString rep;
123 
124  if ( precision < 0 )
125  {
126  precision = 0;
127  if ( ( width() < 10 || height() < 10 ) && ( width() > 0 && height() > 0 ) )
128  {
129  precision = static_cast<int>( std::ceil( -1.0 * std::log10( std::min( width(), height() ) ) ) ) + 1;
130  // sanity check
131  if ( precision > 20 )
132  precision = 20;
133  }
134  }
135 
136  if ( isEmpty() )
137  rep = QStringLiteral( "Empty" );
138  else
139  rep = QStringLiteral( "%1,%2 : %3,%4" )
140  .arg( mXmin, 0, 'f', precision )
141  .arg( mYmin, 0, 'f', precision )
142  .arg( mXmax, 0, 'f', precision )
143  .arg( mYmax, 0, 'f', precision );
144 
145  QgsDebugMsgLevel( QString( "Extents : %1" ).arg( rep ), 4 );
146 
147  return rep;
148 }
