43 static bool _initWidgetFunction( const QString &name, QgsSymbolLayerWidgetFunc f )
44 {
45  QgsSymbolLayerRegistry *reg = QgsApplication::symbolLayerRegistry();
46 
47  QgsSymbolLayerAbstractMetadata *abstractMetadata = reg->symbolLayerMetadata( name );
48  if ( !abstractMetadata )
49  {
50  QgsDebugMsg( "Failed to find symbol layer's entry in registry: " + name );
51  return false;
52  }
53  QgsSymbolLayerMetadata *metadata = dynamic_cast<QgsSymbolLayerMetadata *>( abstractMetadata );
54  if ( !metadata )
55  {
56  QgsDebugMsg( "Failed to cast symbol layer's metadata: " + name );
57  return false;
58  }
59  metadata->setWidgetFunction( f );
60  return true;
61 }

166 void QgsLayerPropertiesWidget::populateLayerTypes()
167 {
168  QStringList symbolLayerIds = QgsApplication::symbolLayerRegistry()->symbolLayersForType( mSymbol->type() );
169 
170  const auto constSymbolLayerIds = symbolLayerIds;
171  for ( const QString &symbolLayerId : constSymbolLayerIds )
172  cboLayerType->addItem( QgsApplication::symbolLayerRegistry()->symbolLayerMetadata( symbolLayerId )->visibleName(), symbolLayerId );
173 
174  if ( mSymbol->type() == QgsSymbol::Fill )
175  {
176  QStringList lineLayerIds = QgsApplication::symbolLayerRegistry()->symbolLayersForType( QgsSymbol::Line );
177  const auto constLineLayerIds = lineLayerIds;
178  for ( const QString &lineLayerId : constLineLayerIds )
179  {
180  QgsSymbolLayerAbstractMetadata *layerInfo = QgsApplication::symbolLayerRegistry()->symbolLayerMetadata( lineLayerId );
181  if ( layerInfo->type() != QgsSymbol::Hybrid )
182  {
183  QString visibleName = layerInfo->visibleName();
184  QString name = QString( tr( "Outline: %1" ) ).arg( visibleName );
185  cboLayerType->addItem( name, lineLayerId );
186  }
187  }
188  }
189 }

191 void QgsLayerPropertiesWidget::updateSymbolLayerWidget( QgsSymbolLayer *layer )
192 {
193  if ( stackedWidget->currentWidget() != pageDummy )
194  {
195  // stop updating from the original widget
196  if ( QgsSymbolLayerWidget *w = qobject_cast< QgsSymbolLayerWidget * >( stackedWidget->currentWidget() ) )
197  disconnect( w, &QgsSymbolLayerWidget::changed, this, &QgsLayerPropertiesWidget::emitSignalChanged );
198  stackedWidget->removeWidget( stackedWidget->currentWidget() );
199  }
200 
201  QgsSymbolLayerRegistry *pReg = QgsApplication::symbolLayerRegistry();
202 
203  QString layerType = layer->layerType();
204  QgsSymbolLayerAbstractMetadata *am = pReg->symbolLayerMetadata( layerType );
205  if ( am )
206  {
207  QgsSymbolLayerWidget *w = am->createSymbolLayerWidget( mVectorLayer );
208  if ( w )
209  {
210  w->setContext( mContext );
211  w->setSymbolLayer( layer );
212  stackedWidget->addWidget( w );
213  stackedWidget->setCurrentWidget( w );
214  // start receiving updates from widget
215  connect( w, &QgsSymbolLayerWidget::changed, this, &QgsLayerPropertiesWidget::emitSignalChanged );
216  connect( w, &QgsSymbolLayerWidget::symbolChanged, this, &QgsLayerPropertiesWidget::reloadLayer );
217  return;
218  }
219  }
220  // When anything is not right
221  stackedWidget->setCurrentWidget( pageDummy );
222 }

224 QgsExpressionContext QgsLayerPropertiesWidget::createExpressionContext() const
225 {
226  if ( mContext.expressionContext() )
227  return *mContext.expressionContext();
228 
229  QgsExpressionContext expContext;
230  expContext << QgsExpressionContextUtils::globalScope()
231  << QgsExpressionContextUtils::projectScope( QgsProject::instance() )
232  << QgsExpressionContextUtils::atlasScope( nullptr );
233 
234  if ( mContext.mapCanvas() )
235  {
236  expContext << QgsExpressionContextUtils::mapSettingsScope( mContext.mapCanvas()->mapSettings() )
237  << new QgsExpressionContextScope( mContext.mapCanvas()->expressionContextScope() );
238  }
239  else
240  {
241  expContext << QgsExpressionContextUtils::mapSettingsScope( QgsMapSettings() );
242  }
243 
244  expContext << QgsExpressionContextUtils::layerScope( mVectorLayer );
245 
246  QgsExpressionContextScope *symbolScope = QgsExpressionContextUtils::updateSymbolScope( nullptr, new QgsExpressionContextScope() );
247  if ( mLayer )
248  {
249  //cheat a bit - set the symbol color variable to match the symbol layer's color (when we should really be using the *symbols*
250  //color, but that's not accessible here). 99% of the time these will be the same anyway
251  symbolScope->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_SYMBOL_COLOR, mLayer->color(), true ) );
252  }
253  expContext << symbolScope;
254  expContext.lastScope()->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_PART_COUNT, 1, true ) );
255  expContext.lastScope()->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_PART_NUM, 1, true ) );
256  expContext.lastScope()->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_POINT_COUNT, 1, true ) );
257  expContext.lastScope()->addVariable( QgsExpressionContextScope::StaticVariable( QgsExpressionContext::EXPR_GEOMETRY_POINT_NUM, 1, true ) );
258 
259  // additional scopes
260  const auto constAdditionalExpressionContextScopes = mContext.additionalExpressionContextScopes();
261  for ( const QgsExpressionContextScope &scope : constAdditionalExpressionContextScopes )
262  {
263  expContext.appendScope( new QgsExpressionContextScope( scope ) );
264  }
265 
266  //TODO - show actual value
267  expContext.setOriginalValueVariable( QVariant() );
268 
269  expContext.setHighlightedVariables( QStringList() << QgsExpressionContext::EXPR_ORIGINAL_VALUE << QgsExpressionContext::EXPR_SYMBOL_COLOR
270  << QgsExpressionContext::EXPR_GEOMETRY_PART_COUNT << QgsExpressionContext::EXPR_GEOMETRY_PART_NUM
271  << QgsExpressionContext::EXPR_GEOMETRY_POINT_COUNT << QgsExpressionContext::EXPR_GEOMETRY_POINT_NUM
272  << QgsExpressionContext::EXPR_CLUSTER_COLOR << QgsExpressionContext::EXPR_CLUSTER_SIZE );
273 
274  return expContext;
275 }

292 void QgsLayerPropertiesWidget::layerTypeChanged()
293 {
294  QgsSymbolLayer *layer = mLayer;
295  if ( !layer )
296  return;
297  QString newLayerType = cboLayerType->currentData().toString();
298  if ( layer->layerType() == newLayerType )
299  return;
300 
301  // get creation function for new layer from registry
302  QgsSymbolLayerRegistry *pReg = QgsApplication::symbolLayerRegistry();
303  QgsSymbolLayerAbstractMetadata *am = pReg->symbolLayerMetadata( newLayerType );
304  if ( !am ) // check whether the metadata is assigned
305  return;
306 
307  // change layer to a new (with different type)
308  // base new layer on existing layer's properties
309  QgsSymbolLayer *newLayer = am->createSymbolLayer( layer->properties() );
310  if ( !newLayer )
311  return;
312 
313  updateSymbolLayerWidget( newLayer );
314  emit changeLayer( newLayer );
315 }
