40 QgsGeometryCheckError::QgsGeometryCheckError( const QgsGeometryCheck *check,
41  const QgsGeometryCheckerUtils::LayerFeature &layerFeature,
42  const QgsPointXY &errorLocation,
43  QgsVertexId vidx,
44  const QVariant &value,
45  ValueType valueType )
46  : mCheck( check )
47  , mLayerId( layerFeature.layerId() )
48  , mFeatureId( layerFeature.feature().id() )
49  , mErrorLocation( errorLocation )
50  , mVidx( vidx )
51  , mValue( value )
52  , mValueType( valueType )
53  , mStatus( StatusPending )
54 {
55  if ( vidx.part != -1 )
56  {
57  const QgsGeometry geom = layerFeature.geometry();
58  mGeometry = QgsGeometry( QgsGeometryCheckerUtils::getGeomPart( geom.constGet(), vidx.part )->clone() );
59  }
60  else
61  {
62  mGeometry = layerFeature.geometry();
63  }
64  if ( !layerFeature.useMapCrs() )
65  {
66  QgsVectorLayer *vl = layerFeature.layer().data();
67  if ( vl )
68  {
69  QgsCoordinateTransform ct( vl->crs(), check->context()->mapCrs, check->context()->transformContext );
70  try
71  {
72  mGeometry.transform( ct );
73  mErrorLocation = ct.transform( mErrorLocation );
74  }
75  catch ( const QgsCsException & )
76  {
77  QgsDebugMsg( QStringLiteral( "Can not show error in current map coordinate reference system" ) );
78  }
79  }
80  }

124 bool QgsGeometryCheckError::handleChanges( const QgsGeometryCheck::Changes &changes )
125 {
126  if ( status() == StatusObsolete )
127  {
128  return false;
129  }
130 
131  for ( const QgsGeometryCheck::Change &change : changes.value( layerId() ).value( featureId() ) )
132  {
133  if ( change.what == QgsGeometryCheck::ChangeFeature )
134  {
135  if ( change.type == QgsGeometryCheck::ChangeRemoved )
136  {
137  return false;
138  }
139  else if ( change.type == QgsGeometryCheck::ChangeChanged )
140  {
141  // If the check is checking the feature at geometry nodes level, the
142  // error almost certainly invalid after a geometry change. In the other
143  // cases, it might likely still be valid.
144  return mCheck->checkType() != QgsGeometryCheck::FeatureNodeCheck;
145  }
146  }
147  else if ( change.what == QgsGeometryCheck::ChangePart )
148  {
149  if ( mVidx.part == change.vidx.part )
150  {
151  return false;
152  }
153  else if ( mVidx.part > change.vidx.part )
154  {
155  mVidx.part += change.type == QgsGeometryCheck::ChangeAdded ? 1 : -1;
156  }
157  }
158  else if ( change.what == QgsGeometryCheck::ChangeRing )
159  {
160  if ( mVidx.partEqual( change.vidx ) )
161  {
162  if ( mVidx.ring == change.vidx.ring )
163  {
164  return false;
165  }
166  else if ( mVidx.ring > change.vidx.ring )
167  {
168  mVidx.ring += change.type == QgsGeometryCheck::ChangeAdded ? 1 : -1;
169  }
170  }
171  }
172  else if ( change.what == QgsGeometryCheck::ChangeNode )
173  {
174  if ( mVidx.ringEqual( change.vidx ) )
175  {
176  if ( mVidx.vertex == change.vidx.vertex )
177  {
178  return false;
179  }
180  else if ( mVidx.vertex > change.vidx.vertex )
181  {
182  mVidx.vertex += change.type == QgsGeometryCheck::ChangeAdded ? 1 : -1;
183  }
184  }
185  }
186  }
187  return true;
188 }
