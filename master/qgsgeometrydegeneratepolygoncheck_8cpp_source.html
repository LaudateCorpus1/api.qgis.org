22 void QgsGeometryDegeneratePolygonCheck::collectErrors( const QMap<QString, QgsFeaturePool *> &featurePools, QList<QgsGeometryCheckError *> &errors, QStringList &messages, QgsFeedback *feedback, const LayerFeatureIds &ids ) const
23 {
24  Q_UNUSED( messages )
25 
26  QMap<QString, QgsFeatureIds> featureIds = ids.isEmpty() ? allLayerFeatureIds( featurePools ) : ids.toMap();
27  QgsGeometryCheckerUtils::LayerFeatures layerFeatures( featurePools, featureIds, compatibleGeometryTypes(), feedback, mContext );
28  for ( const QgsGeometryCheckerUtils::LayerFeature &layerFeature : layerFeatures )
29  {
30  const QgsAbstractGeometry *geom = layerFeature.geometry().constGet();
31  for ( int iPart = 0, nParts = geom->partCount(); iPart < nParts; ++iPart )
32  {
33  for ( int iRing = 0, nRings = geom->ringCount( iPart ); iRing < nRings; ++iRing )
34  {
35  if ( QgsGeometryCheckerUtils::polyLineSize( geom, iPart, iRing ) < 3 )
36  {
37  QgsVertexId vidx( iPart, iRing );
38  errors.append( new QgsGeometryCheckError( this, layerFeature, geom->vertexAt( vidx ), vidx ) );
39  }
40  }
41  }
42  }

45 void QgsGeometryDegeneratePolygonCheck::fixError( const QMap<QString, QgsFeaturePool *> &featurePools, QgsGeometryCheckError *error, int method, const QMap<QString, int> & /*mergeAttributeIndices*/, Changes &changes ) const
46 {
47  QgsFeaturePool *featurePool = featurePools[ error->layerId() ];
48  QgsFeature feature;
49  if ( !featurePool->getFeature( error->featureId(), feature ) )
50  {
51  error->setObsolete();
52  return;
53  }
54  QgsGeometry featureGeometry = feature.geometry();
55  const QgsAbstractGeometry *geom = featureGeometry.constGet();
56  QgsVertexId vidx = error->vidx();
57 
58  // Check if ring still exists
59  if ( !vidx.isValid( geom ) )
60  {
61  error->setObsolete();
62  return;
63  }
64 
65  // Check if error still applies
66  if ( QgsGeometryCheckerUtils::polyLineSize( geom, vidx.part, vidx.ring ) >= 3 )
67  {
68  error->setObsolete();
69  return;
70  }
71 
72  // Fix error
73  if ( method == NoChange )
74  {
75  error->setFixed( method );
76  }
77  else if ( method == DeleteRing )
78  {
79  deleteFeatureGeometryRing( featurePools, error->layerId(), feature, vidx.part, vidx.ring, changes );
80  error->setFixed( method );
81  }
82  else
83  {
84  error->setFixFailed( tr( "Unknown method" ) );
85  }
86 }
