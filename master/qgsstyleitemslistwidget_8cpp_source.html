271 void QgsStyleItemsListWidget::populateGroups()
272 {
273  if ( !mStyle )
274  return;
275 
276  mUpdatingGroups = true;
277  groupsCombo->blockSignals( true );
278  groupsCombo->clear();
279 
280  groupsCombo->addItem( tr( "Favorites" ), QVariant( "favorite" ) );
281 
282  QString allText = tr( "All Symbols" );
283  if ( mModel->entityFilterEnabled() )
284  {
285  switch ( mModel->entityFilter() )
286  {
287  case QgsStyle::SymbolEntity:
288  allText = tr( "All Symbols" );
289  break;
290 
291  case QgsStyle::ColorrampEntity:
292  allText = tr( "All Color Ramps" );
293  break;
294 
295  case QgsStyle::TextFormatEntity:
296  allText = tr( "All Text Formats" );
297  break;
298 
299  case QgsStyle::LabelSettingsEntity:
300  allText = tr( "All Label Settings" );
301  break;
302 
303  case QgsStyle::TagEntity:
304  case QgsStyle::SmartgroupEntity:
305  break;
306  }
307  }
308 
309  groupsCombo->addItem( allText, QVariant( "all" ) );
310 
311  int index = 2;
312  QStringList tags = mStyle->tags();
313  if ( tags.count() > 0 )
314  {
315  tags.sort();
316  groupsCombo->insertSeparator( index );
317  const auto constTags = tags;
318  for ( const QString &tag : constTags )
319  {
320  groupsCombo->addItem( tag, QVariant( "tag" ) );
321  index++;
322  }
323  }
324 
325  QStringList groups = mStyle->smartgroupNames();
326  if ( groups.count() > 0 )
327  {
328  groups.sort();
329  groupsCombo->insertSeparator( index + 1 );
330  const auto constGroups = groups;
331  for ( const QString &group : constGroups )
332  {
333  groupsCombo->addItem( group, QVariant( "smartgroup" ) );
334  }
335  }
336  groupsCombo->blockSignals( false );
337 
338  QgsSettings settings;
339  index = settings.value( QStringLiteral( "qgis/symbolsListGroupsIndex" ), 0 ).toInt();
340  groupsCombo->setCurrentIndex( index );
341 
342  mUpdatingGroups = false;
343 
344  updateModelFilters();
345 }

347 void QgsStyleItemsListWidget::updateModelFilters()
348 {
349  if ( mUpdatingGroups || !mModel )
350  return;
351 
352  const QString text = groupsCombo->currentText();
353  const bool isFreeText = text != groupsCombo->itemText( groupsCombo->currentIndex() );
354 
355  if ( isFreeText )
356  {
357  mModel->setFavoritesOnly( false );
358  mModel->setTagId( -1 );
359  mModel->setSmartGroupId( -1 );
360  mModel->setFilterString( groupsCombo->currentText() );
361  }
362  else if ( groupsCombo->currentData().toString() == QLatin1String( "favorite" ) )
363  {
364  mModel->setFavoritesOnly( true );
365  mModel->setTagId( -1 );
366  mModel->setSmartGroupId( -1 );
367  mModel->setFilterString( QString() );
368  }
369  else if ( groupsCombo->currentData().toString() == QLatin1String( "all" ) )
370  {
371  mModel->setFavoritesOnly( false );
372  mModel->setTagId( -1 );
373  mModel->setSmartGroupId( -1 );
374  mModel->setFilterString( QString() );
375  }
376  else if ( groupsCombo->currentData().toString() == QLatin1String( "smartgroup" ) )
377  {
378  mModel->setFavoritesOnly( false );
379  mModel->setTagId( -1 );
380  mModel->setSmartGroupId( mStyle->smartgroupId( text ) );
381  mModel->setFilterString( QString() );
382  }
383  else
384  {
385  mModel->setFavoritesOnly( false );
386  mModel->setTagId( mStyle->tagId( text ) );
387  mModel->setSmartGroupId( -1 );
388  mModel->setFilterString( QString() );
389  }
390 }

392 void QgsStyleItemsListWidget::openStyleManager()
393 {
394  // prefer to use global window manager to open the style manager, if possible!
395  // this allows reuse of an existing non-modal window instead of opening a new modal window.
396  // Note that we only use the non-modal dialog if we're open in the panel -- if we're already
397  // open as part of a modal dialog, then we MUST use another modal dialog or the result will
398  // not be focusable!
399  QgsPanelWidget *panel = QgsPanelWidget::findParentPanel( this );
400  if ( !panel || !panel->dockMode()
401  || !QgsGui::windowManager()
402  || !QgsGui::windowManager()->openStandardDialog( QgsWindowManagerInterface::DialogStyleManager ) )
403  {
404  // fallback to modal dialog
405  QgsStyleManagerDialog dlg( mStyle, this );
406  dlg.exec();
407 
408  updateModelFilters(); // probably not needed -- the model should automatically update if any changes were made
409  }
410 }
