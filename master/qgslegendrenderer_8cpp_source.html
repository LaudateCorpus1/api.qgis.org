59 void QgsLegendRenderer::exportLegendToJson( const QgsRenderContext &context, QgsLayerTreeGroup *nodeGroup, QJsonObject &json )
60 {
61  QJsonArray nodes;
62  const QList<QgsLayerTreeNode *> childNodes = nodeGroup->children();
63  for ( QgsLayerTreeNode *node : childNodes )
64  {
65  if ( QgsLayerTree::isGroup( node ) )
66  {
67  QgsLayerTreeGroup *nodeGroup = QgsLayerTree::toGroup( node );
68  const QModelIndex idx = mLegendModel->node2index( nodeGroup );
69  const QString text = mLegendModel->data( idx, Qt::DisplayRole ).toString();
70 
71  QJsonObject group;
72  group[ QStringLiteral( "type" ) ] = QStringLiteral( "group" );
73  group[ QStringLiteral( "title" ) ] = text;
74  exportLegendToJson( context, nodeGroup, group );
75  nodes.append( group );
76  }
77  else if ( QgsLayerTree::isLayer( node ) )
78  {
79  QJsonObject group;
80  group[ QStringLiteral( "type" ) ] = QStringLiteral( "layer" );
81 
82  QgsLayerTreeLayer *nodeLayer = QgsLayerTree::toLayer( node );
83 
84  QString text;
85  if ( nodeLegendStyle( nodeLayer ) != QgsLegendStyle::Hidden )
86  {
87  const QModelIndex idx = mLegendModel->node2index( nodeLayer );
88  text = mLegendModel->data( idx, Qt::DisplayRole ).toString();
89  }
90 
91  QList<QgsLayerTreeModelLegendNode *> legendNodes = mLegendModel->layerLegendNodes( nodeLayer );
92 
93  if ( legendNodes.isEmpty() && mLegendModel->legendFilterMapSettings() )
94  continue;
95 
96  if ( legendNodes.count() == 1 )
97  {
98  legendNodes.at( 0 )->exportToJson( mSettings, context, group );
99  nodes.append( group );
100  }
101  else if ( legendNodes.count() > 1 )
102  {
103  QJsonArray symbols;
104  for ( int j = 0; j < legendNodes.count(); j++ )
105  {
106  QgsLayerTreeModelLegendNode *legendNode = legendNodes.at( j );
107  QJsonObject symbol;
108  legendNode->exportToJson( mSettings, context, symbol );
109  symbols.append( symbol );
110  }
111  group[ QStringLiteral( "title" ) ] = text;
112  group[ QStringLiteral( "symbols" ) ] = symbols;
113  nodes.append( group );
114  }
115  }
116  }
117 
118  json[QStringLiteral( "nodes" )] = nodes;
119 }

126 QSizeF QgsLegendRenderer::paintAndDetermineSizeInternal( QgsRenderContext *context, QPainter *painter )
127 {
128  QSizeF size( 0, 0 );
129  QgsLayerTreeGroup *rootGroup = mLegendModel->rootGroup();
130  if ( !rootGroup )
131  return size;
132 
133  QList<LegendComponentGroup> componentGroups = createComponentGroupList( rootGroup, mSettings.splitLayer() );
134 
135  setColumns( componentGroups );
136 
137  QMap< int, double > maxColumnWidths;
138  qreal maxEqualColumnWidth = 0;
139  // another iteration -- this one is required to calculate the maximum item width for each
140  // column. Unfortunately, we can't trust the component group widths at this stage, as they are minimal widths
141  // only. When actually rendering a symbol node, the text is aligned according to the WIDEST
142  // symbol in a column. So that means we can't possibly determine the exact size of legend components
143  // until now. BUUUUUUUUUUUUT. Because everything sucks, we can't even start the actual render of items
144  // at the same time we calculate this -- legend items REQUIRE the REAL width of the columns in order to
145  // correctly align right or center-aligned symbols/text. Bah -- A triple iteration it is!
146  // temporarily remove painter from context -- we don't need to actually draw anything yet. But we DO need
147  // to send the full render context so that an expression context is available during the size calculation
148  QPainter *prevPainter = context ? context->painter() : nullptr;
149  if ( context )
150  context->setPainter( nullptr );
151  for ( const LegendComponentGroup &group : qgis::as_const( componentGroups ) )
152  {
153  const QSizeF actualSize = drawGroup( group, context, ColumnContext() );
154  maxEqualColumnWidth = std::max( actualSize.width(), maxEqualColumnWidth );
155  maxColumnWidths[ group.column ] = std::max( actualSize.width(), maxColumnWidths.value( group.column, 0 ) );
156  }
157  if ( context )
158  context->setPainter( prevPainter );
159 
160  if ( mSettings.columnCount() < 2 )
161  {
162  // single column - use the full available width
163  maxEqualColumnWidth = std::max( maxEqualColumnWidth, mLegendSize.width() - 2 * mSettings.boxSpace() );
164  maxColumnWidths[ 0 ] = maxEqualColumnWidth;
165  }
166 
167  //calculate size of title
168  QSizeF titleSize = drawTitle();
169  //add title margin to size of title text
170  titleSize.rwidth() += mSettings.boxSpace() * 2.0;
171  double columnTop = mSettings.boxSpace() + titleSize.height() + mSettings.style( QgsLegendStyle::Title ).margin( QgsLegendStyle::Bottom );
172 
173  bool firstInColumn = true;
174  double columnMaxHeight = 0;
175  qreal columnWidth = 0;
176  int column = -1;
177  ColumnContext columnContext;
178  columnContext.left = mSettings.boxSpace();
179  columnContext.right = std::max( mLegendSize.width() - mSettings.boxSpace(), mSettings.boxSpace() );
180  double currentY = columnTop;
181 
182  for ( const LegendComponentGroup &group : qgis::as_const( componentGroups ) )
183  {
184  if ( group.column > column )
185  {
186  // Switch to next column
187  columnContext.left = group.column > 0 ? columnContext.right + mSettings.columnSpace() : mSettings.boxSpace();
188  columnWidth = mSettings.equalColumnWidth() ? maxEqualColumnWidth : maxColumnWidths.value( group.column );
189  columnContext.right = columnContext.left + columnWidth;
190  currentY = columnTop;
191  column++;
192  firstInColumn = true;
193  }
194  if ( !firstInColumn )
195  {
196  currentY += spaceAboveGroup( group );
197  }
198 
199  if ( context )
200  drawGroup( group, context, columnContext, currentY );
201  else if ( painter )
202  drawGroup( group, columnContext, painter, currentY );
203 
204  currentY += group.size.height();
205  columnMaxHeight = std::max( currentY - columnTop, columnMaxHeight );
206 
207  firstInColumn = false;
208  }
209  const double totalWidth = columnContext.right + mSettings.boxSpace();
210 
211  size.rheight() = columnTop + columnMaxHeight + mSettings.boxSpace();
212  size.rwidth() = totalWidth;
213  if ( !mSettings.title().isEmpty() )
214  {
215  size.rwidth() = std::max( titleSize.width(), size.width() );
216  }
217 
218  // override the size if it was set by the user
219  if ( mLegendSize.isValid() )
220  {
221  qreal w = std::max( size.width(), mLegendSize.width() );
222  qreal h = std::max( size.height(), mLegendSize.height() );
223  size = QSizeF( w, h );
224  }
225 
226  // Now we have set the correct total item width and can draw the title centered
227  if ( !mSettings.title().isEmpty() )
228  {
229  if ( context )
230  drawTitle( context, mSettings.boxSpace(), mSettings.titleAlignment(), size.width() );
231  else
232  drawTitle( painter, mSettings.boxSpace(), mSettings.titleAlignment(), size.width() );
233  }
234 
235  return size;
236 }

238 void QgsLegendRenderer::widthAndOffsetForTitleText( const Qt::AlignmentFlag halignment, const double legendWidth, double &textBoxWidth, double &textBoxLeft )
239 {
240  switch ( halignment )
241  {
242  default:
243  textBoxLeft = mSettings.boxSpace();
244  textBoxWidth = legendWidth - 2 * mSettings.boxSpace();
245  break;
246 
247  case Qt::AlignHCenter:
248  {
249  // not sure on this logic, I just moved it -- don't blame me for it being totally obscure!
250  const double centerX = legendWidth / 2;
251  textBoxWidth = ( std::min( static_cast< double >( centerX ), legendWidth - centerX ) - mSettings.boxSpace() ) * 2.0;
252  textBoxLeft = centerX - textBoxWidth / 2.;
253  break;
254  }
255  }

258 QList<QgsLegendRenderer::LegendComponentGroup> QgsLegendRenderer::createComponentGroupList( QgsLayerTreeGroup *parentGroup, bool splitLayer )
259 {
260  QList<LegendComponentGroup> componentGroups;
261 
262  if ( !parentGroup )
263  return componentGroups;
264 
265  const QList<QgsLayerTreeNode *> childNodes = parentGroup->children();
266  for ( QgsLayerTreeNode *node : childNodes )
267  {
268  if ( QgsLayerTree::isGroup( node ) )
269  {
270  QgsLayerTreeGroup *nodeGroup = QgsLayerTree::toGroup( node );
271 
272  // Group subitems
273  QList<LegendComponentGroup> subgroups = createComponentGroupList( nodeGroup, splitLayer );
274  bool hasSubItems = !subgroups.empty();
275 
276  if ( nodeLegendStyle( nodeGroup ) != QgsLegendStyle::Hidden )
277  {
278  LegendComponent component;
279  component.item = node;
280  component.size = drawGroupTitle( nodeGroup );
281 
282  if ( !subgroups.isEmpty() )
283  {
284  // Add internal space between this group title and the next component
285  subgroups[0].size.rheight() += spaceAboveGroup( subgroups[0] );
286  // Prepend this group title to the first group
287  subgroups[0].components.prepend( component );
288  subgroups[0].size.rheight() += component.size.height();
289  subgroups[0].size.rwidth() = std::max( component.size.width(), subgroups[0].size.width() );
290  }
291  else
292  {
293  // no subitems, create new group
294  LegendComponentGroup group;
295  group.components.append( component );
296  group.size.rwidth() += component.size.width();
297  group.size.rheight() += component.size.height();
298  group.size.rwidth() = std::max( component.size.width(), group.size.width() );
299  subgroups.append( group );
300  }
301  }
302 
303  if ( hasSubItems ) //leave away groups without content
304  {
305  componentGroups.append( subgroups );
306  }
307 
308  }
309  else if ( QgsLayerTree::isLayer( node ) )
310  {
311  QgsLayerTreeLayer *nodeLayer = QgsLayerTree::toLayer( node );
312 
313  LegendComponentGroup group;
314 
315  if ( nodeLegendStyle( nodeLayer ) != QgsLegendStyle::Hidden )
316  {
317  LegendComponent component;
318  component.item = node;
319  component.size = drawLayerTitle( nodeLayer );
320  group.components.append( component );
321  group.size.rwidth() = component.size.width();
322  group.size.rheight() = component.size.height();
323  }
324 
325  QList<QgsLayerTreeModelLegendNode *> legendNodes = mLegendModel->layerLegendNodes( nodeLayer );
326 
327  // workaround for the issue that "filtering by map" does not remove layer nodes that have no symbols present
328  // on the map. We explicitly skip such layers here. In future ideally that should be handled directly
329  // in the layer tree model
330  if ( legendNodes.isEmpty() && mLegendModel->legendFilterMapSettings() )
331  continue;
332 
333  QList<LegendComponentGroup> layerGroups;
334  layerGroups.reserve( legendNodes.count() );
335 
336  for ( int j = 0; j < legendNodes.count(); j++ )
337  {
338  QgsLayerTreeModelLegendNode *legendNode = legendNodes.at( j );
339 
340  LegendComponent symbolComponent = drawSymbolItem( legendNode );
341 
342  if ( !mSettings.splitLayer() || j == 0 )
343  {
344  // append to layer group
345  // the width is not correct at this moment, we must align all symbol labels
346  group.size.rwidth() = std::max( symbolComponent.size.width(), group.size.width() );
347  // Add symbol space only if there is already title or another item above
348  if ( !group.components.isEmpty() )
349  {
350  // TODO: for now we keep Symbol and SymbolLabel Top margin in sync
351  group.size.rheight() += mSettings.style( QgsLegendStyle::Symbol ).margin( QgsLegendStyle::Top );
352  }
353  group.size.rheight() += symbolComponent.size.height();
354  group.components.append( symbolComponent );
355  }
356  else
357  {
358  LegendComponentGroup symbolGroup;
359  symbolGroup.components.append( symbolComponent );
360  symbolGroup.size.rwidth() = symbolComponent.size.width();
361  symbolGroup.size.rheight() = symbolComponent.size.height();
362  layerGroups.append( symbolGroup );
363  }
364  }
365  layerGroups.prepend( group );
366  componentGroups.append( layerGroups );
367  }
368  }
369 
370  return componentGroups;
371 }

374 void QgsLegendRenderer::setColumns( QList<LegendComponentGroup> &componentGroups )
375 {
376  if ( mSettings.columnCount() == 0 )
377  return;
378 
379  // Divide groups to columns
380  double totalHeight = 0;
381  qreal maxGroupHeight = 0;
382  for ( const LegendComponentGroup &group : qgis::as_const( componentGroups ) )
383  {
384  totalHeight += spaceAboveGroup( group );
385  totalHeight += group.size.height();
386  maxGroupHeight = std::max( group.size.height(), maxGroupHeight );
387  }
388 
389  // We know height of each group and we have to split them into columns
390  // minimizing max column height. It is sort of bin packing problem, NP-hard.
391  // We are using simple heuristic, brute fore appeared to be to slow,
392  // the number of combinations is N = n!/(k!*(n-k)!) where n = groupCount-1
393  // and k = columnsCount-1
394  double maxColumnHeight = 0;
395  int currentColumn = 0;
396  int currentColumnGroupCount = 0; // number of groups in current column
397  double currentColumnHeight = 0;
398  double closedColumnsHeight = 0;
399 
400  for ( int i = 0; i < componentGroups.size(); i++ )
401  {
402  // Recalc average height for remaining columns including current
403  double avgColumnHeight = ( totalHeight - closedColumnsHeight ) / ( mSettings.columnCount() - currentColumn );
404 
405  LegendComponentGroup group = componentGroups.at( i );
406  double currentHeight = currentColumnHeight;
407  if ( currentColumnGroupCount > 0 )
408  currentHeight += spaceAboveGroup( group );
409  currentHeight += group.size.height();
410 
411  bool canCreateNewColumn = ( currentColumnGroupCount > 0 ) // do not leave empty column
412  && ( currentColumn < mSettings.columnCount() - 1 ); // must not exceed max number of columns
413 
414  bool shouldCreateNewColumn = ( currentHeight - avgColumnHeight ) > group.size.height() / 2 // center of current group is over average height
415  && currentColumnGroupCount > 0 // do not leave empty column
416  && currentHeight > maxGroupHeight // no sense to make smaller columns than max group height
417  && currentHeight > maxColumnHeight; // no sense to make smaller columns than max column already created
418 
419  // also should create a new column if the number of items left < number of columns left
420  // in this case we should spread the remaining items out over the remaining columns
421  shouldCreateNewColumn |= ( componentGroups.size() - i < mSettings.columnCount() - currentColumn );
422 
423  if ( canCreateNewColumn && shouldCreateNewColumn )
424  {
425  // New column
426  currentColumn++;
427  currentColumnGroupCount = 0;
428  closedColumnsHeight += currentColumnHeight;
429  currentColumnHeight = group.size.height();
430  }
431  else
432  {
433  currentColumnHeight = currentHeight;
434  }
435  componentGroups[i].column = currentColumn;
436  currentColumnGroupCount++;
437  maxColumnHeight = std::max( currentColumnHeight, maxColumnHeight );
438  }
439 
440  // Align labels of symbols for each layr/column to the same labelXOffset
441  QMap<QString, qreal> maxSymbolWidth;
442  for ( int i = 0; i < componentGroups.size(); i++ )
443  {
444  LegendComponentGroup &group = componentGroups[i];
445  for ( int j = 0; j < group.components.size(); j++ )
446  {
447  if ( QgsLayerTreeModelLegendNode *legendNode = qobject_cast<QgsLayerTreeModelLegendNode *>( group.components.at( j ).item ) )
448  {
449  QString key = QStringLiteral( "%1-%2" ).arg( reinterpret_cast< qulonglong >( legendNode->layerNode() ) ).arg( group.column );
450  maxSymbolWidth[key] = std::max( group.components.at( j ).symbolSize.width(), maxSymbolWidth[key] );
451  }
452  }
453  }
454  for ( int i = 0; i < componentGroups.size(); i++ )
455  {
456  LegendComponentGroup &group = componentGroups[i];
457  for ( int j = 0; j < group.components.size(); j++ )
458  {
459  if ( QgsLayerTreeModelLegendNode *legendNode = qobject_cast<QgsLayerTreeModelLegendNode *>( group.components.at( j ).item ) )
460  {
461  QString key = QStringLiteral( "%1-%2" ).arg( reinterpret_cast< qulonglong >( legendNode->layerNode() ) ).arg( group.column );
462  double space = mSettings.style( QgsLegendStyle::Symbol ).margin( QgsLegendStyle::Right ) +
463  mSettings.style( QgsLegendStyle::SymbolLabel ).margin( QgsLegendStyle::Left );
464  group.components[j].labelXOffset = maxSymbolWidth[key] + space;
465  group.components[j].maxSiblingSymbolWidth = maxSymbolWidth[key];
466  group.components[j].size.rwidth() = maxSymbolWidth[key] + space + group.components.at( j ).labelSize.width();
467  }
468  }
469  }
470 }

477 QSizeF QgsLegendRenderer::drawTitleInternal( QgsRenderContext *context, QPainter *painter, const double top, Qt::AlignmentFlag halignment, double legendWidth )
478 {
479  QSizeF size( 0, 0 );
480  if ( mSettings.title().isEmpty() )
481  {
482  return size;
483  }
484 
485  QStringList lines = mSettings.splitStringForWrapping( mSettings.title() );
486  double y = top;
487 
488  if ( context && context->painter() )
489  {
490  context->painter()->setPen( mSettings.fontColor() );
491  }
492  else if ( painter )
493  {
494  painter->setPen( mSettings.fontColor() );
495  }
496 
497  //calculate width and left pos of rectangle to draw text into
498  double textBoxWidth;
499  double textBoxLeft;
500  widthAndOffsetForTitleText( halignment, legendWidth, textBoxWidth, textBoxLeft );
501 
502  QFont titleFont = mSettings.style( QgsLegendStyle::Title ).font();
503 
504  for ( QStringList::Iterator titlePart = lines.begin(); titlePart != lines.end(); ++titlePart )
505  {
506  //last word is not drawn if rectangle width is exactly text width, so add 1
507  //TODO - correctly calculate size of italicized text, since QFontMetrics does not
508  qreal width = mSettings.textWidthMillimeters( titleFont, *titlePart ) + 1;
509  qreal height = mSettings.fontAscentMillimeters( titleFont ) + mSettings.fontDescentMillimeters( titleFont );
510 
511  QRectF r( textBoxLeft, y, textBoxWidth, height );
512 
513  if ( context && context->painter() )
514  {
515  mSettings.drawText( context->painter(), r, *titlePart, titleFont, halignment, Qt::AlignVCenter, Qt::TextDontClip );
516  }
517  else if ( painter )
518  {
519  mSettings.drawText( painter, r, *titlePart, titleFont, halignment, Qt::AlignVCenter, Qt::TextDontClip );
520  }
521 
522  //update max width of title
523  size.rwidth() = std::max( width, size.rwidth() );
524 
525  y += height;
526  if ( titlePart != ( lines.end() - 1 ) )
527  {
528  y += mSettings.lineSpacing();
529  }
530  }
531  size.rheight() = y - top;
532 
533  return size;
534 }

537 double QgsLegendRenderer::spaceAboveGroup( const LegendComponentGroup &group )
538 {
539  if ( group.components.isEmpty() ) return 0;
540 
541  LegendComponent component = group.components.first();
542 
543  if ( QgsLayerTreeGroup *nodeGroup = qobject_cast<QgsLayerTreeGroup *>( component.item ) )
544  {
545  return mSettings.style( nodeLegendStyle( nodeGroup ) ).margin( QgsLegendStyle::Top );
546  }
547  else if ( QgsLayerTreeLayer *nodeLayer = qobject_cast<QgsLayerTreeLayer *>( component.item ) )
548  {
549  return mSettings.style( nodeLegendStyle( nodeLayer ) ).margin( QgsLegendStyle::Top );
550  }
551  else if ( qobject_cast<QgsLayerTreeModelLegendNode *>( component.item ) )
552  {
553  // TODO: use Symbol or SymbolLabel Top margin
554  return mSettings.style( QgsLegendStyle::Symbol ).margin( QgsLegendStyle::Top );
555  }
556 
557  return 0;
558 }

565 QSizeF QgsLegendRenderer::drawGroupInternal( const LegendComponentGroup &group, QgsRenderContext *context, ColumnContext columnContext, QPainter *painter, const double top )
566 {
567  bool first = true;
568  QSizeF size = QSizeF( group.size );
569  double currentY = top;
570  for ( const LegendComponent &component : qgis::as_const( group.components ) )
571  {
572  if ( QgsLayerTreeGroup *groupItem = qobject_cast<QgsLayerTreeGroup *>( component.item ) )
573  {
574  QgsLegendStyle::Style s = nodeLegendStyle( groupItem );
575  if ( s != QgsLegendStyle::Hidden )
576  {
577  if ( !first )
578  {
579  currentY += mSettings.style( s ).margin( QgsLegendStyle::Top );
580  }
581  QSizeF groupSize;
582  if ( context )
583  groupSize = drawGroupTitle( groupItem, context, columnContext, currentY );
584  else
585  groupSize = drawGroupTitle( groupItem, columnContext, painter, currentY );
586  size.rwidth() = std::max( groupSize.width(), size.width() );
587  }
588  }
589  else if ( QgsLayerTreeLayer *layerItem = qobject_cast<QgsLayerTreeLayer *>( component.item ) )
590  {
591  QgsLegendStyle::Style s = nodeLegendStyle( layerItem );
592  if ( s != QgsLegendStyle::Hidden )
593  {
594  if ( !first )
595  {
596  currentY += mSettings.style( s ).margin( QgsLegendStyle::Top );
597  }
598  QSizeF subGroupSize;
599  if ( context )
600  subGroupSize = drawLayerTitle( layerItem, context, columnContext, currentY );
601  else
602  subGroupSize = drawLayerTitle( layerItem, columnContext, painter, currentY );
603  size.rwidth() = std::max( subGroupSize.width(), size.width() );
604  }
605  }
606  else if ( QgsLayerTreeModelLegendNode *legendNode = qobject_cast<QgsLayerTreeModelLegendNode *>( component.item ) )
607  {
608  if ( !first )
609  {
610  currentY += mSettings.style( QgsLegendStyle::Symbol ).margin( QgsLegendStyle::Top );
611  }
612 
613  LegendComponent symbolComponent = context ? drawSymbolItem( legendNode, context, columnContext, currentY, component.maxSiblingSymbolWidth )
614  : drawSymbolItem( legendNode, columnContext, painter, currentY, component.maxSiblingSymbolWidth );
615  // expand width, it may be wider because of label offsets
616  size.rwidth() = std::max( symbolComponent.size.width(), size.width() );
617  }
618  currentY += component.size.height();
619  first = false;
620  }
621  return size;

682 QSizeF QgsLegendRenderer::drawLayerTitleInternal( QgsLayerTreeLayer *nodeLayer, ColumnContext columnContext, QgsRenderContext *context, QPainter *painter, const double top )
683 {
684  QSizeF size( 0, 0 );
685  QModelIndex idx = mLegendModel->node2index( nodeLayer );
686 
687  //Let the user omit the layer title item by having an empty layer title string
688  if ( mLegendModel->data( idx, Qt::DisplayRole ).toString().isEmpty() )
689  return size;
690 
691  double y = top;
692 
693  if ( context && context->painter() )
694  context->painter()->setPen( mSettings.layerFontColor() );
695  else if ( painter )
696  painter->setPen( mSettings.layerFontColor() );
697 
698  QFont layerFont = mSettings.style( nodeLegendStyle( nodeLayer ) ).font();
699 
700  QgsExpressionContextScope *layerScope = nullptr;
701  if ( context && nodeLayer->layer() )
702  {
703  layerScope = QgsExpressionContextUtils::layerScope( nodeLayer->layer() );
704  context->expressionContext().appendScope( layerScope );
705  }
706 
707  QgsExpressionContext tempContext;
708 
709  const QStringList lines = mSettings.evaluateItemText( mLegendModel->data( idx, Qt::DisplayRole ).toString(),
710  context ? context->expressionContext() : tempContext );
711  int i = 0;
712  for ( QStringList::ConstIterator layerItemPart = lines.constBegin(); layerItemPart != lines.constEnd(); ++layerItemPart )
713  {
714  y += mSettings.fontAscentMillimeters( layerFont );
715  if ( QPainter *destPainter = context && context->painter() ? context->painter() : painter )
716  {
717  double x = columnContext.left;
718  if ( mSettings.style( nodeLegendStyle( nodeLayer ) ).alignment() != Qt::AlignLeft )
719  {
720  const double labelWidth = mSettings.textWidthMillimeters( layerFont, *layerItemPart );
721  if ( mSettings.style( nodeLegendStyle( nodeLayer ) ).alignment() == Qt::AlignRight )
722  x = columnContext.right - labelWidth;
723  else if ( mSettings.style( nodeLegendStyle( nodeLayer ) ).alignment() == Qt::AlignHCenter )
724  x = columnContext.left + ( columnContext.right - columnContext.left - labelWidth ) / 2;
725  }
726  mSettings.drawText( destPainter, x, y, *layerItemPart, layerFont );
727  }
728  qreal width = mSettings.textWidthMillimeters( layerFont, *layerItemPart );
729  size.rwidth() = std::max( width, size.width() );
730  if ( layerItemPart != ( lines.end() - 1 ) )
731  {
732  y += mSettings.lineSpacing();
733  }
734  i++;
735  }
736  size.rheight() = y - top;
737  size.rheight() += mSettings.style( nodeLegendStyle( nodeLayer ) ).margin( QgsLegendStyle::Side::Bottom );
738 
739  if ( layerScope )
740  delete context->expressionContext().popScope();
741 
742  return size;
743 }
