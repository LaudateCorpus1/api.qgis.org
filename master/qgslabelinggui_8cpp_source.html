60 static bool _initCalloutWidgetFunction( const QString &name, QgsCalloutWidgetFunc f )
61 {
62  QgsCalloutRegistry *registry = QgsApplication::calloutRegistry();
63 
64  QgsCalloutAbstractMetadata *abstractMetadata = registry->calloutMetadata( name );
65  if ( !abstractMetadata )
66  {
67  QgsDebugMsg( QStringLiteral( "Failed to find callout entry in registry: %1" ).arg( name ) );
68  return false;
69  }
70  QgsCalloutMetadata *metadata = dynamic_cast<QgsCalloutMetadata *>( abstractMetadata );
71  if ( !metadata )
72  {
73  QgsDebugMsg( QStringLiteral( "Failed to cast callout's metadata: " ) .arg( name ) );
74  return false;
75  }
76  metadata->setWidgetFunction( f );
77  return true;
78 }

86 void QgsLabelingGui::updateCalloutWidget( QgsCallout *callout )
87 {
88  if ( !callout )
89  {
90  mCalloutStackedWidget->setCurrentWidget( pageDummy );
91  return;
92  }
93 
94  if ( mCalloutStackedWidget->currentWidget() != pageDummy )
95  {
96  // stop updating from the original widget
97  if ( QgsCalloutWidget *pew = qobject_cast< QgsCalloutWidget * >( mCalloutStackedWidget->currentWidget() ) )
98  disconnect( pew, &QgsCalloutWidget::changed, this, &QgsLabelingGui::updatePreview );
99  }
100 
101  QgsCalloutRegistry *registry = QgsApplication::calloutRegistry();
102  if ( QgsCalloutAbstractMetadata *am = registry->calloutMetadata( callout->type() ) )
103  {
104  if ( QgsCalloutWidget *w = am->createCalloutWidget( mLayer ) )
105  {
106 
107  QgsWkbTypes::GeometryType geometryType = mGeomType;
108  if ( mGeometryGeneratorGroupBox->isChecked() )
109  geometryType = mGeometryGeneratorType->currentData().value<QgsWkbTypes::GeometryType>();
110  else if ( mLayer )
111  geometryType = mLayer->geometryType();
112  w->setGeometryType( geometryType );
113  w->setCallout( callout );
114 
115  w->setContext( context() );
116  mCalloutStackedWidget->addWidget( w );
117  mCalloutStackedWidget->setCurrentWidget( w );
118  // start receiving updates from widget
119  connect( w, &QgsCalloutWidget::changed, this, &QgsLabelingGui::updatePreview );
120  return;
121  }
122  }
123  // When anything is not right
124  mCalloutStackedWidget->setCurrentWidget( pageDummy );
125 }

776 void QgsLabelingGui::validateGeometryGeneratorExpression()
777 {
778  bool valid = true;
779 
780  if ( mGeometryGeneratorGroupBox->isChecked() )
781  {
782  if ( !mPreviewFeature.isValid() && mLayer )
783  mLayer->getFeatures( QgsFeatureRequest().setLimit( 1 ) ).nextFeature( mPreviewFeature );
784 
785  QgsExpression expression( mGeometryGenerator->text() );
786  QgsExpressionContext context = createExpressionContext();
787  context.setFeature( mPreviewFeature );
788 
789  expression.prepare( &context );
790

795  }
796  else
797  {
798  const QVariant result = expression.evaluate( &context );
799  const QgsGeometry geometry = result.value<QgsGeometry>();

801  if ( geometry.isNull() )

809  tr( "Result of the expression does not match configured geometry type." ),
810  tr( "Change to %1" ).arg( QgsWkbTypes::geometryDisplayString( geometry.type() ) ) ) );
811  valid = false;
812  }
813  }
814  }
815 
816  // The collapsible groupbox internally changes the visibility of this
817  // Work around by setting the visibility deferred in the next event loop cycle.
818  QTimer *timer = new QTimer();
819  connect( timer, &QTimer::timeout, this, [this, valid]()
820  {
