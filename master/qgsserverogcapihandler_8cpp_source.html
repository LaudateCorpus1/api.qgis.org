40 QVariantMap QgsServerOgcApiHandler::values( const QgsServerApiContext &context ) const
41 {
42  QVariantMap result ;
43  QVariantList positional;
44  const auto constParameters { parameters( context ) };
45  for ( const auto &p : constParameters )
46  {
47  // value() calls the validators and throw an exception if validation fails
48  result[p.name()] = p.value( context );
49  }
50  const auto match { path().match( context.request()->url().toString() ) };
51  if ( match.hasMatch() )
52  {
53  const auto constNamed { path().namedCaptureGroups() };
54  // Get named path parameters
55  for ( const auto &name : constNamed )
56  {
57  if ( ! name.isEmpty() )
58  result[name] = QUrlQuery( match.captured( name ) ).toString() ;
59  }
60  }
61  return result;

129 std::string QgsServerOgcApiHandler::href( const QgsServerApiContext &context, const QString &extraPath, const QString &extension ) const
130 {
131  QUrl url { context.request()->url() };
132  QString urlBasePath { context.matchedPath() };
133  const auto match { path().match( url.path() ) };
134  if ( match.captured().count() > 0 )
135  {
136  url.setPath( urlBasePath + match.captured( 0 ) );
137  }
138  else
139  {
140  url.setPath( urlBasePath );
141  }
142 
143  // Remove any existing extension
144  const auto suffixLength { QFileInfo( url.path() ).completeSuffix().length() };
145  if ( suffixLength > 0 )
146  {
147  auto path {url.path()};
148  path.truncate( path.length() - ( suffixLength + 1 ) );
149  url.setPath( path );
150  }
151 
152  // Add extra path
153  url.setPath( url.path() + extraPath );
154 
155  // (re-)add extension
156  // JSON is the default anyway, we don'n need to add it
157  if ( ! extension.isEmpty() )
158  {
159  // Remove trailing slashes if any.
160  QString path { url.path() };
161  while ( path.endsWith( '/' ) )
162  {
163  path.chop( 1 );
164  }
165  url.setPath( path + '.' + extension );
166  }
167  return QgsServerOgcApi::sanitizeUrl( url ).toString( QUrl::FullyEncoded ).toStdString();
168

256 void QgsServerOgcApiHandler::htmlDump( const json &data, const QgsServerApiContext &context ) const
257 {
258  context.response()->setHeader( QStringLiteral( "Content-Type" ), QStringLiteral( "text/html" ) );
259  auto path { templatePath( context ) };
260  if ( ! QFile::exists( path ) )
261  {
262  QgsMessageLog::logMessage( QStringLiteral( "Template not found error: %1" ).arg( path ), QStringLiteral( "Server" ), Qgis::Critical );
263  throw QgsServerApiBadRequestException( QStringLiteral( "Template not found: %1" ).arg( QFileInfo( path ).fileName() ) );
264  }
265 
266  QFile f( path );
267  if ( ! f.open( QFile::ReadOnly | QFile::Text ) )
268  {
269  QgsMessageLog::logMessage( QStringLiteral( "Could not open template file: %1" ).arg( path ), QStringLiteral( "Server" ), Qgis::Critical );
270  throw QgsServerApiInternalServerError( QStringLiteral( "Could not open template file: %1" ).arg( QFileInfo( path ).fileName() ) );
271  }
272 
273  try
274  {
275  // Get the template directory and the file name
276  QFileInfo pathInfo { path };
277  Environment env { ( pathInfo.dir().path() + QDir::separator() ).toStdString() };
278 
279  // For template debugging:
280  env.add_callback( "json_dump", 0, [ = ]( Arguments & )
281  {
282  return data.dump();
283  } );
284 
285  // Path manipulation: appends a directory path to the current url
286  env.add_callback( "path_append", 1, [ = ]( Arguments & args )
287  {
288  auto url { context.request()->url() };
289  QFileInfo fi{ url.path() };
290  auto suffix { fi.suffix() };
291  auto fName { fi.filePath()};
292  fName.chop( suffix.length() + 1 );
293  fName += '/' + QString::number( args.at( 0 )->get<QgsFeatureId>( ) );
294  if ( !suffix.isEmpty() )
295  {
296  fName += '.' + suffix;
297  }
298  fi.setFile( fName );
299  url.setPath( fi.filePath() );
300  return url.toString().toStdString();
301  } );
302 
303  // Path manipulation: removes the specified number of directory components from the current url path
304  env.add_callback( "path_chomp", 1, [ = ]( Arguments & args )
305  {
306  QUrl url { QString::fromStdString( args.at( 0 )->get<std::string>( ) ) };
307  QFileInfo fi{ url.path() };
308  auto suffix { fi.suffix() };
309  auto fName { fi.filePath()};
310  fName.chop( suffix.length() + 1 );
311  // Chomp last segment
312  fName = fName.replace( QRegularExpression( R"raw(\/[^/]+$)raw" ), QString() );
313  if ( !suffix.isEmpty() )
314  {
315  fName += '.' + suffix;
316  }
317  fi.setFile( fName );
318  url.setPath( fi.filePath() );
319  return url.toString().toStdString();
320  } );
321 
322  // Returns filtered links from a link list
323  // links_filter( <links>, <key>, <value> )
324  env.add_callback( "links_filter", 3, [ = ]( Arguments & args )
325  {
326  json links = args.at( 0 )->get<json>( );
327  if ( ! links.is_array() )
328  {
329  links = json::array();
330  }
331  std::string key { args.at( 1 )->get<std::string>( ) };
332  std::string value { args.at( 2 )->get<std::string>( ) };
333  json result = json::array();
334  for ( const auto &l : links )
335  {
336  if ( l[key] == value )
337  {
338  result.push_back( l );
339  }
340  }
341  return result;
342  } );
343 
344  // Returns a short name from content types
345  env.add_callback( "content_type_name", 1, [ = ]( Arguments & args )
346  {
347  const QgsServerOgcApi::ContentType ct { QgsServerOgcApi::contenTypeFromExtension( args.at( 0 )->get<std::string>( ) ) };
348  return QgsServerOgcApi::contentTypeToStdString( ct );
349  } );
350 
351 
352  // Static: returns the full URL to the specified static <path>
353  env.add_callback( "static", 1, [ = ]( Arguments & args )
354  {
355  auto asset( args.at( 0 )->get<std::string>( ) );
356  return context.matchedPath().toStdString() + "/static/" + asset;
357  } );
358 
359  context.response()->write( env.render_file( pathInfo.fileName().toStdString(), data ) );
360  }
361  catch ( std::exception &e )
362  {
363  QgsMessageLog::logMessage( QStringLiteral( "Error parsing template file: %1 - %2" ).arg( path, e.what() ), QStringLiteral( "Server" ), Qgis::Critical );
364  throw QgsServerApiInternalServerError( QStringLiteral( "Error parsing template file: %1" ).arg( e.what() ) );
365  }
366 }

368 QgsServerOgcApi::ContentType QgsServerOgcApiHandler::contentTypeFromRequest( const QgsServerRequest *request ) const
369 {
370  // Fallback to default
371  QgsServerOgcApi::ContentType result { defaultContentType() };
372  bool found { false };
373  // First file extension ...
374  const QString extension { QFileInfo( request->url().path() ).completeSuffix().toUpper() };
375  if ( ! extension.isEmpty() )
376  {
377  static QMetaEnum metaEnum { QMetaEnum::fromType<QgsServerOgcApi::ContentType>() };
378  bool ok { false };
379  const int ct { metaEnum.keyToValue( extension.toLocal8Bit().constData(), &ok ) };
380  if ( ok )
381  {
382  result = static_cast<QgsServerOgcApi::ContentType>( ct );
383  found = true;
384  }
385  else
386  {
387  QgsMessageLog::logMessage( QStringLiteral( "The client requested an unsupported extension: %1" ).arg( extension ), QStringLiteral( "Server" ), Qgis::Warning );
388  }
389  }
390  // ... then "Accept"
391  const QString accept { request->header( QStringLiteral( "Accept" ) ) };
392  if ( ! found && ! accept.isEmpty() )
393  {
394  const QString ctFromAccept { contentTypeForAccept( accept ) };
395  if ( ! ctFromAccept.isEmpty() )
396  {
397  result = QgsServerOgcApi::contentTypeMimes().key( ctFromAccept );
398  found = true;
399  }
400  else
401  {
402  QgsMessageLog::logMessage( QStringLiteral( "The client requested an unsupported content type in Accept header: %1" ).arg( accept ), QStringLiteral( "Server" ), Qgis::Warning );
403  }
404  }
405  // Validation: check if the requested content type (or an alias) is supported by the handler
406  if ( ! contentTypes().contains( result ) )
407  {
408  // Check aliases
409  bool found { false };
410  if ( QgsServerOgcApi::contentTypeAliases().keys().contains( result ) )
411  {
412  const QList<QgsServerOgcApi::ContentType> constCt { contentTypes() };
413  for ( const auto &ct : constCt )
414  {
415  if ( QgsServerOgcApi::contentTypeAliases()[result].contains( ct ) )
416  {
417  result = ct;
418  found = true;
419  break;
420  }
421  }
422  }
423 
424  if ( ! found )
425  {
426  QgsMessageLog::logMessage( QStringLiteral( "Unsupported Content-Type: %1" ).arg( QgsServerOgcApi::contentTypeToString( result ) ), QStringLiteral( "Server" ), Qgis::Info );
427  throw QgsServerApiBadRequestException( QStringLiteral( "Unsupported Content-Type: %1" ).arg( QgsServerOgcApi::contentTypeToString( result ) ) );
428  }
429  }
430  return result;
431 }

433 QString QgsServerOgcApiHandler::parentLink( const QUrl &url, int levels )
434 {
435  QString path { url.path() };
436  const QFileInfo fi { path };
437  const QString suffix { fi.suffix() };
438  if ( ! suffix.isEmpty() )
439  {
440  path.chop( suffix.length() + 1 );
441  }
442  while ( path.endsWith( '/' ) )
443  {
444  path.chop( 1 );
445  }
446  QRegularExpression re( R"raw(\/[^/]+$)raw" );
447  for ( int i = 0; i < levels ; i++ )
448  {
449  path = path.replace( re, QString() );
450  }
451  QUrl result( url );
452  QList<QPair<QString, QString> > qi;
453  const auto constItems { result.queryItems( ) };
454  for ( const auto &i : constItems )
455  {
456  if ( i.first.compare( QStringLiteral( "MAP" ), Qt::CaseSensitivity::CaseInsensitive ) == 0 )
457  {
458  qi.push_back( i );
459  }
460  }
461  result.setQueryItems( qi );
462  result.setPath( path );
463  return result.toString();
464 }
