140 bool QgsCoordinateTransformPrivate::initialize()
141 {
142  invalidate();
143  if ( !mSourceCRS.isValid() )
144  {
145  // Pass through with no projection since we have no idea what the layer
146  // coordinates are and projecting them may not be appropriate
147  QgsDebugMsgLevel( QStringLiteral( "Source CRS is invalid!" ), 4 );
148  return false;
149  }
150 
151  if ( !mDestCRS.isValid() )
152  {
153  //No destination projection is set so we set the default output projection to
154  //be the same as input proj.
155  mDestCRS = mSourceCRS;
156  QgsDebugMsgLevel( QStringLiteral( "Destination CRS is invalid!" ), 4 );
157  return false;
158  }
159 
160  mIsValid = true;
161 
162  // init the projections (destination and source)
163  freeProj();
164 
165 #if PROJ_VERSION_MAJOR < 6
166  Q_NOWARN_DEPRECATED_PUSH
167  int sourceDatumTransform = mSourceDatumTransform;
168  int destDatumTransform = mDestinationDatumTransform;
169  bool useDefaultDatumTransform = ( sourceDatumTransform == - 1 && destDatumTransform == -1 );
170 
171  mSourceProjString = mSourceCRS.toProj4();
172  if ( !useDefaultDatumTransform )
173  {
174  mSourceProjString = stripDatumTransform( mSourceProjString );
175  }
176  if ( sourceDatumTransform != -1 )
177  {
178  mSourceProjString += ( ' ' + QgsDatumTransform::datumTransformToProj( sourceDatumTransform ) );
179  }
180 
181  mDestProjString = mDestCRS.toProj4();
182  if ( !useDefaultDatumTransform )
183  {
184  mDestProjString = stripDatumTransform( mDestProjString );
185  }
186  if ( destDatumTransform != -1 )
187  {
188  mDestProjString += ( ' ' + QgsDatumTransform::datumTransformToProj( destDatumTransform ) );
189  }
190 
191  if ( !useDefaultDatumTransform )
192  {
193  addNullGridShifts( mSourceProjString, mDestProjString, sourceDatumTransform, destDatumTransform );
194  }
195  Q_NOWARN_DEPRECATED_POP
196 #endif
197 
198  // create proj projections for current thread
199  ProjData res = threadLocalProjData();
200 
201 #ifdef COORDINATE_TRANSFORM_VERBOSE
202  QgsDebugMsg( "From proj : " + mSourceCRS.toProj4() );
203  QgsDebugMsg( "To proj : " + mDestCRS.toProj4() );
204 #endif
205 
206 #if PROJ_VERSION_MAJOR>=6
207  if ( !res )
208  mIsValid = false;
209 #else
210  if ( !res.first || !res.second )
211  {
212  mIsValid = false;
213  }
214 #endif
215 
216 #ifdef COORDINATE_TRANSFORM_VERBOSE
217  if ( mIsValid )
218  {
219  QgsDebugMsg( QStringLiteral( "------------------------------------------------------------" ) );
220  QgsDebugMsg( QStringLiteral( "The OGR Coordinate transformation for this layer was set to" ) );
221  QgsLogger::debug<QgsCoordinateReferenceSystem>( "Input", mSourceCRS, __FILE__, __FUNCTION__, __LINE__ );
222  QgsLogger::debug<QgsCoordinateReferenceSystem>( "Output", mDestCRS, __FILE__, __FUNCTION__, __LINE__ );
223  QgsDebugMsg( QStringLiteral( "------------------------------------------------------------" ) );
224  }
225  else
226  {
227  QgsDebugMsg( QStringLiteral( "------------------------------------------------------------" ) );
228  QgsDebugMsg( QStringLiteral( "The OGR Coordinate transformation FAILED TO INITIALIZE!" ) );
229  QgsDebugMsg( QStringLiteral( "------------------------------------------------------------" ) );
230  }
231 #else
232  if ( !mIsValid )
233  {
234  QgsDebugMsg( QStringLiteral( "Coordinate transformation failed to initialize!" ) );
235  }
236 #endif
237 
238  //XXX todo overload == operator for QgsCoordinateReferenceSystem
239  //at the moment srs.parameters contains the whole proj def...soon it won't...
240  //if (mSourceCRS->toProj4() == mDestCRS->toProj4())
241  if ( mSourceCRS == mDestCRS )
242  {
243  // If the source and destination projection are the same, set the short
244  // circuit flag (no transform takes place)
245  mShortCircuit = true;
246  QgsDebugMsgLevel( QStringLiteral( "Source/Dest CRS equal, shortcircuit is set." ), 3 );
247  }
248  else
249  {
250  // Transform must take place
251  mShortCircuit = false;
252  QgsDebugMsgLevel( QStringLiteral( "Source/Dest CRS not equal, shortcircuit is not set." ), 3 );
253  }
254  return mIsValid;
255 }

291 ProjData QgsCoordinateTransformPrivate::threadLocalProjData()
292 {
293  QgsReadWriteLocker locker( mProjLock, QgsReadWriteLocker::Read );
294 
295 #if PROJ_VERSION_MAJOR>=6
296  PJ_CONTEXT *context = QgsProjContext::get();
297  QMap < uintptr_t, ProjData >::const_iterator it = mProjProjections.constFind( reinterpret_cast< uintptr_t>( context ) );
298 #else
299 #ifdef USE_THREAD_LOCAL
300  QMap < uintptr_t, QPair< projPJ, projPJ > >::const_iterator it = mProjProjections.constFind( reinterpret_cast< uintptr_t>( mProjContext.get() ) );
301 #else
302  projCtx pContext = nullptr;
303  if ( mProjContext.hasLocalData() )
304  {
305  pContext = mProjContext.localData()->get();
306  }
307  else
308  {
309  mProjContext.setLocalData( new QgsProjContextStore() );
310  pContext = mProjContext.localData()->get();
311  }
312  QMap < uintptr_t, QPair< projPJ, projPJ > >::const_iterator it = mProjProjections.constFind( reinterpret_cast< uintptr_t>( pContext ) );
313 #endif
314 #endif
315 
316  if ( it != mProjProjections.constEnd() )
317  {
318  ProjData res = it.value();
319  return res;
320  }
321 
322  // proj projections don't exist yet, so we need to create
323  locker.changeMode( QgsReadWriteLocker::Write );
324 
325 #if PROJ_VERSION_MAJOR>=6
326  // use a temporary proj error collector
327  QStringList projErrors;
328  proj_log_func( context, &projErrors, proj_collecting_logger );
329 
330  QgsProjUtils::proj_pj_unique_ptr transform;
331  if ( !mProjCoordinateOperation.isEmpty() )
332  {
333  transform.reset( proj_create( context, mProjCoordinateOperation.toUtf8().constData() ) );
334  if ( !transform || !proj_coordoperation_is_instantiable( context, transform.get() ) )
335  {
336  if ( sMissingGridUsedByContextHandler )
337  {
338  QgsDatumTransform::TransformDetails desired;
339  desired.proj = mProjCoordinateOperation;
340  desired.accuracy = -1; //unknown, can't retrieve from proj as we can't instantiate the op
341  desired.grids = QgsProjUtils::gridsUsed( mProjCoordinateOperation );
342  sMissingGridUsedByContextHandler( mSourceCRS, mDestCRS, desired );
343  }
344  else
345  {
346  const QString err = QObject::tr( "Could not use operation specified in project between %1 and %2. (Wanted to use: %3)." ).arg( mSourceCRS.authid(),
347  mDestCRS.authid(),
348  mProjCoordinateOperation );
349  QgsMessageLog::logMessage( err, QString(), Qgis::Critical );
350  }
351 
352  transform.reset();
353  }
354  else
355  {
356  // transform may have either the source or destination CRS using swapped axis order. For QGIS, we ALWAYS need regular x/y axis order
357  transform.reset( proj_normalize_for_visualization( context, transform.get() ) );
358  if ( !transform )
359  {
360  const QString err = QObject::tr( "Cannot normalize transform between %1 and %2" ).arg( mSourceCRS.authid(),
361  mDestCRS.authid() );
362  QgsMessageLog::logMessage( err, QString(), Qgis::Critical );
363  }
364  }
365  }
366 
367  QString nonAvailableError;
368  if ( !transform ) // fallback on default proj pathway
369  {
370  if ( !mSourceCRS.projObject() || ! mDestCRS.projObject() )
371  {
372  proj_log_func( context, nullptr, nullptr );
373  return nullptr;
374  }
375 
376  PJ_OPERATION_FACTORY_CONTEXT *operationContext = proj_create_operation_factory_context( context, nullptr );
377 
378  // We want to check ALL grids, not just those available for use
379  proj_operation_factory_context_set_grid_availability_use( context, operationContext, PROJ_GRID_AVAILABILITY_IGNORED );
380 
381  // See https://lists.osgeo.org/pipermail/proj/2019-May/008604.html
382  proj_operation_factory_context_set_spatial_criterion( context, operationContext, PROJ_SPATIAL_CRITERION_PARTIAL_INTERSECTION );
383 
384  if ( PJ_OBJ_LIST *ops = proj_create_operations( context, mSourceCRS.projObject(), mDestCRS.projObject(), operationContext ) )
385  {
386  int count = proj_list_get_count( ops );
387  if ( count < 1 )
388  {
389  // huh?
390  int errNo = proj_context_errno( context );
391  if ( errNo && errNo != -61 )
392  {
393  nonAvailableError = QString( proj_errno_string( errNo ) );
394  }
395  else
396  {
397  nonAvailableError = QObject::tr( "No coordinate operations are available between these two reference systems" );
398  }
399  }
400  else if ( count == 1 )
401  {
402  // only a single operation available. Can we use it?
403  transform.reset( proj_list_get( context, ops, 0 ) );
404  if ( transform )
405  {
406  if ( !proj_coordoperation_is_instantiable( context, transform.get() ) )
407  {
408  // uh oh :( something is missing! find what it is
409  for ( int j = 0; j < proj_coordoperation_get_grid_used_count( context, transform.get() ); ++j )
410  {
411  const char *shortName = nullptr;
412  const char *fullName = nullptr;
413  const char *packageName = nullptr;
414  const char *url = nullptr;
415  int directDownload = 0;
416  int openLicense = 0;
417  int isAvailable = 0;
418  proj_coordoperation_get_grid_used( context, transform.get(), j, &shortName, &fullName, &packageName, &url, &directDownload, &openLicense, &isAvailable );
419  if ( !isAvailable )
420  {
421  // found it!
422  if ( sMissingRequiredGridHandler )
423  {
424  QgsDatumTransform::GridDetails gridDetails;
425  gridDetails.shortName = QString( shortName );
426  gridDetails.fullName = QString( fullName );
427  gridDetails.packageName = QString( packageName );
428  gridDetails.url = QString( url );
429  gridDetails.directDownload = directDownload;
430  gridDetails.openLicense = openLicense;
431  gridDetails.isAvailable = isAvailable;
432  sMissingRequiredGridHandler( mSourceCRS, mDestCRS, gridDetails );
433  }
434  else
435  {
436  const QString err = QObject::tr( "Cannot create transform between %1 and %2, missing required grid %3" ).arg( mSourceCRS.authid(),
437  mDestCRS.authid(),
438  shortName );
439  QgsMessageLog::logMessage( err, QString(), Qgis::Critical );
440  }
441  break;
442  }
443  }
444  }
445  else
446  {
447 
448  // transform may have either the source or destination CRS using swapped axis order. For QGIS, we ALWAYS need regular x/y axis order
449  transform.reset( proj_normalize_for_visualization( context, transform.get() ) );
450  if ( !transform )
451  {
452  const QString err = QObject::tr( "Cannot normalize transform between %1 and %2" ).arg( mSourceCRS.authid(),
453  mDestCRS.authid() );
454  QgsMessageLog::logMessage( err, QString(), Qgis::Critical );
455  }
456  }
457  }
458  }
459  else
460  {
461  // multiple operations available. Can we use the best one?
462  QgsDatumTransform::TransformDetails preferred;
463  bool missingPreferred = false;
464  for ( int i = 0; i < count; ++ i )
465  {
466  transform.reset( proj_list_get( context, ops, i ) );
467  const bool isInstantiable = transform && proj_coordoperation_is_instantiable( context, transform.get() );
468  if ( i == 0 && transform && !isInstantiable )
469  {
470  // uh oh :( something is missing blocking us from the preferred operation!
471  missingPreferred = true;
472  preferred = QgsDatumTransform::transformDetailsFromPj( transform.get() );
473  }
474  if ( transform && isInstantiable )
475  {
476  // found one
477  break;
478  }
479  transform.reset();
480  }
481 
482  if ( transform && missingPreferred )
483  {
484  // found a transform, but it's not the preferred
485  QgsDatumTransform::TransformDetails available = QgsDatumTransform::transformDetailsFromPj( transform.get() );
486  if ( sMissingPreferredGridHandler )
487  {
488  sMissingPreferredGridHandler( mSourceCRS, mDestCRS, preferred, available );
489  }
490  else
491  {
492  const QString err = QObject::tr( "Using non-preferred coordinate operation between %1 and %2. Using %3, preferred %4." ).arg( mSourceCRS.authid(),
493  mDestCRS.authid(),
494  available.proj,
495  preferred.proj );
496  QgsMessageLog::logMessage( err, QString(), Qgis::Critical );
497  }
498  }
499 
500  // transform may have either the source or destination CRS using swapped axis order. For QGIS, we ALWAYS need regular x/y axis order
501  if ( transform )
502  transform.reset( proj_normalize_for_visualization( context, transform.get() ) );
503  if ( !transform )
504  {
505  const QString err = QObject::tr( "Cannot normalize transform between %1 and %2" ).arg( mSourceCRS.authid(),
506  mDestCRS.authid() );
507  QgsMessageLog::logMessage( err, QString(), Qgis::Critical );
508  }
509  }
510  proj_list_destroy( ops );
511  }
512  proj_operation_factory_context_destroy( operationContext );
513  }
514 
515  if ( !transform && nonAvailableError.isEmpty() )
516  {
517  int errNo = proj_context_errno( context );
518  if ( errNo && errNo != -61 )
519  {
520  nonAvailableError = QString( proj_errno_string( errNo ) );
521  }
522  else if ( !projErrors.empty() )
523  {
524  nonAvailableError = projErrors.constLast();
525  }
526 
527  if ( nonAvailableError.isEmpty() )
528  {
529  nonAvailableError = QObject::tr( "No coordinate operations are available between these two reference systems" );
530  }
531  else
532  {
533  // strip proj prefixes from error string, so that it's nicer for users
534  nonAvailableError = nonAvailableError.remove( QStringLiteral( "internal_proj_create_operations: " ) );
535  }
536  }
537 
538  if ( !nonAvailableError.isEmpty() )
539  {
540  if ( sCoordinateOperationCreationErrorHandler )
541  {
542  sCoordinateOperationCreationErrorHandler( mSourceCRS, mDestCRS, nonAvailableError );
543  }
544  else
545  {
546  const QString err = QObject::tr( "Cannot create transform between %1 and %2: %3" ).arg( mSourceCRS.authid(),
547  mDestCRS.authid(),
548  nonAvailableError );
549  QgsMessageLog::logMessage( err, QString(), Qgis::Critical );
550  }
551  }
552 
553  // reset logger to terminal output
554  proj_log_func( context, nullptr, proj_logger );
555 
556  if ( !transform )
557  {
558  // ouch!
559  return nullptr;
560  }
561 
562  ProjData res = transform.release();
563  mProjProjections.insert( reinterpret_cast< uintptr_t>( context ), res );
564 #else
565 #ifdef USE_THREAD_LOCAL
566  Q_NOWARN_DEPRECATED_PUSH
567  QPair<projPJ, projPJ> res = qMakePair( pj_init_plus_ctx( mProjContext.get(), mSourceProjString.toUtf8() ),
568  pj_init_plus_ctx( mProjContext.get(), mDestProjString.toUtf8() ) );
569  Q_NOWARN_DEPRECATED_POP
570  mProjProjections.insert( reinterpret_cast< uintptr_t>( mProjContext.get() ), res );
571 #else
572  QPair<projPJ, projPJ> res = qMakePair( pj_init_plus_ctx( pContext, mSourceProjString.toUtf8() ),
573  pj_init_plus_ctx( pContext, mDestProjString.toUtf8() ) );
574  mProjProjections.insert( reinterpret_cast< uintptr_t>( pContext ), res );
575 #endif
576 #endif
577  return res;
578 }

602 {
603  QStringList parameterSplit = proj4.split( '+', QString::SkipEmptyParts );
604  QString currentParameter;
605  QString newProjString;
606

608  {

610  if ( !currentParameter.startsWith( QLatin1String( "towgs84" ), Qt::CaseInsensitive )
611  && !currentParameter.startsWith( QLatin1String( "nadgrids" ), Qt::CaseInsensitive ) )
612  {
613  newProjString.append( '+' );
614  newProjString.append( currentParameter );
615  newProjString.append( ' ' );

621 void QgsCoordinateTransformPrivate::addNullGridShifts( QString &srcProjString, QString &destProjString,
622  int sourceDatumTransform, int destinationDatumTransform ) const
623 {
624  //if one transformation uses ntv2, the other one needs to be null grid shift
