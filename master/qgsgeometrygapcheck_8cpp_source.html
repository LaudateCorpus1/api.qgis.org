51 void QgsGeometryGapCheck::collectErrors( const QMap<QString, QgsFeaturePool *> &featurePools, QList<QgsGeometryCheckError *> &errors, QStringList &messages, QgsFeedback *feedback, const LayerFeatureIds &ids ) const
52 {
53  if ( feedback )
54  feedback->setProgress( feedback->progress() + 1.0 );
55 
56  std::unique_ptr<QgsAbstractGeometry> allowedGapsGeom;
57  std::unique_ptr<QgsGeometryEngine> allowedGapsGeomEngine;
58 
59  if ( mAllowedGapsSource )
60  {
61  QVector<QgsGeometry> allowedGaps;
62  QgsFeatureRequest request;
63  request.setSubsetOfAttributes( QgsAttributeList() );
64  QgsFeatureIterator iterator = mAllowedGapsSource->getFeatures( request );
65  QgsFeature feature;
66 
67  while ( iterator.nextFeature( feature ) )
68  {
69  QgsGeometry geom = feature.geometry();
70  QgsGeometry gg = geom.buffer( mAllowedGapsBuffer, 20 );
71  allowedGaps.append( gg );
72  }
73 
74  std::unique_ptr< QgsGeometryEngine > allowedGapsEngine = QgsGeometryCheckerUtils::createGeomEngine( nullptr, mContext->tolerance );
75 
76  // Create union of allowed gaps
77  QString errMsg;
78  allowedGapsGeom.reset( allowedGapsEngine->combine( allowedGaps, &errMsg ) );
79  allowedGapsGeomEngine = QgsGeometryCheckerUtils::createGeomEngine( allowedGapsGeom.get(), mContext->tolerance );
80  allowedGapsGeomEngine->prepareGeometry();
81  }
82 
83  QVector<QgsGeometry> geomList;
84  QMap<QString, QgsFeatureIds> featureIds = ids.isEmpty() ? allLayerFeatureIds( featurePools ) : ids.toMap();
85  const QgsGeometryCheckerUtils::LayerFeatures layerFeatures( featurePools, featureIds, compatibleGeometryTypes(), nullptr, mContext, true );
86  for ( const QgsGeometryCheckerUtils::LayerFeature &layerFeature : layerFeatures )
87  {
88  geomList.append( layerFeature.geometry() );
89 
90  if ( feedback && feedback->isCanceled() )
91  {
92  geomList.clear();
93  break;
94  }
95  }
96 
97  if ( geomList.isEmpty() )
98  {
99  return;
100  }
101 
102  std::unique_ptr< QgsGeometryEngine > geomEngine = QgsGeometryCheckerUtils::createGeomEngine( nullptr, mContext->tolerance );
103 
104  // Create union of geometry
105  QString errMsg;
106  std::unique_ptr<QgsAbstractGeometry> unionGeom( geomEngine->combine( geomList, &errMsg ) );
107  if ( !unionGeom )
108  {
109  messages.append( tr( "Gap check: %1" ).arg( errMsg ) );
110  return;
111  }
112 
113  // Get envelope of union
114  geomEngine = QgsGeometryCheckerUtils::createGeomEngine( unionGeom.get(), mContext->tolerance );
115  geomEngine->prepareGeometry();
116  std::unique_ptr<QgsAbstractGeometry> envelope( geomEngine->envelope( &errMsg ) );
117  if ( !envelope )
118  {
119  messages.append( tr( "Gap check: %1" ).arg( errMsg ) );
120  return;
121  }
122 
123  // Buffer envelope
124  geomEngine = QgsGeometryCheckerUtils::createGeomEngine( envelope.get(), mContext->tolerance );
125  geomEngine->prepareGeometry();
126  QgsAbstractGeometry *bufEnvelope = geomEngine->buffer( 2, 0, GEOSBUF_CAP_SQUARE, GEOSBUF_JOIN_MITRE, 4. ); //#spellok //#spellok
127  envelope.reset( bufEnvelope );
128 
129  // Compute difference between envelope and union to obtain gap polygons
130  geomEngine = QgsGeometryCheckerUtils::createGeomEngine( envelope.get(), mContext->tolerance );
131  geomEngine->prepareGeometry();
132  std::unique_ptr<QgsAbstractGeometry> diffGeom( geomEngine->difference( unionGeom.get(), &errMsg ) );
133  if ( !diffGeom )
134  {
135  messages.append( tr( "Gap check: %1" ).arg( errMsg ) );
136  return;
137  }
138 
139  // For each gap polygon which does not lie on the boundary, get neighboring polygons and add error
140  QgsGeometryPartIterator parts = diffGeom->parts();
141  while ( parts.hasNext() )
142  {
143  const QgsAbstractGeometry *gapGeom = parts.next();
144  // Skip the gap between features and boundingbox
145  const double spacing = context()->tolerance;
146  if ( gapGeom->boundingBox().snappedToGrid( spacing ) == envelope->boundingBox().snappedToGrid( spacing ) )
147  {
148  continue;
149  }
150 
151  // Skip gaps above threshold
152  if ( ( mGapThresholdMapUnits > 0 && gapGeom->area() > mGapThresholdMapUnits ) || gapGeom->area() < mContext->reducedTolerance )
153  {
154  continue;
155  }
156 
157  QgsRectangle gapAreaBBox = gapGeom->boundingBox();
158 
159  // Get neighboring polygons
160  QMap<QString, QgsFeatureIds> neighboringIds;
161  const QgsGeometryCheckerUtils::LayerFeatures layerFeatures( featurePools, featureIds.keys(), gapAreaBBox, compatibleGeometryTypes(), mContext );
162  for ( const QgsGeometryCheckerUtils::LayerFeature &layerFeature : layerFeatures )
163  {
164  const QgsGeometry geom = layerFeature.geometry();
165  if ( QgsGeometryCheckerUtils::sharedEdgeLength( gapGeom, geom.constGet(), mContext->reducedTolerance ) > 0 )
166  {
167  neighboringIds[layerFeature.layer()->id()].insert( layerFeature.feature().id() );
168  gapAreaBBox.combineExtentWith( layerFeature.geometry().boundingBox() );
169  }
170  }
171 
172  if ( neighboringIds.isEmpty() )
173  {
174  continue;
175  }
176 
177  if ( allowedGapsGeomEngine && allowedGapsGeomEngine->contains( gapGeom ) )
178  {
179  continue;
180  }
181 
182  // Add error
183  double area = gapGeom->area();
184  QgsRectangle gapBbox = gapGeom->boundingBox();
185  errors.append( new QgsGeometryGapCheckError( this, QString(), QgsGeometry( gapGeom->clone() ), neighboringIds, area, gapBbox, gapAreaBBox ) );
186  }

189 void QgsGeometryGapCheck::fixError( const QMap<QString, QgsFeaturePool *> &featurePools, QgsGeometryCheckError *error, int method, const QMap<QString, int> & /*mergeAttributeIndices*/, Changes &changes ) const
190 {
191  QMetaEnum metaEnum = QMetaEnum::fromType<QgsGeometryGapCheck::ResolutionMethod>();
192  if ( !metaEnum.isValid() || !metaEnum.valueToKey( method ) )
193  {
194  error->setFixFailed( tr( "Unknown method" ) );
195  }
196  else
197  {
198  ResolutionMethod methodValue = static_cast<ResolutionMethod>( method );
199  switch ( methodValue )
200  {
201  case NoChange:
202  error->setFixed( method );
203  break;
204 
205  case MergeLongestEdge:
206  {
207  QString errMsg;
208  if ( mergeWithNeighbor( featurePools, static_cast<QgsGeometryGapCheckError *>( error ), changes, errMsg ) )
209  {
210  error->setFixed( method );
211  }
212  else
213  {
214  error->setFixFailed( tr( "Failed to merge with neighbor: %1" ).arg( errMsg ) );
215  }
216  break;
217  }
218 
219  case AddToAllowedGaps:
220  {
221  QgsVectorLayer *layer = qobject_cast<QgsVectorLayer *>( mAllowedGapsLayer.data() );
222  if ( layer )
223  {
224  if ( !layer->isEditable() && !layer->startEditing() )
225  {
226  error->setFixFailed( tr( "Could not start editing layer %1" ).arg( layer->name() ) );
227  }
228  else
229  {
230  QgsFeature feature = QgsVectorLayerUtils::createFeature( layer, error->geometry() );
231  if ( !layer->addFeature( feature ) )
232  {
233  error->setFixFailed( tr( "Could not add feature to layer %1" ).arg( layer->name() ) );
234  }
235  else
236  {
237  error->setFixed( method );
238  }
239  }
240  }
241  else
242  {
243  error->setFixFailed( tr( "Allowed gaps layer could not be resolved" ) );
244  }
245  break;
246  }
247  }
248  }

251 bool QgsGeometryGapCheck::mergeWithNeighbor( const QMap<QString, QgsFeaturePool *> &featurePools,
252  QgsGeometryGapCheckError *err,
253  Changes &changes, QString &errMsg ) const
254 {
255  double maxVal = 0.;
256  QString mergeLayerId;
257  QgsFeature mergeFeature;
258  int mergePartIdx = -1;
259 
260  const QgsGeometry geometry = err->geometry();
261  const QgsAbstractGeometry *errGeometry = QgsGeometryCheckerUtils::getGeomPart( geometry.constGet(), 0 );
262 
263  const auto layerIds = err->neighbors().keys();
264  // Search for touching neighboring geometries
265  for ( const QString &layerId : layerIds )
266  {
267  QgsFeaturePool *featurePool = featurePools.value( layerId );
268  std::unique_ptr<QgsAbstractGeometry> errLayerGeom( errGeometry->clone() );
269  QgsCoordinateTransform ct( featurePool->crs(), mContext->mapCrs, mContext->transformContext );
270  errLayerGeom->transform( ct, QgsCoordinateTransform::ReverseTransform );
271 
272  const auto featureIds = err->neighbors().value( layerId );
273 
274  for ( QgsFeatureId testId : featureIds )
275  {
276  QgsFeature testFeature;
277  if ( !featurePool->getFeature( testId, testFeature ) )
278  {
279  continue;
280  }
281  const QgsGeometry featureGeom = testFeature.geometry();
282  const QgsAbstractGeometry *testGeom = featureGeom.constGet();
283  for ( int iPart = 0, nParts = testGeom->partCount(); iPart < nParts; ++iPart )
284  {
285  double len = QgsGeometryCheckerUtils::sharedEdgeLength( errLayerGeom.get(), QgsGeometryCheckerUtils::getGeomPart( testGeom, iPart ), mContext->reducedTolerance );
286  if ( len > maxVal )
287  {
288  maxVal = len;
289  mergeFeature = testFeature;
290  mergePartIdx = iPart;
291  mergeLayerId = layerId;
292  }
293  }
294  }
295  }
296 
297  if ( maxVal == 0. )
298  {
299  return false;
300  }
301 
302  // Merge geometries
303  QgsFeaturePool *featurePool = featurePools[ mergeLayerId ];
304  std::unique_ptr<QgsAbstractGeometry> errLayerGeom( errGeometry->clone() );
305  QgsCoordinateTransform ct( featurePool->crs(), mContext->mapCrs, mContext->transformContext );
306  errLayerGeom->transform( ct, QgsCoordinateTransform::ReverseTransform );
307  const QgsGeometry mergeFeatureGeom = mergeFeature.geometry();
308  const QgsAbstractGeometry *mergeGeom = mergeFeatureGeom.constGet();
309  std::unique_ptr< QgsGeometryEngine > geomEngine = QgsGeometryCheckerUtils::createGeomEngine( errLayerGeom.get(), mContext->reducedTolerance );
310  std::unique_ptr<QgsAbstractGeometry> combinedGeom( geomEngine->combine( QgsGeometryCheckerUtils::getGeomPart( mergeGeom, mergePartIdx ), &errMsg ) );
311  if ( !combinedGeom || combinedGeom->isEmpty() || !QgsWkbTypes::isSingleType( combinedGeom->wkbType() ) )
312  {
313  return false;
314  }
315 
316  // Add merged polygon to destination geometry
317  replaceFeatureGeometryPart( featurePools, mergeLayerId, mergeFeature, mergePartIdx, combinedGeom.release(), changes );
318 
319  return true;
320 }
