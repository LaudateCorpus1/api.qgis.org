72 QVariant QgsStyleModel::data( const QModelIndex &index, int role ) const
73 {
74  if ( index.row() < 0 || index.row() >= rowCount( QModelIndex() ) )
75  return QVariant();
76 
77 
78  QgsStyle::StyleEntity entityType = entityTypeFromRow( index.row() );
79 
80  QString name;
81  switch ( entityType )
82  {
83  case QgsStyle::SymbolEntity:
84  name = mSymbolNames.value( index.row() );
85  break;
86 
87  case QgsStyle::ColorrampEntity:
88  name = mRampNames.value( index.row() - mSymbolNames.size() );
89  break;
90 
91  case QgsStyle::TextFormatEntity:
92  name = mTextFormatNames.value( index.row() - mSymbolNames.size() - mRampNames.size() );
93  break;
94 
95  case QgsStyle::LabelSettingsEntity:
96  name = mLabelSettingsNames.value( index.row() - mSymbolNames.size() - mRampNames.size() - mTextFormatNames.size() );
97  break;
98 
99  case QgsStyle::TagEntity:
100  case QgsStyle::SmartgroupEntity:
101  break;
102  }
103 
104  switch ( role )
105  {
106  case Qt::DisplayRole:
107  case Qt::ToolTipRole:
108  case Qt::EditRole:
109  {
110  switch ( index.column() )
111  {
112  case Name:
113  {
114  const QStringList tags = mStyle->tagsOfSymbol( entityType, name );
115 
116  if ( role == Qt::ToolTipRole )
117  {
118  QString tooltip = QStringLiteral( "<h3>%1</h3><p><i>%2</i>" ).arg( name,
119  tags.count() > 0 ? tags.join( QStringLiteral( ", " ) ) : tr( "Not tagged" ) );
120 
121  switch ( entityType )
122  {
123  case QgsStyle::SymbolEntity:
124  {
125  // create very large preview image
126  std::unique_ptr< QgsSymbol > symbol( mStyle->symbol( name ) );
127  if ( symbol )
128  {
129  int width = static_cast< int >( Qgis::UI_SCALE_FACTOR * QFontMetrics( data( index, Qt::FontRole ).value< QFont >() ).width( 'X' ) * 23 );
130  int height = static_cast< int >( width / 1.61803398875 ); // golden ratio
131  QPixmap pm = QgsSymbolLayerUtils::symbolPreviewPixmap( symbol.get(), QSize( width, height ), height / 20 );
132  QByteArray data;
133  QBuffer buffer( &data );
134  pm.save( &buffer, "PNG", 100 );
135  tooltip += QStringLiteral( "<p><img src='data:image/png;base64, %3'>" ).arg( QString( data.toBase64() ) );
136  }
137  break;
138  }
139 
140  case QgsStyle::TextFormatEntity:
141  {
142  int width = static_cast< int >( Qgis::UI_SCALE_FACTOR * QFontMetrics( data( index, Qt::FontRole ).value< QFont >() ).width( 'X' ) * 23 );
143  int height = static_cast< int >( width / 1.61803398875 ); // golden ratio
144  const QgsTextFormat format = mStyle->textFormat( name );
145  QPixmap pm = QgsTextFormat::textFormatPreviewPixmap( format, QSize( width, height ), QString(), height / 20 );
146  QByteArray data;
147  QBuffer buffer( &data );
148  pm.save( &buffer, "PNG", 100 );
149  tooltip += QStringLiteral( "<p><img src='data:image/png;base64, %3'>" ).arg( QString( data.toBase64() ) );
150  break;
151  }
152 
153  case QgsStyle::LabelSettingsEntity:
154  {
155  int width = static_cast< int >( Qgis::UI_SCALE_FACTOR * QFontMetrics( data( index, Qt::FontRole ).value< QFont >() ).width( 'X' ) * 23 );
156  int height = static_cast< int >( width / 1.61803398875 ); // golden ratio
157  const QgsPalLayerSettings settings = mStyle->labelSettings( name );
158  QPixmap pm = QgsPalLayerSettings::labelSettingsPreviewPixmap( settings, QSize( width, height ), QString(), height / 20 );
159  QByteArray data;
160  QBuffer buffer( &data );
161  pm.save( &buffer, "PNG", 100 );
162  tooltip += QStringLiteral( "<p><img src='data:image/png;base64, %3'>" ).arg( QString( data.toBase64() ) );
163  break;
164  }
165 
166  case QgsStyle::ColorrampEntity:
167  case QgsStyle::TagEntity:
168  case QgsStyle::SmartgroupEntity:
169  break;
170  }
171  return tooltip;
172  }
173  else
174  {
175  return name;
176  }
177  }
178  case Tags:
179  return mStyle->tagsOfSymbol( entityType, name ).join( QStringLiteral( ", " ) );
180  }
181  return QVariant();
182  }
183 
184  case Qt::DecorationRole:
185  {
186  // Generate icons at all additional sizes specified for the model.
187  // This allows the model to have size responsive icons.
188  switch ( index.column() )
189  {
190  case Name:
191  switch ( entityType )
192  {
193  case QgsStyle::SymbolEntity:
194  {
195  // use cached icon if possible
196  QIcon icon = mSymbolIconCache.value( name );
197  if ( !icon.isNull() )
198  return icon;
199 
200  std::unique_ptr< QgsSymbol > symbol( mStyle->symbol( name ) );
201  if ( symbol )
202  {
203  if ( mAdditionalSizes.isEmpty() )
204  icon.addPixmap( QgsSymbolLayerUtils::symbolPreviewPixmap( symbol.get(), QSize( 24, 24 ), 1 ) );
205 
206  for ( const QVariant &size : mAdditionalSizes )
207  {
208  QSize s = size.toSize();
209  icon.addPixmap( QgsSymbolLayerUtils::symbolPreviewPixmap( symbol.get(), s, static_cast< int >( s.width() * ICON_PADDING_FACTOR ) ) );
210  }
211 
212  }
213  mSymbolIconCache.insert( name, icon );
214  return icon;
215  }
216  case QgsStyle::ColorrampEntity:
217  {
218  // use cached icon if possible
219  QIcon icon = mColorRampIconCache.value( name );
220  if ( !icon.isNull() )
221  return icon;
222 
223  std::unique_ptr< QgsColorRamp > ramp( mStyle->colorRamp( name ) );
224  if ( ramp )
225  {
226  if ( mAdditionalSizes.isEmpty() )
227  icon.addPixmap( QgsSymbolLayerUtils::colorRampPreviewPixmap( ramp.get(), QSize( 24, 24 ), 1 ) );
228  for ( const QVariant &size : mAdditionalSizes )
229  {
230  QSize s = size.toSize();
231  icon.addPixmap( QgsSymbolLayerUtils::colorRampPreviewPixmap( ramp.get(), s, static_cast< int >( s.width() * ICON_PADDING_FACTOR ) ) );
232  }
233 
234  }
235  mColorRampIconCache.insert( name, icon );
236  return icon;
237  }
238 
239  case QgsStyle::TextFormatEntity:
240  {
241  // use cached icon if possible
242  QIcon icon = mTextFormatIconCache.value( name );
243  if ( !icon.isNull() )
244  return icon;
245 
246  const QgsTextFormat format( mStyle->textFormat( name ) );
247  if ( mAdditionalSizes.isEmpty() )
248  icon.addPixmap( QgsTextFormat::textFormatPreviewPixmap( format, QSize( 24, 24 ), QString(), 1 ) );
249  for ( const QVariant &size : mAdditionalSizes )
250  {
251  QSize s = size.toSize();
252  icon.addPixmap( QgsTextFormat::textFormatPreviewPixmap( format, s, QString(), static_cast< int >( s.width() * ICON_PADDING_FACTOR ) ) );
253  }
254  mTextFormatIconCache.insert( name, icon );
255  return icon;
256  }
257 
258  case QgsStyle::LabelSettingsEntity:
259  {
260  // use cached icon if possible
261  QIcon icon = mLabelSettingsIconCache.value( name );
262  if ( !icon.isNull() )
263  return icon;
264 
265  const QgsPalLayerSettings settings( mStyle->labelSettings( name ) );
266  if ( mAdditionalSizes.isEmpty() )
267  icon.addPixmap( QgsPalLayerSettings::labelSettingsPreviewPixmap( settings, QSize( 24, 24 ), QString(), 1 ) );
268  for ( const QVariant &size : mAdditionalSizes )
269  {
270  QSize s = size.toSize();
271  icon.addPixmap( QgsPalLayerSettings::labelSettingsPreviewPixmap( settings, s, QString(), static_cast< int >( s.width() * ICON_PADDING_FACTOR ) ) );
272  }
273  mLabelSettingsIconCache.insert( name, icon );
274  return icon;
275  }
276 
277  case QgsStyle::TagEntity:
278  case QgsStyle::SmartgroupEntity:
279  return QVariant();
280  }
281  break;
282 
283  case Tags:
284  return QVariant();
285  }
286  return QVariant();
287  }
288 
289  case TypeRole:
290  return entityType;
291 
292  case TagRole:
293  return mStyle->tagsOfSymbol( entityType, name );
294 
295  case IsFavoriteRole:
296  return mStyle->isFavorite( entityType, name );
297 
298  case SymbolTypeRole:
299  {
300  if ( entityType != QgsStyle::SymbolEntity )
301  return QVariant();
302 
303  const QgsSymbol *symbol = mStyle->symbolRef( name );
304  return symbol ? symbol->type() : QVariant();
305  }
306 
307  case LayerTypeRole:
308  {
309  if ( entityType != QgsStyle::LabelSettingsEntity )
310  return QVariant();
311 
312  return mStyle->labelSettingsLayerType( name );
313  }
314 
315  default:
316  return QVariant();
317  }
318 #ifndef _MSC_VER // avoid warning
319  return QVariant(); // avoid warning
320 #endif
321 }

323 bool QgsStyleModel::setData( const QModelIndex &index, const QVariant &value, int role )
324 {
325  if ( index.row() < 0 || index.row() >= rowCount( QModelIndex() ) || role != Qt::EditRole )
326  return false;
327 
328  switch ( index.column() )
329  {
330  case Name:
331  {
332  QgsStyle::StyleEntity entityType = entityTypeFromRow( index.row() );
333  QString name;
334  switch ( entityType )
335  {
336  case QgsStyle::SymbolEntity:
337  name = mSymbolNames.value( index.row() );
338  break;
339 
340  case QgsStyle::ColorrampEntity:
341  name = mRampNames.value( index.row() - mSymbolNames.size() );
342  break;
343 
344  case QgsStyle::TextFormatEntity:
345  name = mTextFormatNames.value( index.row() - mSymbolNames.size() - mRampNames.size() );
346  break;
347 
348  case QgsStyle::LabelSettingsEntity:
349  name = mLabelSettingsNames.value( index.row() - mSymbolNames.size() - mRampNames.size() - mTextFormatNames.size() );
350  break;
351 
352  case QgsStyle::TagEntity:
353  case QgsStyle::SmartgroupEntity:
354  break;
355  }
356 
357  const QString newName = value.toString();
358 
359  switch ( entityType )
360  {
361  case QgsStyle::SymbolEntity:
362  return mStyle->renameSymbol( name, newName );
363 
364  case QgsStyle::ColorrampEntity:
365  return mStyle->renameColorRamp( name, newName );
366 
367  case QgsStyle::TextFormatEntity:
368  return mStyle->renameTextFormat( name, newName );
369 
370  case QgsStyle::LabelSettingsEntity:
371  return mStyle->renameLabelSettings( name, newName );
372 
373  case QgsStyle::TagEntity:
374  case QgsStyle::SmartgroupEntity:
375  return false;
376  }
377  break;
378  }
379 
380  case Tags:
381  return false;
382  }
383 
384  return false;
385 }

840 bool QgsStyleProxyModel::filterAcceptsRow( int source_row, const QModelIndex &source_parent ) const
841 {
842  if ( mFilterString.isEmpty() && !mEntityFilterEnabled && !mSymbolTypeFilterEnabled && mTagId < 0 && mSmartGroupId < 0 && !mFavoritesOnly )
843  return true;
844 
845  QModelIndex index = sourceModel()->index( source_row, 0, source_parent );
846  const QString name = sourceModel()->data( index ).toString();
847  const QStringList tags = sourceModel()->data( index, QgsStyleModel::TagRole ).toStringList();
848 
849  QgsStyle::StyleEntity styleEntityType = static_cast< QgsStyle::StyleEntity >( sourceModel()->data( index, QgsStyleModel::TypeRole ).toInt() );
850  if ( mEntityFilterEnabled && ( mEntityFilters.empty() || !mEntityFilters.contains( styleEntityType ) ) )
851  return false;
852 
853  QgsSymbol::SymbolType symbolType = static_cast< QgsSymbol::SymbolType >( sourceModel()->data( index, QgsStyleModel::SymbolTypeRole ).toInt() );
854  if ( mSymbolTypeFilterEnabled && symbolType != mSymbolType )
855  return false;
856 
857  if ( styleEntityType == QgsStyle::LabelSettingsEntity && mLayerType != QgsWkbTypes::UnknownGeometry &&
858  mLayerType != static_cast< QgsWkbTypes::GeometryType >( sourceModel()->data( index, QgsStyleModel::LayerTypeRole ).toInt() ) )
859  return false;
860 
861  if ( mTagId >= 0 && !mTaggedSymbolNames.contains( name ) )
862  return false;
863 
864  if ( mSmartGroupId >= 0 && !mSmartGroupSymbolNames.contains( name ) )
865  return false;
866 
867  if ( mFavoritesOnly && !sourceModel()->data( index, QgsStyleModel::IsFavoriteRole ).toBool() )
868  return false;
869 
870  if ( !mFilterString.isEmpty() )
871  {
872  // filter by word, in both filter string and style entity name/tags
873  // this allows matching of a filter string "hash line" to the symbol "hashed red lines"
874  const QStringList partsToMatch = mFilterString.trimmed().split( ' ' );
875 
876  QStringList partsToSearch = name.split( ' ' );
877  for ( const QString &tag : tags )
878  {
879  partsToSearch.append( tag.split( ' ' ) );
880  }
881 
882  for ( const QString &part : partsToMatch )
883  {
884  bool found = false;
885  for ( const QString &partToSearch : qgis::as_const( partsToSearch ) )
886  {
887  if ( partToSearch.contains( part, Qt::CaseInsensitive ) )
888  {
889  found = true;
890  break;
891  }
892  }
893  if ( !found )
894  return false; // couldn't find a match for this word, so hide entity
895  }
896  }
897 
898  return true;
899 }
