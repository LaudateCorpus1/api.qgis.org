441 QgsFeatureRenderer *QgsGraduatedSymbolRenderer::create( QDomElement &element, const QgsReadWriteContext &context )
442 {
443  QDomElement symbolsElem = element.firstChildElement( QStringLiteral( "symbols" ) );
444  if ( symbolsElem.isNull() )
445  return nullptr;
446 
447  QDomElement rangesElem = element.firstChildElement( QStringLiteral( "ranges" ) );
448  if ( rangesElem.isNull() )
449  return nullptr;
450 
451  QgsSymbolMap symbolMap = QgsSymbolLayerUtils::loadSymbols( symbolsElem, context );
452  QgsRangeList ranges;
453 
454  QDomElement rangeElem = rangesElem.firstChildElement();
455  while ( !rangeElem.isNull() )
456  {
457  if ( rangeElem.tagName() == QLatin1String( "range" ) )
458  {
459  double lowerValue = rangeElem.attribute( QStringLiteral( "lower" ) ).toDouble();
460  double upperValue = rangeElem.attribute( QStringLiteral( "upper" ) ).toDouble();
461  QString symbolName = rangeElem.attribute( QStringLiteral( "symbol" ) );
462  QString label = rangeElem.attribute( QStringLiteral( "label" ) );
463  bool render = rangeElem.attribute( QStringLiteral( "render" ), QStringLiteral( "true" ) ) != QLatin1String( "false" );
464  if ( symbolMap.contains( symbolName ) )
465  {
466  QgsSymbol *symbol = symbolMap.take( symbolName );
467  ranges.append( QgsRendererRange( lowerValue, upperValue, symbol, label, render ) );
468  }
469  }
470  rangeElem = rangeElem.nextSiblingElement();
471  }
472 
473  QString attrName = element.attribute( QStringLiteral( "attr" ) );
474 
475  QgsGraduatedSymbolRenderer *r = new QgsGraduatedSymbolRenderer( attrName, ranges );
476 
477  QString attrMethod = element.attribute( QStringLiteral( "graduatedMethod" ) );
478  if ( !attrMethod.isEmpty() )
479  {
480  if ( attrMethod == graduatedMethodStr( GraduatedColor ) )
481  r->setGraduatedMethod( GraduatedColor );
482  else if ( attrMethod == graduatedMethodStr( GraduatedSize ) )
483  r->setGraduatedMethod( GraduatedSize );
484  }
485 
486 
487  // delete symbols if there are any more
488  QgsSymbolLayerUtils::clearSymbolMap( symbolMap );
489 
490  // try to load source symbol (optional)
491  QDomElement sourceSymbolElem = element.firstChildElement( QStringLiteral( "source-symbol" ) );
492  if ( !sourceSymbolElem.isNull() )
493  {
494  QgsSymbolMap sourceSymbolMap = QgsSymbolLayerUtils::loadSymbols( sourceSymbolElem, context );
495  if ( sourceSymbolMap.contains( QStringLiteral( "0" ) ) )
496  {
497  r->setSourceSymbol( sourceSymbolMap.take( QStringLiteral( "0" ) ) );
498  }
499  QgsSymbolLayerUtils::clearSymbolMap( sourceSymbolMap );
500  }
501 
502  // try to load color ramp (optional)
503  QDomElement sourceColorRampElem = element.firstChildElement( QStringLiteral( "colorramp" ) );
504  if ( !sourceColorRampElem.isNull() && sourceColorRampElem.attribute( QStringLiteral( "name" ) ) == QLatin1String( "[source]" ) )
505  {
506  r->setSourceColorRamp( QgsSymbolLayerUtils::loadColorRamp( sourceColorRampElem ) );
507  }
508 
509  // try to load mode
510 
511  QDomElement modeElem = element.firstChildElement( QStringLiteral( "mode" ) ); // old format, backward compatibility
512  QDomElement methodElem = element.firstChildElement( QStringLiteral( "classificationMethod" ) );
513  QgsClassificationMethod *method = nullptr;
514 
515  // TODO QGIS 4 Remove
516  // backward compatibility for QGIS project < 3.10
517  if ( !modeElem.isNull() )
518  {
519  QString modeString = modeElem.attribute( QStringLiteral( "name" ) );
520  QString methodId;
521  // the strings saved in the project does not match with the old Mode enum
522  if ( modeString == QLatin1String( "equal" ) )
523  methodId = QStringLiteral( "EqualInterval" );
524  else if ( modeString == QLatin1String( "quantile" ) )
525  methodId = QStringLiteral( "Quantile" );
526  else if ( modeString == QLatin1String( "jenks" ) )
527  methodId = QStringLiteral( "Jenks" );
528  else if ( modeString == QLatin1String( "stddev" ) )
529  methodId = QStringLiteral( "StdDev" );
530  else if ( modeString == QLatin1String( "pretty" ) )
531  methodId = QStringLiteral( "Pretty" );
532 
533  method = QgsApplication::classificationMethodRegistry()->method( methodId );
534 
535  // symmetric mode
536  QDomElement symmetricModeElem = element.firstChildElement( QStringLiteral( "symmetricMode" ) );
537  if ( !symmetricModeElem.isNull() )
538  {
539  // symmetry
540  QString symmetricEnabled = symmetricModeElem.attribute( QStringLiteral( "enabled" ) );
541  QString symmetricPointString = symmetricModeElem.attribute( QStringLiteral( "symmetryPoint" ) );
542  QString astrideEnabled = symmetricModeElem.attribute( QStringLiteral( "astride" ) );
543  method->setSymmetricMode( symmetricEnabled == QLatin1String( "true" ), symmetricPointString.toDouble(), astrideEnabled == QLatin1String( "true" ) );
544  }
545  QDomElement labelFormatElem = element.firstChildElement( QStringLiteral( "labelformat" ) );
546  if ( !labelFormatElem.isNull() )
547  {
548  // label format
549  QString format = labelFormatElem.attribute( QStringLiteral( "format" ), "%1" + QStringLiteral( " - " ) + "%2" );
550  int precision = labelFormatElem.attribute( QStringLiteral( "decimalplaces" ), QStringLiteral( "4" ) ).toInt();
551  bool trimTrailingZeroes = labelFormatElem.attribute( QStringLiteral( "trimtrailingzeroes" ), QStringLiteral( "false" ) ) == QLatin1String( "true" );
552  method->setLabelFormat( format );
553  method->setLabelPrecision( precision );
554  method->setLabelTrimTrailingZeroes( trimTrailingZeroes );
555  }
556  // End of backward compatibility
557  }
558  else
559  {
560  // QGIS project 3.10+
561  method = QgsClassificationMethod::create( methodElem, context );
562  }
563 
564  // apply the method
565  r->setClassificationMethod( method );
566 
567  QDomElement rotationElem = element.firstChildElement( QStringLiteral( "rotation" ) );
568  if ( !rotationElem.isNull() && !rotationElem.attribute( QStringLiteral( "field" ) ).isEmpty() )
569  {
570  for ( const QgsRendererRange &range : qgis::as_const( r->mRanges ) )
571  {
572  convertSymbolRotation( range.symbol(), rotationElem.attribute( QStringLiteral( "field" ) ) );
573  }
574  if ( r->mSourceSymbol )
575  {
576  convertSymbolRotation( r->mSourceSymbol.get(), rotationElem.attribute( QStringLiteral( "field" ) ) );
577  }
578  }
579  QDomElement sizeScaleElem = element.firstChildElement( QStringLiteral( "sizescale" ) );
580  if ( !sizeScaleElem.isNull() && !sizeScaleElem.attribute( QStringLiteral( "field" ) ).isEmpty() )
581  {
582  for ( const QgsRendererRange &range : qgis:: as_const( r->mRanges ) )
583  {
584  convertSymbolSizeScale( range.symbol(),
585  QgsSymbolLayerUtils::decodeScaleMethod( sizeScaleElem.attribute( QStringLiteral( "scalemethod" ) ) ),
586  sizeScaleElem.attribute( QStringLiteral( "field" ) ) );
587  }
588  if ( r->mSourceSymbol && r->mSourceSymbol->type() == QgsSymbol::Marker )
589  {
590  convertSymbolSizeScale( r->mSourceSymbol.get(),
591  QgsSymbolLayerUtils::decodeScaleMethod( sizeScaleElem.attribute( QStringLiteral( "scalemethod" ) ) ),
592  sizeScaleElem.attribute( QStringLiteral( "field" ) ) );
593  }
594  }
595 
596  QDomElement ddsLegendSizeElem = element.firstChildElement( QStringLiteral( "data-defined-size-legend" ) );
597  if ( !ddsLegendSizeElem.isNull() )
598  {
599  r->mDataDefinedSizeLegend.reset( QgsDataDefinedSizeLegend::readXml( ddsLegendSizeElem, context ) );
600  }
601 // TODO: symbol levels
602  return r;
603 }
