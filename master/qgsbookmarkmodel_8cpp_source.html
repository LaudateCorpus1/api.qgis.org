44 QVariant QgsBookmarkManagerModel::data( const QModelIndex &index, int role ) const
45 {
46  if ( !index.isValid() )
47  return QVariant();
48 
49  QgsBookmark b = bookmarkForIndex( index );
50  const int managerCount = mManager->bookmarks().count();
51 
52  switch ( role )
53  {
54  case RoleExtent:
55  return b.extent();
56 
57  case RoleName:
58  return b.name();
59 
60  case RoleId:
61  return b.id();
62 
63  case RoleGroup:
64  return b.group();
65 
66  case Qt::DisplayRole:
67  case Qt::EditRole:
68  {
69  switch ( index.column() )
70  {
71  case ColumnName:
72  return b.name();
73  case ColumnGroup:
74  return b.group();
75  case ColumnXMin:
76  return b.extent().xMinimum();
77  case ColumnYMin:
78  return b.extent().yMinimum();
79  case ColumnXMax:
80  return b.extent().xMaximum();
81  case ColumnYMax:
82  return b.extent().yMaximum();
83  case ColumnCrs:
84  return b.extent().crs().authid();
85  case ColumnStore:
86  return QVariant();
87  }
88  break;
89  }
90 
91  case Qt::CheckStateRole:
92  {
93  if ( index.column() == ColumnStore )
94  return index.row() < managerCount ? Qt::Unchecked : Qt::Checked;
95  break;
96  }
97  }
98  return QVariant();
99 }

120 bool QgsBookmarkManagerModel::setData( const QModelIndex &index, const QVariant &value, int role )
121 {
122  if ( !index.isValid() )
123  return false;
124 
125  QgsBookmark b = bookmarkForIndex( index );
126  const int managerCount = mManager->bookmarks().count();
127 
128  switch ( role )
129  {
130  case Qt::EditRole:
131  {
132  QgsReferencedRectangle extent = b.extent();
133  switch ( index.column() )
134  {
135  case ColumnName:
136  b.setName( value.toString() );
137  break;
138  case ColumnGroup:
139  b.setGroup( value.toString() );
140  break;
141  case ColumnXMin:
142  {
143  bool ok = false;
144  extent.setXMinimum( value.toDouble( &ok ) );
145  if ( !ok )
146  return false;
147  break;
148  }
149  case ColumnYMin:
150  {
151  bool ok = false;
152  extent.setYMinimum( value.toDouble( &ok ) );
153  if ( !ok )
154  return false;
155  break;
156  }
157  case ColumnXMax:
158  {
159  bool ok = false;
160  extent.setXMaximum( value.toDouble( &ok ) );
161  if ( !ok )
162  return false;
163  break;
164  }
165  case ColumnYMax:
166  {
167  bool ok = false;
168  extent.setYMaximum( value.toDouble( &ok ) );
169  if ( !ok )
170  return false;
171  break;
172  }
173  case ColumnCrs:
174  {
175  QgsCoordinateReferenceSystem crs;
176  if ( !crs.createFromString( value.toString() ) )
177  return false;
178  extent.setCrs( crs );
179  break;
180  }
181  default:
182  return false;
183  }
184  b.setExtent( extent );
185  return index.row() < managerCount ? mManager->updateBookmark( b ) : mProjectManager->updateBookmark( b );
186  }
187 
188  case Qt::CheckStateRole:
189  {
190  if ( index.column() != ColumnStore )
191  return false;
192 
193  if ( index.row() < managerCount )
194  {
195  if ( value.toInt() != Qt::Checked )
196  return false;
197  return mManager->moveBookmark( b.id(), mProjectManager );
198  }
199  else
200  {
201  if ( value.toInt() != Qt::Unchecked )
202  return false;
203  return mProjectManager->moveBookmark( b.id(), mManager );
204  }
205  }
206  }
207 
208  return false;
209 }

211 bool QgsBookmarkManagerModel::insertRows( int, int count, const QModelIndex &parent )
212 {
213  // append
214  int oldCount = mManager->bookmarks().count();
215  beginInsertRows( parent, oldCount, oldCount + count );
216  bool result = true;
217  for ( int i = 0; i < count; ++i )
218  {
219  bool res = false;
220  mBlocked = true;
221  mManager->addBookmark( QgsBookmark(), &res );
222  mBlocked = false;
223  result &= res;
224  }
225  endInsertRows();
226  return result;
227 }

229 bool QgsBookmarkManagerModel::removeRows( int row, int count, const QModelIndex &parent )
230 {
231  beginRemoveRows( parent, row, row + count );
232 
233  QList< QgsBookmark > appBookmarks = mManager->bookmarks();
234  QList< QgsBookmark > projectBookmarks = mProjectManager->bookmarks();
235  for ( int r = row + count - 1; r >= row; --r )
236  {
237  if ( r >= appBookmarks.count() )
238  mProjectManager->removeBookmark( projectBookmarks.at( r - appBookmarks.size() ).id() );
239  else
240  mManager->removeBookmark( appBookmarks.at( r ).id() );
241  }
242  endRemoveRows();
243  return true;
244 }

293 void QgsBookmarkManagerModel::bookmarkAboutToBeRemoved( const QString &id )

355 QgsBookmark QgsBookmarkManagerModel::bookmarkForIndex( const QModelIndex &index ) const
356 {
357  if ( !index.isValid() )
358  return QgsBookmark();
359 
360  const int managerCount = mManager->bookmarks().count();
361  const int projectCount = mProjectManager->bookmarks().count();
362  if ( index.row() < managerCount )
363  return mManager->bookmarks().at( index.row() );
364  else if ( index.row() < managerCount + projectCount )
365  return mProjectManager->bookmarks().at( index.row() - managerCount );
366  return QgsBookmark();
367 }
