37 unsigned int QgsRangeWidgetFactory::fieldScore( const QgsVectorLayer *vl, int fieldIdx ) const
38 {
39  const QgsField field = vl->fields().at( fieldIdx );
40  if ( field.type() == QVariant::Int ) return 20;
41  if ( field.type() == QVariant::Double ) return 5; // low priority because the fixed number of decimal places may alter the original data
42  if ( field.isNumeric() ) return 5; // widgets used support only signed 32bits (int) and double
43  return 0;
44 }
