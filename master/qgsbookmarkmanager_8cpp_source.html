161 bool QgsBookmarkManager::removeBookmark( const QString &id )
162 {
163  if ( id.isEmpty() )
164  return false;
165 
166  QString group;
167  int pos = -1;
168  int i = 0;
169  for ( const QgsBookmark &b : qgis::as_const( mBookmarks ) )
170  {
171  if ( b.id() == id )
172  {
173  group = b.group();
174  pos = i;

176  }
177  i++;
178  }
179 
180  if ( pos < 0 )
181  return false;

185  if ( bookmarksByGroup( group ).isEmpty() )
186  mGroups.removeOne( group );
187  emit bookmarkRemoved( id );
188  if ( mProject )
189  {
190  mProject->setDirty( true );
191  }
192 
193  return true;
194 }

196 bool QgsBookmarkManager::updateBookmark( const QgsBookmark &bookmark )
197 {
198  // check for duplicate ID
199  int i = 0;
200  for ( const QgsBookmark &b : qgis::as_const( mBookmarks ) )
201  {
202  if ( b.id() == bookmark.id() )
203  {
204  if ( mBookmarks[i].group() != bookmark.group() )
205  {
206  if ( bookmarksByGroup( mBookmarks[i].group() ).count() == 1 )
207  mGroups.removeOne( mBookmarks[i].group() );
208  if ( !mGroups.contains( bookmark.group() ) )
209  mGroups << bookmark.group();
210  }
211  mBookmarks[i] = bookmark;
212  emit bookmarkChanged( bookmark.id() );
213  if ( mProject )
214  {
215  mProject->setDirty( true );
216  }
217 
218  return true;
219  }
220  i++;
221  }
222  return false;
223 }

265 bool QgsBookmarkManager::readXml( const QDomElement &element, const QDomDocument &doc )
266 {
267  clear();
268 
269  QDomElement bookmarksElem = element;
270  if ( element.tagName() != QStringLiteral( "Bookmarks" ) )
271  {
272  bookmarksElem = element.firstChildElement( QStringLiteral( "Bookmarks" ) );
273  }
274  bool result = true;
275  if ( mProject && bookmarksElem.isNull() )
276  {
277  // handle legacy projects
278  const int count = mProject->readNumEntry( QStringLiteral( "Bookmarks" ), QStringLiteral( "/count" ) );
279  for ( int i = 0; i < count; ++i )
280  {
281  const double minX = mProject->readDoubleEntry( QStringLiteral( "Bookmarks" ), QStringLiteral( "/Row-%1/MinX" ).arg( i ) );
282  const double minY = mProject->readDoubleEntry( QStringLiteral( "Bookmarks" ), QStringLiteral( "/Row-%1/MinY" ).arg( i ) );
283  const double maxX = mProject->readDoubleEntry( QStringLiteral( "Bookmarks" ), QStringLiteral( "/Row-%1/MaxX" ).arg( i ) );
284  const double maxY = mProject->readDoubleEntry( QStringLiteral( "Bookmarks" ), QStringLiteral( "/Row-%1/MaxY" ).arg( i ) );
285  const long srid = mProject->readNumEntry( QStringLiteral( "Bookmarks" ), QStringLiteral( "/Row-%1/SRID" ).arg( i ) );
286  QgsBookmark b;
287  b.setId( QStringLiteral( "bookmark_%1" ).arg( i ) );
288  b.setName( mProject->readEntry( QStringLiteral( "Bookmarks" ), QStringLiteral( "/Row-%1/Name" ).arg( i ) ) );
289  b.setExtent( QgsReferencedRectangle( QgsRectangle( minX, minY, maxX, maxY ), QgsCoordinateReferenceSystem::fromSrsId( srid ) ) );
290 
291  bool added = false;
292  addBookmark( b, &added );
293  result = added && result;
294  }
295  return result;
296  }
297 
298  //restore each
299  QDomNodeList bookmarkNodes = element.elementsByTagName( QStringLiteral( "Bookmark" ) );
300  for ( int i = 0; i < bookmarkNodes.size(); ++i )
301  {
302  QgsBookmark b = QgsBookmark::fromXml( bookmarkNodes.at( i ).toElement(), doc );
303  bool added = false;
304  addBookmark( b, &added );
305  result = added && result;
306  }
307 
308  return result;
309 }

405 bool QgsBookmarkManager::importFromFile( const QString &path )
406 {
407  if ( path.isEmpty() )
408  {
409  return false;
410  }
411 
412  QFile f( path );
413  if ( !f.open( QIODevice::ReadOnly | QIODevice::Text ) )

418  QDomDocument doc;

421  return false;

424 
425  QDomElement docElem = doc.documentElement();

427 
428  bool res = true;
429  for ( int i = 0; i < nodeList.count(); i++ )
430  {
431  QDomNode bookmark = nodeList.at( i );
432  QDomElement name = bookmark.firstChildElement( QStringLiteral( "name" ) );
433  QDomElement prjname = bookmark.firstChildElement( QStringLiteral( "project" ) );
434  QDomElement xmin = bookmark.firstChildElement( QStringLiteral( "xmin" ) );
435  QDomElement ymin = bookmark.firstChildElement( QStringLiteral( "ymin" ) );
436  QDomElement xmax = bookmark.firstChildElement( QStringLiteral( "xmax" ) );
437  QDomElement ymax = bookmark.firstChildElement( QStringLiteral( "ymax" ) );
438  QDomElement srid = bookmark.firstChildElement( QStringLiteral( "sr_id" ) );
439 
440  bool ok = false;
441  QgsBookmark b;
442  b.setName( name.text() );
443  b.setGroup( prjname.text() );
444  QgsCoordinateReferenceSystem crs;
445  crs.createFromSrsId( srid.text().toLongLong() );
446  b.setExtent( QgsReferencedRectangle( QgsRectangle( xmin.text().toDouble(),
447  ymin.text().toDouble(),
448  xmax.text().toDouble(),
449  ymax.text().toDouble() ), crs ) );
450  addBookmark( b, &ok );
451  res = res && ok;
452  }
453 
454  return res;

479 void QgsBookmarkManager::initialize( const QString &filePath )
480 {
481  if ( mInitialized )
482  return;
483 
484  mFilePath = filePath;
485 
486  mInitialized = true;
487 
488  // restore state
489  if ( !QFileInfo::exists( mFilePath ) )
490  {
491  //convert old bookmarks from db
492  sqlite3_database_unique_ptr database;
493  int result = database.open( QgsApplication::qgisUserDatabaseFilePath() );
494  if ( result != SQLITE_OK )
495  {
496  return;
497  }
498 
499  sqlite3_statement_unique_ptr preparedStatement = database.prepare( QStringLiteral( "SELECT name,project_name,xmin,ymin,xmax,ymax,projection_srid FROM tbl_bookmarks" ), result );
500  if ( result == SQLITE_OK )
501  {
502  while ( preparedStatement.step() == SQLITE_ROW )
503  {
504  const QString name = preparedStatement.columnAsText( 0 );
505  const QString group = preparedStatement.columnAsText( 1 );
506  const double xMin = preparedStatement.columnAsDouble( 2 );
507  const double yMin = preparedStatement.columnAsDouble( 3 );
508  const double xMax = preparedStatement.columnAsDouble( 4 );
509  const double yMax = preparedStatement.columnAsDouble( 5 );
510  const long long srid = preparedStatement.columnAsInt64( 6 );
511 
512  QgsBookmark b;
513  b.setName( name );
514  const QgsRectangle extent( xMin, yMin, xMax, yMax );
515  b.setExtent( QgsReferencedRectangle( extent, QgsCoordinateReferenceSystem::fromSrsId( srid ) ) );
516  addBookmark( b );
517  }
518  }
519  store();
520  }
521  else
522  {
523  QFile f( mFilePath );
524  if ( !f.open( QIODevice::ReadOnly | QIODevice::Text ) )
525  {
526  return;
527  }
528 
529  QDomDocument doc;
530  if ( !doc.setContent( &f ) )
531  {
532  return;
533  }
534  f.close();
535 
536  QDomElement elem = doc.documentElement();
537  readXml( elem, doc );
538  }
