171 QgsMeshDatasetValue QgsMeshLayer::datasetValue( const QgsMeshDatasetIndex &index, const QgsPointXY &point ) const
172 {
173  QgsMeshDatasetValue value;
174 
175  if ( mTriangularMesh && dataProvider() && dataProvider()->isValid() && index.isValid() )
176  {
177  int faceIndex = mTriangularMesh->faceIndexForPoint( point ) ;
178  if ( faceIndex >= 0 )
179  {
180  int nativeFaceIndex = mTriangularMesh->trianglesToNativeFaces().at( faceIndex );
181  if ( dataProvider()->isFaceActive( index, nativeFaceIndex ) )
182  {
183 
184  if ( dataProvider()->datasetGroupMetadata( index ).dataType() == QgsMeshDatasetGroupMetadata::DataOnFaces )
185  {
186  int nativeFaceIndex = mTriangularMesh->trianglesToNativeFaces().at( faceIndex );
187  value = dataProvider()->datasetValue( index, nativeFaceIndex );
188  }
189  else
190  {
191  const QgsMeshFace &face = mTriangularMesh->triangles()[faceIndex];
192  const int v1 = face[0], v2 = face[1], v3 = face[2];
193  const QgsPoint p1 = mTriangularMesh->vertices()[v1], p2 = mTriangularMesh->vertices()[v2], p3 = mTriangularMesh->vertices()[v3];
194  const QgsMeshDatasetValue val1 = dataProvider()->datasetValue( index, v1 );
195  const QgsMeshDatasetValue val2 = dataProvider()->datasetValue( index, v2 );
196  const QgsMeshDatasetValue val3 = dataProvider()->datasetValue( index, v3 );
197  const double x = QgsMeshLayerUtils::interpolateFromVerticesData( p1, p2, p3, val1.x(), val2.x(), val3.x(), point );
198  double y = std::numeric_limits<double>::quiet_NaN();
199  bool isVector = dataProvider()->datasetGroupMetadata( index ).isVector();
200  if ( isVector )
201  y = QgsMeshLayerUtils::interpolateFromVerticesData( p1, p2, p3, val1.y(), val2.y(), val3.y(), point );
202 
203  value = QgsMeshDatasetValue( x, y );
204  }
205 
206  }
207  }
208  }
209 
210  return value;
211 }

374 bool QgsMeshLayer::readXml( const QDomNode &layer_node, QgsReadWriteContext &context )
375 {
376  QgsDebugMsgLevel( QStringLiteral( "Datasource in QgsMeshLayer::readXml: %1" ).arg( mDataSource.toLocal8Bit().data() ), 3 );
377 
378  //process provider key
379  QDomNode pkeyNode = layer_node.namedItem( QStringLiteral( "provider" ) );
380 
381  if ( pkeyNode.isNull() )
382  {
383  mProviderKey.clear();
384  }
385  else
386  {
387  QDomElement pkeyElt = pkeyNode.toElement();
388  mProviderKey = pkeyElt.text();
389  }
390 
391  if ( mReadFlags & QgsMapLayer::FlagDontResolveLayers )
392  {
393  return false;
394  }
395 
396  QgsDataProvider::ProviderOptions providerOptions;
397  if ( !setDataProvider( mProviderKey, providerOptions ) )
398  {
399  return false;
400  }
401 
402  QDomElement elemExtraDatasets = layer_node.firstChildElement( QStringLiteral( "extra-datasets" ) );
403  if ( !elemExtraDatasets.isNull() )
404  {
405  QDomElement elemUri = elemExtraDatasets.firstChildElement( QStringLiteral( "uri" ) );
406  while ( !elemUri.isNull() )
407  {
408  QString uri = context.pathResolver().readPath( elemUri.text() );
409 
410  bool res = mDataProvider->addDataset( uri );
411 #ifdef QGISDEBUG
412  QgsDebugMsg( QStringLiteral( "extra dataset (res %1): %2" ).arg( res ).arg( uri ) );
413 #else
414  ( void )res; // avoid unused warning in release builds
415 #endif
416 
417  elemUri = elemUri.nextSiblingElement( QStringLiteral( "uri" ) );
418  }
419  }
420 
421  QString errorMsg;
422  readSymbology( layer_node, errorMsg, context );
423 
424  return mValid; // should be true if read successfully
425 }

427 bool QgsMeshLayer::writeXml( QDomNode &layer_node, QDomDocument &document, const QgsReadWriteContext &context ) const
428 {
429  // first get the layer element so that we can append the type attribute
430  QDomElement mapLayerNode = layer_node.toElement();
431 
432  if ( mapLayerNode.isNull() || ( QLatin1String( "maplayer" ) != mapLayerNode.nodeName() ) )
433  {
434  QgsDebugMsgLevel( QStringLiteral( "can't find <maplayer>" ), 2 );
435  return false;
436  }
437 
438  mapLayerNode.setAttribute( QStringLiteral( "type" ), QStringLiteral( "mesh" ) );
439 
440  // add provider node
441  if ( mDataProvider )
442  {
443  QDomElement provider = document.createElement( QStringLiteral( "provider" ) );
444  QDomText providerText = document.createTextNode( providerType() );
445  provider.appendChild( providerText );
446  layer_node.appendChild( provider );
447 
448  const QStringList extraDatasetUris = mDataProvider->extraDatasets();
449  QDomElement elemExtraDatasets = document.createElement( QStringLiteral( "extra-datasets" ) );
450  for ( const QString &uri : extraDatasetUris )
451  {
452  QString path = context.pathResolver().writePath( uri );
453  QDomElement elemUri = document.createElement( QStringLiteral( "uri" ) );
454  elemUri.appendChild( document.createTextNode( path ) );
455  elemExtraDatasets.appendChild( elemUri );
456  }
457  layer_node.appendChild( elemExtraDatasets );
458  }
459 
460  // renderer specific settings
461  QString errorMsg;
462  return writeSymbology( layer_node, document, errorMsg, context );
463 }

486 bool QgsMeshLayer::setDataProvider( QString const &provider, const QgsDataProvider::ProviderOptions &options )
487 {
488  delete mDataProvider;
489 
490  mProviderKey = provider;
491  QString dataSource = mDataSource;
492 
493  mDataProvider = qobject_cast<QgsMeshDataProvider *>( QgsProviderRegistry::instance()->createProvider( provider, dataSource, options ) );
494  if ( !mDataProvider )
495  {
496  QgsDebugMsgLevel( QStringLiteral( "Unable to get mesh data provider" ), 2 );
497  return false;
498  }
499 
500  mDataProvider->setParent( this );
501  QgsDebugMsgLevel( QStringLiteral( "Instantiated the mesh data provider plugin" ), 2 );
502 
503  mValid = mDataProvider->isValid();
504  if ( !mValid )
505  {
506  QgsDebugMsgLevel( QStringLiteral( "Invalid mesh provider plugin %1" ).arg( QString( mDataSource.toUtf8() ) ), 2 );
507  return false;
508  }
509 
510  setCrs( mDataProvider->crs() );
511 
512  if ( provider == QStringLiteral( "mesh_memory" ) )
513  {
514  // required so that source differs between memory layers
515  mDataSource = mDataSource + QStringLiteral( "&uid=%1" ).arg( QUuid::createUuid().toString() );
516  }
517 
518  for ( int i = 0; i < mDataProvider->datasetGroupCount(); ++i )
519  assignDefaultStyleToDatasetGroup( i );
520 
521  connect( mDataProvider, &QgsMeshDataProvider::dataChanged, this, &QgsMeshLayer::dataChanged );
522  connect( mDataProvider, &QgsMeshDataProvider::datasetGroupsAdded, this, &QgsMeshLayer::onDatasetGroupsAdded );
523 
524  return true;
