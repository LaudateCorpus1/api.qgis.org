31 QgsRectangle QgsServerApiUtils::parseBbox( const QString &bbox )
32 {
33  const auto parts { bbox.split( ',', QString::SplitBehavior::SkipEmptyParts ) };
34  // Note: Z is ignored
35  auto ok { true };
36  if ( parts.count() == 4 || parts.count() == 6 )
37  {
38  const auto hasZ { parts.count() == 6 };
39  auto toDouble = [ & ]( const int i ) -> double
40  {
41  if ( ! ok )
42  return 0;
43  return parts[i].toDouble( &ok );
44  };
45  QgsRectangle rect;
46  if ( hasZ )
47  {
48  rect = QgsRectangle( toDouble( 0 ), toDouble( 1 ),
49  toDouble( 3 ), toDouble( 4 ) );
50  }
51  else
52  {
53  rect = QgsRectangle( toDouble( 0 ), toDouble( 1 ),
54  toDouble( 2 ), toDouble( 3 ) );
55  }
56  if ( ok )
57  {
58  return rect;
59  }
60  }
61  return QgsRectangle();
62 }

73  return {{ extent.xMinimum(), extent.yMinimum(), extent.xMaximum(), extent.yMaximum() }};

76 QgsCoordinateReferenceSystem QgsServerApiUtils::parseCrs( const QString &bboxCrs )
77 {
78  QgsCoordinateReferenceSystem crs;
79  // We get this:
80  // http://www.opengis.net/def/crs/OGC/1.3/CRS84
81  // We want this:
82  // "urn:ogc:def:crs:<auth>:[<version>]:<code>"
83  const auto parts { QUrl( bboxCrs ).path().split( '/' ) };
84  if ( parts.count() == 6 )
85  {
86  return crs.fromOgcWmsCrs( QStringLiteral( "urn:ogc:def:crs:%1:%2:%3" ).arg( parts[3], parts[4], parts[5] ) );
87  }
88  else
89  {
90  return crs;
91  }
92 }

151 QString QgsServerApiUtils::crsToOgcUri( const QgsCoordinateReferenceSystem &crs )
152 {
153  const auto parts { crs.authid().split( ':' ) };
154  if ( parts.length() == 2 )
155  {
156  if ( parts[0] == QStringLiteral( "EPSG" ) )
157  return QStringLiteral( "http://www.opengis.net/def/crs/EPSG/9.6.2/%1" ).arg( parts[1] ) ;
158  else if ( parts[0] == QStringLiteral( "OGC" ) )
159  {
160  return QStringLiteral( "http://www.opengis.net/def/crs/OGC/1.3/%1" ).arg( parts[1] ) ;
161  }
162  else
163  {
164  QgsMessageLog::logMessage( QStringLiteral( "Error converting published CRS to URI %1: (not OGC or EPSG)" ).arg( crs.authid() ), QStringLiteral( "Server" ), Qgis::Critical );
165  }
166  }
167  else
168  {
169  QgsMessageLog::logMessage( QStringLiteral( "Error converting published CRS to URI: %1" ).arg( crs.authid() ), QStringLiteral( "Server" ), Qgis::Critical );
170  }
171  return QString();
172 }

174 QString QgsServerApiUtils::appendMapParameter( const QString &path, const QUrl &requestUrl )
175 {
176  QList<QPair<QString, QString> > qi;
177  QString result { path };
178  const auto constItems { requestUrl.queryItems( ) };
179  for ( const auto &i : constItems )
180  {
181  if ( i.first.compare( QStringLiteral( "MAP" ), Qt::CaseSensitivity::CaseInsensitive ) == 0 )
182  {
183  qi.push_back( i );
184  }
185  }
186  if ( ! qi.empty() )
187  {
188  if ( ! path.endsWith( '?' ) )
189  {
190  result += '?';
191  }
192  result.append( QStringLiteral( "MAP=%1" ).arg( qi.first().second ) );
193  }
194  return result;
195 }
