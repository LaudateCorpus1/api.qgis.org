225 inline QString qgsDoubleToString( double a, int precision = 17 )
226 {
227  if ( precision )
228  {
229  QString str = QString::number( a, 'f', precision );
230  if ( str.contains( QLatin1Char( '.' ) ) )
231  {
232  // remove ending 0s
233  int idx = str.length() - 1;
234  while ( str.at( idx ) == '0' && idx > 1 )
235  {
236  idx--;
237  }
238  if ( idx < str.length() - 1 )
239  str.truncate( str.at( idx ) == '.' ? idx : idx + 1 );
240  }
241  return str;
242  }
243  else
244  {
245  // avoid printing -0
246  // see https://bugreports.qt.io/browse/QTBUG-71439
247  const QString str( QString::number( a, 'f', precision ) );
248  if ( str == QLatin1String( "-0" ) )
249  {
250  return QLatin1String( "0" );
251  }
252  else
253  {
254  return str;
255  }
256  }
257 }

265 inline bool qgsDoubleNear( double a, double b, double epsilon = 4 * std::numeric_limits<double>::epsilon() )
266 {
267  if ( std::isnan( a ) || std::isnan( b ) )
268  return std::isnan( a ) && std::isnan( b ) ;
269 
270  const double diff = a - b;
271  return diff > -epsilon && diff <= epsilon;
272 }

290 inline bool qgsDoubleNearSig( double a, double b, int significantDigits = 10 )
291 {
292  if ( std::isnan( a ) || std::isnan( b ) )
293  return std::isnan( a ) && std::isnan( b ) ;
294 
295  // The most simple would be to print numbers as %.xe and compare as strings
296  // but that is probably too costly
297  // Then the fastest would be to set some bits directly, but little/big endian
298  // has to be considered (maybe TODO)
299  // Is there a better way?
300  int aexp, bexp;
301  double ar = std::frexp( a, &aexp );
302  double br = std::frexp( b, &bexp );
303 
304  return aexp == bexp &&
305  std::round( ar * std::pow( 10.0, significantDigits ) ) == std::round( br * std::pow( 10.0, significantDigits ) );
306 }

313 inline double qgsRound( double number, int places )
314 {
315  double m = ( number < 0.0 ) ? -1.0 : 1.0;
316  double scaleFactor = std::pow( 10.0, places );
317  return ( std::round( number * m * scaleFactor ) / scaleFactor ) * m;
318 }

419 template<class T> const QMap<T, QString> qgsEnumMap() SIP_SKIP
420 {
421  QMetaEnum metaEnum = QMetaEnum::fromType<T>();
422  Q_ASSERT( metaEnum.isValid() );
423  QMap<T, QString> enumMap;
424  for ( int idx = 0; idx < metaEnum.keyCount(); ++idx )
425  {
426  const char *enumKey = metaEnum.key( idx );
427  enumMap.insert( static_cast<T>( metaEnum.keyToValue( enumKey ) ), QString( enumKey ) );
428  }
429  return enumMap;
430 }
