81 QSizeF QgsDiagram::sizeForValue( double value, const QgsDiagramSettings &s, const QgsDiagramInterpolationSettings &is ) const
82 {
83  double scaledValue = value;
84  double scaledLowerValue = is.lowerValue;
85  double scaledUpperValue = is.upperValue;
86 
87  // interpolate the squared value if scale by area
88  if ( s.scaleByArea )
89  {
90  scaledValue = std::sqrt( scaledValue );
91  scaledLowerValue = std::sqrt( scaledLowerValue );
92  scaledUpperValue = std::sqrt( scaledUpperValue );
93  }
94 
95  //interpolate size
96  double scaledRatio = ( scaledValue - scaledLowerValue ) / ( scaledUpperValue - scaledLowerValue );
97 
98  QSizeF size = QSizeF( is.upperSize.width() * scaledRatio + is.lowerSize.width() * ( 1 - scaledRatio ),
99  is.upperSize.height() * scaledRatio + is.lowerSize.height() * ( 1 - scaledRatio ) );
100 
101  // Scale, if extension is smaller than the specified minimum
102  if ( size.width() <= s.minimumSize && size.height() <= s.minimumSize )
103  {
104  bool p = false; // preserve height == width
105  if ( qgsDoubleNear( size.width(), size.height() ) )
106  p = true;
107 
108  size.scale( s.minimumSize, s.minimumSize, Qt::KeepAspectRatio );
109 
110  // If height == width, recover here (overwrite floating point errors)
111  if ( p )
112  size.setWidth( size.height() );
113  }
114 
115  return size;
116 }
