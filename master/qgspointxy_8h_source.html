388 inline bool operator==( const QgsPointXY &p1, const QgsPointXY &p2 ) SIP_SKIP
389 {
390  const bool nan1X = std::isnan( p1.x() );
391  const bool nan2X = std::isnan( p2.x() );
392  if ( nan1X != nan2X )
393  return false;
394  if ( !nan1X && !qgsDoubleNear( p1.x(), p2.x(), 1E-8 ) )
395  return false;
396

399  if ( nan1Y != nan2Y )
