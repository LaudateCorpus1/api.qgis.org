49 void QgsLegendStyle::writeXml( const QString &name, QDomElement &elem, QDomDocument &doc ) const
50 {
51  if ( elem.isNull() )
52  return;
53 
54  QDomElement styleElem = doc.createElement( QStringLiteral( "style" ) );
55 
56  styleElem.setAttribute( QStringLiteral( "name" ), name );
57  styleElem.setAttribute( QStringLiteral( "alignment" ), QString::number( mAlignment ) );
58 
59  if ( !qgsDoubleNear( mMarginMap[Top], 0.0 ) )
60  styleElem.setAttribute( QStringLiteral( "marginTop" ), QString::number( mMarginMap[Top] ) );
61  if ( !qgsDoubleNear( mMarginMap[Bottom], 0.0 ) )
62  styleElem.setAttribute( QStringLiteral( "marginBottom" ), QString::number( mMarginMap[Bottom] ) );
63  if ( !qgsDoubleNear( mMarginMap[Left], 0.0 ) )
64  styleElem.setAttribute( QStringLiteral( "marginLeft" ), QString::number( mMarginMap[Left] ) );
65  if ( !qgsDoubleNear( mMarginMap[Right], 0.0 ) )
66  styleElem.setAttribute( QStringLiteral( "marginRight" ), QString::number( mMarginMap[Right] ) );
67 
68  styleElem.appendChild( QgsFontUtils::toXmlElement( mFont, doc, QStringLiteral( "styleFont" ) ) );
69 
70  elem.appendChild( styleElem );
71 }

73 void QgsLegendStyle::readXml( const QDomElement &elem, const QDomDocument &doc )
74 {
75  Q_UNUSED( doc )
76  if ( elem.isNull() ) return;
77 
78  if ( !QgsFontUtils::setFromXmlChildNode( mFont, elem, QStringLiteral( "styleFont" ) ) )
79  {
80  mFont.fromString( elem.attribute( QStringLiteral( "font" ) ) );
81  }
82 
83  mMarginMap[Top] = elem.attribute( QStringLiteral( "marginTop" ), QStringLiteral( "0" ) ).toDouble();
84  mMarginMap[Bottom] = elem.attribute( QStringLiteral( "marginBottom" ), QStringLiteral( "0" ) ).toDouble();
85  mMarginMap[Left] = elem.attribute( QStringLiteral( "marginLeft" ), QStringLiteral( "0" ) ).toDouble();
86  mMarginMap[Right] = elem.attribute( QStringLiteral( "marginRight" ), QStringLiteral( "0" ) ).toDouble();
87 
88  mAlignment = static_cast< Qt::Alignment >( elem.attribute( QStringLiteral( "alignment" ), QString::number( Qt::AlignLeft ) ).toInt() );
