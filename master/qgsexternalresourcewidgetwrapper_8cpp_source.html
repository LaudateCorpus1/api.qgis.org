80 void QgsExternalResourceWidgetWrapper::setFeature( const QgsFeature &feature )
81 {
82  if ( mQgsWidget && mPropertyCollection.hasActiveProperties() )
83  {
84  QgsExpressionContext expressionContext( QgsExpressionContextUtils::globalProjectLayerScopes( layer() ) );
85  expressionContext.setFeature( feature );
86  bool ok = false;
87 
88  if ( mPropertyCollection.isActive( QgsEditorWidgetWrapper::RootPath ) )
89  {
90  QString path = mPropertyCollection.valueAsString( QgsEditorWidgetWrapper::RootPath, expressionContext, QString(), &ok );
91  if ( ok )
92  {
93  mQgsWidget->setDefaultRoot( path );
94  }
95  }
96  if ( mPropertyCollection.isActive( QgsEditorWidgetWrapper::DocumentViewerContent ) )
97  {
98  QString dvcString = mPropertyCollection.valueAsString( QgsEditorWidgetWrapper::DocumentViewerContent, expressionContext, QStringLiteral( "NoContent" ), &ok );
99  if ( ok )
100  {
101  QgsExternalResourceWidget::DocumentViewerContent dvc;
102  if ( dvcString.compare( QLatin1String( "image" ), Qt::CaseInsensitive ) == 0 )
103  {
104  dvc = QgsExternalResourceWidget::Image;
105  }
106  else if ( dvcString.compare( QLatin1String( "web" ), Qt::CaseInsensitive ) == 0 )
107  {
108  dvc = QgsExternalResourceWidget::Web;
109  }
110  else
111  {
112  dvc = QgsExternalResourceWidget::NoContent;
113  }
114  mQgsWidget->setDocumentViewerContent( dvc );
115  }
116  }
117  }
118 
119  QgsEditorWidgetWrapper::setFeature( feature );
120 }

127 void QgsExternalResourceWidgetWrapper::initWidget( QWidget *editor )
128 {
129  mLineEdit = qobject_cast<QLineEdit *>( editor );
130  mLabel = qobject_cast<QLabel *>( editor );
131  mQgsWidget = qobject_cast<QgsExternalResourceWidget *>( editor );
132 
133  if ( mLineEdit )
134  {
135  QgsFilterLineEdit *fle = qobject_cast<QgsFilterLineEdit *>( editor );
136  if ( fle )
137  {
138  fle->setNullValue( QgsApplication::nullRepresentation() );
139  }
140  }
141  else
142  mLineEdit = editor->findChild<QLineEdit *>();
143 
144  if ( mQgsWidget )
145  {
146  mQgsWidget->fileWidget()->setStorageMode( QgsFileWidget::GetFile );
147 
148  QVariantMap cfg = config();
149 
150  if ( cfg.contains( QStringLiteral( "UseLink" ) ) )
151  {
152  mQgsWidget->fileWidget()->setUseLink( cfg.value( QStringLiteral( "UseLink" ) ).toBool() );
153  }
154  if ( cfg.contains( QStringLiteral( "FullUrl" ) ) )
155  {
156  mQgsWidget->fileWidget()->setFullUrl( cfg.value( QStringLiteral( "FullUrl" ) ).toBool() );
157  }
158 
159  mPropertyCollection.loadVariant( cfg.value( QStringLiteral( "PropertyCollection" ) ), propertyDefinitions() );
160  if ( !mPropertyCollection.isActive( QgsWidgetWrapper::RootPath ) )
161  {
162  mQgsWidget->setDefaultRoot( cfg.value( QStringLiteral( "DefaultRoot" ) ).toString() );
163  }
164  if ( cfg.contains( QStringLiteral( "StorageMode" ) ) )
165  {
166  mQgsWidget->fileWidget()->setStorageMode( ( QgsFileWidget::StorageMode )cfg.value( QStringLiteral( "StorageMode" ) ).toInt() );
167  }
168  if ( cfg.contains( QStringLiteral( "RelativeStorage" ) ) )
169  {
170  mQgsWidget->setRelativeStorage( ( QgsFileWidget::RelativeStorage )cfg.value( QStringLiteral( "RelativeStorage" ) ).toInt() );
171  }
172  if ( cfg.contains( QStringLiteral( "FileWidget" ) ) )
173  {
174  mQgsWidget->setFileWidgetVisible( cfg.value( QStringLiteral( "FileWidget" ) ).toBool() );
175  }
176  if ( cfg.contains( QStringLiteral( "FileWidgetButton" ) ) )
177  {
178  // Prevent from showing the button in the attribute table, see https://github.com/qgis/QGIS/issues/26948
179  mQgsWidget->fileWidget()->setFileWidgetButtonVisible( cfg.value( QStringLiteral( "FileWidgetButton" ) ).toBool() && context().formMode() != QgsAttributeEditorContext::Popup );
180  }
181  if ( cfg.contains( QStringLiteral( "DocumentViewer" ) ) )
182  {
183  mQgsWidget->setDocumentViewerContent( ( QgsExternalResourceWidget::DocumentViewerContent )cfg.value( QStringLiteral( "DocumentViewer" ) ).toInt() );
184  }
185  if ( cfg.contains( QStringLiteral( "FileWidgetFilter" ) ) )
186  {
187  mQgsWidget->fileWidget()->setFilter( cfg.value( QStringLiteral( "FileWidgetFilter" ) ).toString() );
188  }
189  }
190 
191  if ( mLineEdit )
192  connect( mLineEdit, &QLineEdit::textChanged, this, [ = ]( const QString & value )
193  {
194  Q_NOWARN_DEPRECATED_PUSH
195  emit valueChanged( value );
196  Q_NOWARN_DEPRECATED_POP
197  emit valuesChanged( value );
198  } );
199 
200 }

248 void QgsExternalResourceWidgetWrapper::updateConstraintWidgetStatus()
249 {
250  if ( mLineEdit )
251  {
252  if ( !constraintResultVisible() )
253  {
254  widget()->setStyleSheet( QString() );
255  }
256  else
257  {
258  switch ( constraintResult() )
259  {
260  case ConstraintResultPass:
261  mLineEdit->setStyleSheet( QString() );
262  break;
263 
264  case ConstraintResultFailHard:
265  mLineEdit->setStyleSheet( QStringLiteral( "QgsFilterLineEdit { background-color: #dd7777; }" ) );
266  break;
267 
268  case ConstraintResultFailSoft:
269  mLineEdit->setStyleSheet( QStringLiteral( "QgsFilterLineEdit { background-color: #ffd85d; }" ) );
270  break;
271  }
272  }
273  }
