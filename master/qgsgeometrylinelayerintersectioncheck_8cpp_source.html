23 void QgsGeometryLineLayerIntersectionCheck::collectErrors( const QMap<QString, QgsFeaturePool *> &featurePools, QList<QgsGeometryCheckError *> &errors, QStringList &messages, QgsFeedback *feedback, const LayerFeatureIds &ids ) const
24 {
25  Q_UNUSED( messages )
26 
27  QMap<QString, QgsFeatureIds> featureIds = ids.isEmpty() ? allLayerFeatureIds( featurePools ) : ids.toMap();
28  featureIds.remove( mCheckLayer ); // Don't check layer against itself
29  QgsGeometryCheckerUtils::LayerFeatures layerFeatures( featurePools, featureIds, compatibleGeometryTypes(), feedback, mContext, true );
30  for ( const QgsGeometryCheckerUtils::LayerFeature &layerFeature : layerFeatures )
31  {
32  const QgsAbstractGeometry *geom = layerFeature.geometry().constGet();
33  for ( int iPart = 0, nParts = geom->partCount(); iPart < nParts; ++iPart )
34  {
35  const QgsLineString *line = dynamic_cast<const QgsLineString *>( QgsGeometryCheckerUtils::getGeomPart( geom, iPart ) );
36  if ( !line )
37  {
38  // Should not happen
39  continue;
40  }
41 
42  // Check whether the line intersects with any other features of the specified layer
43  QgsGeometryCheckerUtils::LayerFeatures checkFeatures( featurePools, QStringList() << mCheckLayer, line->boundingBox(), {QgsWkbTypes::LineGeometry, QgsWkbTypes::PolygonGeometry}, mContext );
44  for ( const QgsGeometryCheckerUtils::LayerFeature &checkFeature : checkFeatures )
45  {
46  const QgsAbstractGeometry *testGeom = checkFeature.geometry().constGet();
47  for ( int jPart = 0, mParts = testGeom->partCount(); jPart < mParts; ++jPart )
48  {
49  const QgsAbstractGeometry *part = QgsGeometryCheckerUtils::getGeomPart( testGeom, jPart );
50  if ( const QgsLineString *testLine = dynamic_cast<const QgsLineString *>( part ) )
51  {
52  const QList< QgsPoint > intersections = QgsGeometryCheckerUtils::lineIntersections( line, testLine, mContext->tolerance );
53  for ( const QgsPoint &inter : intersections )
54  {
55  errors.append( new QgsGeometryCheckError( this, layerFeature, inter, QgsVertexId( iPart ), checkFeature.id() ) );
56  }
57  }
58  else if ( const QgsPolygon *polygon = dynamic_cast<const QgsPolygon *>( part ) )
59  {
60  const QList< const QgsLineString * > rings = QgsGeometryCheckerUtils::polygonRings( polygon );
61  for ( const QgsLineString *ring : rings )
62  {
63  const QList< QgsPoint > intersections = QgsGeometryCheckerUtils::lineIntersections( line, ring, mContext->tolerance );
64  for ( const QgsPoint &inter : intersections )
65  {
66  errors.append( new QgsGeometryCheckError( this, layerFeature, inter, QgsVertexId( iPart ), checkFeature.id() ) );
67  }
68  }
69  }
70  }
71  }
72  }
73  }
74 }
