70 {
71  // GML2 does not support curves

74  if ( isEmpty() )

113 json QgsMultiSurface::asJsonObject( int precision ) const
114 {
115  json polygons( json::array( ) );
116  for ( const QgsAbstractGeometry *geom : qgis::as_const( mGeometries ) )
117  {
118  if ( qgsgeometry_cast<const QgsSurface *>( geom ) )
119  {
120  json coordinates( json::array( ) );
121  std::unique_ptr< QgsPolygon >polygon( static_cast<const QgsSurface *>( geom )->surfaceToPolygon() );
122  std::unique_ptr< QgsLineString > exteriorLineString( polygon->exteriorRing()->curveToLine() );
123  QgsPointSequence exteriorPts;
124  exteriorLineString->points( exteriorPts );
125  coordinates.push_back( QgsGeometryUtils::pointsToJson( exteriorPts, precision ) );
126 
127  std::unique_ptr< QgsLineString > interiorLineString;
128  for ( int i = 0, n = polygon->numInteriorRings(); i < n; ++i )
129  {
130  interiorLineString.reset( polygon->interiorRing( i )->curveToLine() );
131  QgsPointSequence interiorPts;
132  interiorLineString->points( interiorPts );
133  coordinates.push_back( QgsGeometryUtils::pointsToJson( interiorPts, precision ) );
134  }
135  polygons.push_back( coordinates );
136  }
137  }
138  return
139  {
140  { "type", "MultiPolygon" },
141  { "coordinates", polygons }
142  };

145 bool QgsMultiSurface::addGeometry( QgsAbstractGeometry *g )
146 {
147  if ( !qgsgeometry_cast<QgsSurface *>( g ) )
148  {
149  delete g;
150  return false;
151  }
152 
153  if ( mGeometries.empty() )
154  {
155  setZMTypeFromSubGeometry( g, QgsWkbTypes::MultiSurface );
156  }
157  if ( is3D() && !g->is3D() )
158  g->addZValue();
159  else if ( !is3D() && g->is3D() )
160  g->dropZValue();
161  if ( isMeasure() && !g->isMeasure() )
162  g->addMValue();
163  else if ( !isMeasure() && g->isMeasure() )
164  g->dropMValue();
165 
166  return QgsGeometryCollection::addGeometry( g );
167 }

180 QgsAbstractGeometry *QgsMultiSurface::boundary() const
181 {
182  std::unique_ptr< QgsMultiCurve > multiCurve( new QgsMultiCurve() );
183  multiCurve->reserve( mGeometries.size() );
184  for ( int i = 0; i < mGeometries.size(); ++i )
185  {
186  if ( QgsSurface *surface = qgsgeometry_cast<QgsSurface *>( mGeometries.at( i ) ) )
187  {
188  multiCurve->addGeometry( surface->boundary() );
189  }
190  }
191  if ( multiCurve->numGeometries() == 0 )
192  {
193  return nullptr;
194  }
195  return multiCurve.release();
196 }
