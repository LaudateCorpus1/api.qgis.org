132 QStringList QgsSymbolLayerV2Registry::symbolLayersForType( QgsSymbolV2::SymbolType type )
133 {
134  QStringList lst;
135  QMap<QString, QgsSymbolLayerV2AbstractMetadata*>::ConstIterator it = mMetadata.constBegin();
136  for ( ; it != mMetadata.constEnd(); ++it )
137  {
138  if ( it.value()->type() == type || it.value()->type() == QgsSymbolV2::Hybrid )
139  lst.append( it.key() );
