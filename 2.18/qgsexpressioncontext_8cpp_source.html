482 QgsExpressionContextScope* QgsExpressionContextUtils::globalScope()
483 {
484  QgsExpressionContextScope* scope = new QgsExpressionContextScope( QObject::tr( "Global" ) );
485 
486  //read values from QSettings
487  QSettings settings;
488 
489  //check if settings contains any variables
490  if ( settings.contains( QString( "/variables/values" ) ) )
491  {
492  QList< QVariant > customVariableVariants = settings.value( QString( "/variables/values" ) ).toList();
493  QList< QVariant > customVariableNames = settings.value( QString( "/variables/names" ) ).toList();
494  int variableIndex = 0;
495  for ( QList< QVariant >::const_iterator it = customVariableVariants.constBegin();
496  it != customVariableVariants.constEnd(); ++it )
497  {
498  if ( variableIndex >= customVariableNames.length() )
499  {
500  break;
501  }
502 
503  QVariant value = ( *it );
504  QString name = customVariableNames.at( variableIndex ).toString();
505 
506  scope->setVariable( name, value );
507  variableIndex++;
508  }
509  }
510 
511  //add some extra global variables
512  scope->addVariable( QgsExpressionContextScope::StaticVariable( "qgis_version", QGis::QGIS_VERSION, true ) );
513  scope->addVariable( QgsExpressionContextScope::StaticVariable( "qgis_version_no", QGis::QGIS_VERSION_INT, true ) );
514  scope->addVariable( QgsExpressionContextScope::StaticVariable( "qgis_release_name", QGis::QGIS_RELEASE_NAME, true ) );
515  scope->addVariable( QgsExpressionContextScope::StaticVariable( "qgis_platform", QgsApplication::platform(), true ) );
516  scope->addVariable( QgsExpressionContextScope::StaticVariable( "qgis_os_name", QgsApplication::osName(), true ) );
517  scope->addVariable( QgsExpressionContextScope::StaticVariable( "user_account_name", QgsApplication::userLoginName(), true ) );
518  scope->addVariable( QgsExpressionContextScope::StaticVariable( "user_full_name", QgsApplication::userFullName(), true ) );
519 
520  return scope;
