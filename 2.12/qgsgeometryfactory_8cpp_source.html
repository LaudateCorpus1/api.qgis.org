32 QgsAbstractGeometryV2* QgsGeometryFactory::geomFromWkb( const unsigned char* wkb )
33 {
34  if ( !wkb )
35  {
36  return 0;
37  }

39  //find out type (bytes 2-5)

41  memcpy( &type, wkb + 1, sizeof( int ) );

43 
44  geom = geomFromWkbType( QgsWKBTypes::Type( type ) );
