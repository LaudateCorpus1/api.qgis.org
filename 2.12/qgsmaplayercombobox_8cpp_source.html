31 void QgsMapLayerComboBox::setLayer( QgsMapLayer *layer )
32 {
33  QModelIndex idx = mProxyModel->sourceLayerModel()->indexFromLayer( layer );

35  {
36  QModelIndex proxyIdx = mProxyModel->mapFromSource( idx );
