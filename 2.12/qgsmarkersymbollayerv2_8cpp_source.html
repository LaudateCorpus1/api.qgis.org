808 bool QgsSimpleMarkerSymbolLayerV2::writeDxf( QgsDxfExport& e, double mmMapUnitScaleFactor, const QString& layerName, QgsSymbolV2RenderContext *context, const QgsFeature*, const QPointF& shift ) const
809 {
810  //data defined size?
811  double size = mSize;
812 
813  bool hasDataDefinedSize = false;
814  if ( context )
815  {
816  hasDataDefinedSize = context->renderHints() & QgsSymbolV2::DataDefinedSizeScale || hasDataDefinedProperty( QgsSymbolLayerV2::EXPR_SIZE );
817  }
818 
819  //data defined size
820  bool ok = true;
821  if ( hasDataDefinedSize )
822  {
823  if ( hasDataDefinedProperty( QgsSymbolLayerV2::EXPR_SIZE ) )
824  {
825  context->setOriginalValueVariable( mSize );
826  size = evaluateDataDefinedProperty( QgsSymbolLayerV2::EXPR_SIZE, *context, mSize, &ok ).toDouble();
827  }
828 
829  if ( ok )
830  {
831  switch ( mScaleMethod )
832  {
833  case QgsSymbolV2::ScaleArea:
834  size = sqrt( size );
835  break;
836  case QgsSymbolV2::ScaleDiameter:
837  break;
838  }
839  }
840 
841  size = QgsSymbolLayerV2Utils::convertToPainterUnits( context->renderContext(), size, mSizeUnit, mSizeMapUnitScale );
842  }
843  if ( mSizeUnit == QgsSymbolV2::MM )
844  {
845  size *= mmMapUnitScaleFactor;
846  }

848

851

859  outlineWidth *= mmMapUnitScaleFactor;
860  }
861

864  QColor bc = mBrush.color();
865  if ( context && hasDataDefinedProperty( QgsSymbolLayerV2::EXPR_COLOR ) )
866  {

876  if ( ok )
877  pc = QgsSymbolLayerV2Utils::decodeColor( colorString );
878  }
879 
880  //offset
881  double offsetX = 0;
882  double offsetY = 0;
883  if ( context )
884  {
885  markerOffset( *context, offsetX, offsetY );
886  }
887  QPointF off( offsetX, offsetY );
888 
889  //angle
890  double angle = mAngle + mLineAngle;
891  if ( context && hasDataDefinedProperty( QgsSymbolLayerV2::EXPR_ANGLE ) )
892  {
893  context->setOriginalValueVariable( mAngle );
894  angle = evaluateDataDefinedProperty( QgsSymbolLayerV2::EXPR_ANGLE, *context, mAngle ).toDouble() + mLineAngle;
895  }

902  }

905  if ( angle )
906  off = _rotatedOffset( off, angle );

909  {
910  off *= mmMapUnitScaleFactor;
911  }

914  t.translate( shift.x() + offsetX, shift.y() + offsetY );
915 
916  if ( angle != 0 )
917  t.rotate( angle );

921  {
922  t.scale( halfSize, -halfSize );

928  p[0].resize( polygon.size() + 1 );

932  p[0][i] = p[0][0];

935  e.writePolygon( p, layerName, "SOLID", bc );
