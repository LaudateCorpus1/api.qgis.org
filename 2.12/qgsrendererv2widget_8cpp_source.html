303  a->setCheckable( true );

309  menu->addAction( a );
310  }
311  }
312 
313  if ( !hasField )

315  if ( fieldName.isEmpty() )
316  {
317  aNo->setChecked( true );
318  }
319  else
320  {
321  aExpr->setChecked( true );
