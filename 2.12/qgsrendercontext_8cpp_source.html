46 QgsRenderContext QgsRenderContext::fromMapSettings( const QgsMapSettings& mapSettings )
47 {
48  QgsRenderContext ctx;
49  ctx.setMapToPixel( mapSettings.mapToPixel() );
50  ctx.setExtent( mapSettings.visibleExtent() );
51  ctx.setDrawEditingInformation( mapSettings.testFlag( QgsMapSettings::DrawEditingInfo ) );
52  ctx.setForceVectorOutput( mapSettings.testFlag( QgsMapSettings::ForceVectorOutput ) );
53  ctx.setUseAdvancedEffects( mapSettings.testFlag( QgsMapSettings::UseAdvancedEffects ) );
54  ctx.setUseRenderingOptimization( mapSettings.testFlag( QgsMapSettings::UseRenderingOptimization ) );
55  ctx.setCoordinateTransform( 0 );
56  ctx.setSelectionColor( mapSettings.selectionColor() );
57  ctx.setShowSelection( mapSettings.testFlag( QgsMapSettings::DrawSelection ) );
58  ctx.setRasterScaleFactor( 1.0 );
59  ctx.setScaleFactor( mapSettings.outputDpi() / 25.4 ); // = pixels per mm
60  ctx.setRendererScale( mapSettings.scale() );
61  ctx.setExpressionContext( mapSettings.expressionContext() );
62 
63  //this flag is only for stopping during the current rendering progress,
64  //so must be false at every new render operation
65  ctx.setRenderingStopped( false );
66 
67  return ctx;
68 }
