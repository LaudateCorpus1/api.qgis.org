33 bool QgsVectorLayerEditPassthrough::modify( QgsVectorLayerUndoPassthroughCommand *cmd )
34 {
35  L->undoStack()->push( cmd ); // push takes owneship -> no need for cmd to be a smart ptr
36  if ( cmd->hasError() )
37  return false;
38 
39  if ( !mModified )
40  {
41  mModified = true;
42  emit layerModified();
43  }
44 
45  return true;
46 }
