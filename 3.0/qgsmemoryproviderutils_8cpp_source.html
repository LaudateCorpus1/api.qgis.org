53 QgsVectorLayer *QgsMemoryProviderUtils::createMemoryLayer( const QString &name, const QgsFields &fields, QgsWkbTypes::Type geometryType, const QgsCoordinateReferenceSystem &crs )
54 {
55  QString geomType = QgsWkbTypes::displayString( geometryType );
56  if ( geomType.isNull() )
57  geomType = QStringLiteral( "none" );
58 
59  QStringList parts;
60  if ( crs.isValid() )
61  {
62  parts << QStringLiteral( "crs=" ) + crs.authid();
63  }
64  for ( const auto &field : fields )
65  {
66  QString lengthPrecision;
67  if ( field.length() > 0 && field.precision() > 0 )
68  {
69  lengthPrecision = QStringLiteral( "(%1,%2)" ).arg( field.length() ).arg( field.precision() );
70  }
71  else if ( field.length() > 0 )
72  {
73  lengthPrecision = QStringLiteral( "(%1)" ).arg( field.length() );
74  }
75  parts << QStringLiteral( "field=%1:%2%3" ).arg( field.name(), memoryLayerFieldType( field.type() ), lengthPrecision );
76  }
77 
78  QString uri = geomType + '?' + parts.join( '&' );
79 
80  return new QgsVectorLayer( uri, name, QStringLiteral( "memory" ) );
