80  QgsRuleBasedRenderer::RenderLevel &operator=( const QgsRuleBasedRenderer::RenderLevel &rh )
81  {
82  zIndex = rh.zIndex;
83  qDeleteAll( jobs );
84  jobs.clear();
85  Q_FOREACH ( RenderJob *job, rh.jobs )

87  jobs << new RenderJob( *job );

89  return *this;
90  }
