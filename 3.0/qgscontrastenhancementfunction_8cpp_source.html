38 int QgsContrastEnhancementFunction::enhance( double value )
39 {
40  if ( mQgsRasterDataType == Qgis::Byte )
41  {
42  return static_cast<int>( value );
43  }
44  else
45  {
46  return static_cast<int>( ( ( ( value - QgsContrastEnhancement::minimumValuePossible( mQgsRasterDataType ) ) / ( QgsContrastEnhancement::maximumValuePossible( mQgsRasterDataType ) - QgsContrastEnhancement::minimumValuePossible( mQgsRasterDataType ) ) ) * 255.0 ) );
47  }
48 }

50 bool QgsContrastEnhancementFunction::isValueInDisplayableRange( double value )
51 {
52  //A default check is to see if the provided value is with the range for the data type
