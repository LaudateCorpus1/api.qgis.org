34 std::unique_ptr<QgsAbstractGeometry> QgsGeometryFactory::geomFromWkb( QgsConstWkbPtr &wkbPtr )
35 {
36  if ( !wkbPtr )
37  return nullptr;
38 
39  //find out type (bytes 2-5)
40  QgsWkbTypes::Type type = QgsWkbTypes::Unknown;
41  try
42  {
43  type = wkbPtr.readHeader();
44  }
45  catch ( const QgsWkbException &e )
46  {
47  Q_UNUSED( e );
48  QgsDebugMsg( "WKB exception while reading header: " + e.what() );
49  return nullptr;
50  }
51  wkbPtr -= 1 + sizeof( int );
52 
53  std::unique_ptr< QgsAbstractGeometry > geom = geomFromWkbType( type );
54 
55  if ( geom )
56  {
57  try
58  {
59  geom->fromWkb( wkbPtr ); // also updates wkbPtr
60  }
61  catch ( const QgsWkbException &e )
62  {
63  Q_UNUSED( e );
64  QgsDebugMsg( "WKB exception: " + e.what() );
65  geom.reset();
66  }
67  }
68 
69  return geom;
70 }

167 std::unique_ptr<QgsPolygon> QgsGeometryFactory::fromPolygonXY( const QgsPolygonXY &polygon )
168 {
169  std::unique_ptr< QgsPolygon > poly = qgis::make_unique< QgsPolygon >();
170 
171  QVector<QgsCurve *> holes;
172  for ( int i = 0; i < polygon.size(); ++i )
173  {
174  std::unique_ptr< QgsLineString > l = linestringFromPolyline( polygon.at( i ) );
175  l->close();
176 
177  if ( i == 0 )
178  {
179  poly->setExteriorRing( l.release() );
180  }
181  else
182  {
183  holes.push_back( l.release() );
184  }
185  }
186  poly->setInteriorRings( holes );
187  return poly;
188 }

200 std::unique_ptr<QgsLineString> QgsGeometryFactory::linestringFromPolyline( const QgsPolylineXY &polyline )
201 {

204  QVector< double > y;

252 std::unique_ptr<QgsGeometryCollection> QgsGeometryFactory::createCollectionOfType( QgsWkbTypes::Type t )
253 {
254  QgsWkbTypes::Type type = QgsWkbTypes::flatType( QgsWkbTypes::multiType( t ) );
255  std::unique_ptr< QgsGeometryCollection > collect;
256  switch ( type )
257  {

259  collect = qgis::make_unique< QgsMultiPoint >();
260  break;
261  case QgsWkbTypes::MultiLineString:
262  collect = qgis::make_unique< QgsMultiLineString >();
263  break;
264  case QgsWkbTypes::MultiCurve:
265  collect = qgis::make_unique< QgsMultiCurve >();
266  break;
267  case QgsWkbTypes::MultiPolygon:
268  collect = qgis::make_unique< QgsMultiPolygon >();
269  break;
270  case QgsWkbTypes::MultiSurface:
271  collect = qgis::make_unique< QgsMultiSurface >();
272  break;
273  case QgsWkbTypes::GeometryCollection:
274  collect = qgis::make_unique< QgsGeometryCollection >();
275  break;
276  default:
277  // should not be possible
278  return nullptr;
279  }
280  if ( QgsWkbTypes::hasM( t ) )
281  collect->addMValue();
282  if ( QgsWkbTypes::hasZ( t ) )
283  collect->addZValue();
284 
285  return collect;
