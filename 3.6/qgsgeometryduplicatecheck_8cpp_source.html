42 void QgsGeometryDuplicateCheck::collectErrors( const QMap<QString, QgsFeaturePool *> &featurePools, QList<QgsGeometryCheckError *> &errors, QStringList &messages, QgsFeedback *feedback, const LayerFeatureIds &ids ) const
43 {
44  QMap<QString, QgsFeatureIds> featureIds = ids.isEmpty() ? allLayerFeatureIds( featurePools ) : ids.toMap();
45  QgsGeometryCheckerUtils::LayerFeatures layerFeaturesA( featurePools, featureIds, compatibleGeometryTypes(), feedback, mContext, true );
46  QList<QString> layerIds = featureIds.keys();
47  for ( const QgsGeometryCheckerUtils::LayerFeature &layerFeatureA : layerFeaturesA )
48  {
49  // Ensure each pair of layers only gets compared once: remove the current layer from the layerIds, but add it to the layerList for layerFeaturesB
50  layerIds.removeOne( layerFeatureA.layer()->id() );
51 
52  QgsRectangle bboxA = layerFeatureA.geometry().boundingBox();
53  std::unique_ptr< QgsGeometryEngine > geomEngineA = QgsGeometryCheckerUtils::createGeomEngine( layerFeatureA.geometry().constGet(), mContext->tolerance );
54  if ( !geomEngineA->isValid() )
55  {
56  messages.append( tr( "Duplicate check failed for (%1): the geometry is invalid" ).arg( layerFeatureA.id() ) );
57  continue;
58  }
59  QMap<QString, QList<QgsFeatureId>> duplicates;
60 
61  QgsWkbTypes::GeometryType geomType = layerFeatureA.feature().geometry().type();
62  QgsGeometryCheckerUtils::LayerFeatures layerFeaturesB( featurePools, QList<QString>() << layerFeatureA.layer()->id() << layerIds, bboxA, {geomType}, mContext );
63  for ( const QgsGeometryCheckerUtils::LayerFeature &layerFeatureB : layerFeaturesB )
64  {
65  // > : only report overlaps within same layer once
66  if ( layerFeatureA.layer()->id() == layerFeatureB.layer()->id() && layerFeatureB.feature().id() >= layerFeatureA.feature().id() )
67  {
68  continue;
69  }
70  QString errMsg;
71  QgsAbstractGeometry *diffGeom = geomEngineA->symDifference( layerFeatureB.geometry().constGet(), &errMsg );
72  if ( errMsg.isEmpty() && diffGeom && diffGeom->isEmpty() )
73  {
74  duplicates[layerFeatureB.layer()->id()].append( layerFeatureB.feature().id() );
75  }
76  else if ( !errMsg.isEmpty() )
77  {
78  messages.append( tr( "Duplicate check failed for (%1, %2): %3" ).arg( layerFeatureA.id(), layerFeatureB.id(), errMsg ) );
79  }
80  delete diffGeom;
81  }
82  if ( !duplicates.isEmpty() )
83  {
84  errors.append( new QgsGeometryDuplicateCheckError( this, layerFeatureA, layerFeatureA.geometry().constGet()->centroid(), featurePools, duplicates ) );
85  }
86  }
87 }
