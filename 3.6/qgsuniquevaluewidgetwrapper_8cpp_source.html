57 void QgsUniqueValuesWidgetWrapper::initWidget( QWidget *editor )
58 {
59  mComboBox = qobject_cast<QComboBox *>( editor );
60  mLineEdit = qobject_cast<QLineEdit *>( editor );
61 
62  QStringList sValues;
63 
64  QSet< QVariant> values = layer()->uniqueValues( fieldIdx() );
65 
66  Q_FOREACH ( const QVariant &v, values )
67  {
68  if ( mComboBox )
69  {
70  mComboBox->addItem( v.toString(), v );
71  }
72 
73  if ( mLineEdit )
74  {
75  sValues << v.toString();
76  }
77  }
78 
79  if ( mLineEdit )
80  {
81  QgsFilterLineEdit *fle = qobject_cast<QgsFilterLineEdit *>( editor );
82  if ( fle && !( field().type() == QVariant::Int || field().type() == QVariant::Double || field().type() == QVariant::LongLong || field().type() == QVariant::Date ) )
83  {
84  fle->setNullValue( QgsApplication::nullRepresentation() );
85  }
86 
87  QCompleter *c = new QCompleter( sValues );
88  c->setCaseSensitivity( Qt::CaseInsensitive );
89  c->setCompletionMode( QCompleter::PopupCompletion );
90  mLineEdit->setCompleter( c );
91 
92  connect( mLineEdit, &QLineEdit::textChanged, this, [ = ]( const QString & value ) { emit valueChanged( value ); } );
93  }
94 
95  if ( mComboBox )
96  {
97  connect( mComboBox, static_cast<void ( QComboBox::* )( int )>( &QComboBox::currentIndexChanged ),
