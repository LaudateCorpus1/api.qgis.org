66 bool QgsLegendSymbolItem::isScaleOK( double scale ) const
67 {
68  if ( scale <= 0 )
69  return true;
70  if ( mScaleMinDenom <= 0 && mScaleMaxDenom <= 0 )
71  return true;
72  if ( mScaleMinDenom > 0 && mScaleMinDenom > scale )

74  if ( mScaleMaxDenom > 0 && mScaleMaxDenom < scale )
