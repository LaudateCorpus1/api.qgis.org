59 void QgsHtmlAnnotation::setSourceFile( const QString &htmlFile )
60 {
61  QFile file( htmlFile );
62  mHtmlFile = htmlFile;
63  if ( !file.open( QIODevice::ReadOnly | QIODevice::Text ) )
64  {
65  mHtmlSource.clear();
66  }
67  else
68  {
69  QTextStream in( &file );
70  in.setCodec( "UTF-8" );
71  mHtmlSource = in.readAll();
72  }
73 
74  file.close();
75  setAssociatedFeature( associatedFeature() );
76  emit appearanceChanged();
77 }

113 void QgsHtmlAnnotation::readXml( const QDomElement &itemElem, const QgsReadWriteContext &context )
114 {
115  mHtmlFile = itemElem.attribute( QStringLiteral( "htmlfile" ), QString() );
116  QDomElement annotationElem = itemElem.firstChildElement( QStringLiteral( "AnnotationItem" ) );
117  if ( !annotationElem.isNull() )
118  {
119  _readXml( annotationElem, context );
120  }
121 
122  // upgrade old layer
123  if ( !mapLayer() && itemElem.hasAttribute( QStringLiteral( "vectorLayer" ) ) )
124  {
125  setMapLayer( QgsProject::instance()->mapLayer( itemElem.attribute( QStringLiteral( "vectorLayer" ) ) ) );
126  }
127 
128  if ( mWebPage )
129  {
130  setSourceFile( mHtmlFile );
131  }
132 }

134 void QgsHtmlAnnotation::setAssociatedFeature( const QgsFeature &feature )
135 {
136  QgsAnnotation::setAssociatedFeature( feature );
137  QString newText;
138  QgsVectorLayer *vectorLayer = qobject_cast< QgsVectorLayer * >( mapLayer() );
139  if ( feature.isValid() && vectorLayer )
140  {
141  QgsExpressionContext context( QgsExpressionContextUtils::globalProjectLayerScopes( vectorLayer ) );
142  context.setFeature( feature );
143  newText = QgsExpression::replaceExpressionText( mHtmlSource, &context );
144  }
145  else
146  {
147  newText = mHtmlSource;
148  }
149  mWebPage->mainFrame()->setHtml( newText );
150  emit appearanceChanged();
151 }
