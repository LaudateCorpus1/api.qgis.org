69  void executeRequest( const QgsServerRequest &request, QgsServerResponse &response,
70  const QgsProject *project ) override
71  {
72  const QgsWmsParameters parameters( QUrlQuery( request.url() ) );
73 
74  QString version = parameters.version();
75  if ( version.isEmpty() )
76  {
77  // WMTVER needs to be supported by WMS 1.1.1 for backwards
78  // compatibility with WMS 1.0.0
79  version = parameters.wmtver();
80  }
81 
82  // Set the default version
83  if ( version.isEmpty() || !parameters.versionIsValid( version ) )
84  {
85  version = mVersion;
86  }
87 
88  // Get the request
89  const QString req = parameters.request();
90  if ( req.isEmpty() )
91  {
92  throw QgsServiceException( QStringLiteral( "OperationNotSupported" ),
93  QStringLiteral( "Please check the value of the REQUEST parameter" ) );
94  }
95 
96  if ( ( mVersion.compare( QLatin1String( "1.1.1" ) ) == 0 \
97  && req.compare( QLatin1String( "capabilities" ) ) == 0 )
98  || req.compare( QLatin1String( "GetCapabilities" ) ) == 0 )
99  {
100  writeGetCapabilities( mServerIface, project, version, request, response, false );
101  }
102  else if ( QSTR_COMPARE( req, "GetProjectSettings" ) )
103  {
104  //getProjectSettings extends WMS 1.3.0 capabilities
105  version = QStringLiteral( "1.3.0" );
106  writeGetCapabilities( mServerIface, project, version, request, response, true );
107  }
108  else if ( QSTR_COMPARE( req, "GetMap" ) )
109  {
110  QString format = parameters.formatAsString();
111  if QSTR_COMPARE( format, "application/dxf" )
112  {
113  writeAsDxf( mServerIface, project, version, request, response );
114  }
115  else
116  {
117  writeGetMap( mServerIface, project, version, request, response );
118  }
119  }
120  else if ( QSTR_COMPARE( req, "GetFeatureInfo" ) )
121  {
122  writeGetFeatureInfo( mServerIface, project, version, request, response );
123  }
124  else if ( QSTR_COMPARE( req, "GetContext" ) )
125  {
126  writeGetContext( mServerIface, project, version, request, response );
127  }
128  else if ( QSTR_COMPARE( req, "GetSchemaExtension" ) )
129  {
130  writeGetSchemaExtension( mServerIface, version, request, response );
131  }
132  else if ( QSTR_COMPARE( req, "GetStyle" ) )
133  {
134  writeGetStyle( mServerIface, project, version, request, response );
135  }
136  else if ( QSTR_COMPARE( req, "GetStyles" ) )
137  {
138  writeGetStyles( mServerIface, project, version, request, response );
139  }
140  else if ( QSTR_COMPARE( req, "DescribeLayer" ) )
141  {
142  writeDescribeLayer( mServerIface, project, version, request, response );
143  }
144  else if ( QSTR_COMPARE( req, "GetLegendGraphic" ) || QSTR_COMPARE( req, "GetLegendGraphics" ) )
145  {
146  writeGetLegendGraphics( mServerIface, project, version, request, response );
147  }
148  else if ( QSTR_COMPARE( req, "GetPrint" ) )
149  {
150  writeGetPrint( mServerIface, project, version, request, response );
151  }
152  else
153  {
154  // Operation not supported
155  throw QgsServiceException( QStringLiteral( "OperationNotSupported" ),
156  QString( "Request %1 is not supported" ).arg( req ) );
157  }
158  }
159 
160  private:
161  QString mVersion;
162  QgsServerInterface *mServerIface = nullptr;
163  };
