112  {
113  const QString name = parameter.name();
114  QString message;
115 
116  switch ( code )
117  {

121  break;

126  break;
127  }
128  case OGC_InvalidFormat:

131  break;

136  break;

146  break;

148  case OGC_LayerNotQueryable:

151  break;
