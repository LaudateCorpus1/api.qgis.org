72 bool QgsProcessingRegistry::addProvider( QgsProcessingProvider *provider )
73 {
74  if ( !provider )
75  return false;
76 
77  if ( mProviders.contains( provider->id() ) )
78  {
79  QgsLogger::warning( QStringLiteral( "Duplicate provider %1 registered" ).arg( provider->id() ) );
80  delete provider;
81  return false;
82  }
83 
84  if ( !provider->load() )
85  {
86  QgsLogger::warning( QStringLiteral( "Provider %1 cannot load" ).arg( provider->id() ) );
87  delete provider;
88  return false;
89  }
90 
91  provider->setParent( this );
92  mProviders[ provider->id()] = provider;
93  emit providerAdded( provider->id() );
94  return true;
95 }

97 bool QgsProcessingRegistry::removeProvider( QgsProcessingProvider *provider )
98 {

103 
104  if ( !mProviders.contains( id ) )

136 const QgsProcessingAlgorithm *QgsProcessingRegistry::algorithmById( const QString &id ) const
137 {
138  QMap<QString, QgsProcessingProvider *>::const_iterator it = mProviders.constBegin();
139  for ( ; it != mProviders.constEnd(); ++it )
140  {
141  const auto constAlgorithms = it.value()->algorithms();
142  for ( const QgsProcessingAlgorithm *alg : constAlgorithms )
143  if ( alg->id() == id )
144  return alg;
145  }
146 
147  // try mapping 'qgis' algs to 'native' algs - this allows us to freely move algorithms
148  // from the python 'qgis' provider to the c++ 'native' provider without breaking API
149  // or existing models
150  if ( id.startsWith( QLatin1String( "qgis:" ) ) )
151  {
152  QString newId = QStringLiteral( "native:" ) + id.mid( 5 );
153  return algorithmById( newId );
154  }
155  return nullptr;
156 }

168 bool QgsProcessingRegistry::addParameterType( QgsProcessingParameterType *type )
169 {
170  if ( !mParameterTypes.contains( type->id() ) )
171  {
172  mParameterTypes.insert( type->id(), type );
173  emit parameterTypeAdded( type );
174  return true;
175  }
176  else
177  {
178  QgsLogger::warning( QStringLiteral( "Duplicate parameter type %1 (\"%2\") registered" ).arg( type->id(), type->name() ) );
179 
180  if ( mParameterTypes.value( type->id() ) != type )
181  delete type;
182 
183  return false;
184  }
