59 bool QgsFields::append( const QgsField &field, FieldOrigin origin, int originIndex )
60 {
61  if ( d->nameToIndex.contains( field.name() ) )
62  return false;
63 
64  if ( originIndex == -1 && origin == OriginProvider )
65  originIndex = d->fields.count();
66  d->fields.append( Field( field, origin, originIndex ) );
67 
68  d->nameToIndex.insert( field.name(), d->fields.count() - 1 );
69  return true;
70 }

73 {

84  d->fields[ fieldIdx ].field.setName( name );
85  d->nameToIndex.remove( oldName );

102 {
103  if ( !exists( fieldIdx ) )
104  return;
105 
106  d->fields.remove( fieldIdx );
107  d->nameToIndex.clear();

324 int QgsFields::lookupField( const QString &fieldName ) const
325 {
326  if ( fieldName.isEmpty() ) //shortcut
327  return -1;
328 
329  for ( int idx = 0; idx < count(); ++idx )
330  {
331  if ( d->fields[idx].field.name() == fieldName )
332  return idx;
333  }
334 
335  for ( int idx = 0; idx < count(); ++idx )
336  {
337  if ( QString::compare( d->fields[idx].field.name(), fieldName, Qt::CaseInsensitive ) == 0 )
338  return idx;
339  }
340 
341  for ( int idx = 0; idx < count(); ++idx )
342  {
343  QString alias = d->fields[idx].field.alias();
344  if ( !alias.isEmpty() && QString::compare( alias, fieldName, Qt::CaseInsensitive ) == 0 )
345  return idx;
346  }
347 
348  return -1;
349 }
