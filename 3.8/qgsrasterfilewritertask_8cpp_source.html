52 bool QgsRasterFileWriterTask::run()
53 {
54  if ( !mPipe )
55  return false;
56 
57  connect( mFeedback.get(), &QgsRasterBlockFeedback::progressChanged, this, &QgsRasterFileWriterTask::setProgress );
58 
59  mError = mWriter.writeRaster( mPipe.get(), mColumns, mRows, mExtent, mCrs, mTransformContext, mFeedback.get() );
60 
61  return mError == QgsRasterFileWriter::NoError;
62 }
