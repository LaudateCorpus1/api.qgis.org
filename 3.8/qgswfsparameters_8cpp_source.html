71  QStringList QgsWfsParameter::toStringListWithExp( const QString &exp ) const
72  {
73  QStringList theList;
74 
75  QString val = mValue.toString();
76  if ( val.isEmpty() )
77  return theList;
78 
79  if ( exp.isEmpty() )
80  theList << val;
81  else
82  {
83  QRegExp rx( exp );
84  if ( rx.indexIn( val, 0 ) == -1 )
85  {
86  theList << val;
87  }
88  else
89  {
90  int pos = 0;
91  while ( ( pos = rx.indexIn( val, pos ) ) != -1 )
92  {
93  theList << rx.cap( 1 );
94  pos += rx.matchedLength();
95  }
96  }
97  }
98 
99  return theList;
100  }

227  QgsWfsParameters::Format QgsWfsParameters::outputFormat() const
228  {
229  QString fStr = outputFormatAsString();
230 
231  if ( fStr.isEmpty() )
232  {
233  if ( versionAsNumber() >= QgsProjectVersion( 1, 1, 0 ) )
234  return Format::GML3;
235  else
236  return Format::GML2;
237  }
238 
239  Format f = Format::NONE;
240  if ( fStr.compare( QLatin1String( "text/xml; subtype=gml/2.1.2" ), Qt::CaseInsensitive ) == 0 )
241  f = Format::GML2;
242  else if ( fStr.compare( QLatin1String( "text/xml; subtype=gml/3.1.1" ), Qt::CaseInsensitive ) == 0 )
243  f = Format::GML3;
244  else if ( fStr.compare( QLatin1String( "application/vnd.geo+json" ), Qt::CaseInsensitive ) == 0 )
245  f = Format::GeoJSON;
246  else if ( fStr.compare( QLatin1String( "gml2" ), Qt::CaseInsensitive ) == 0 )
247  f = Format::GML2;
248  else if ( fStr.compare( QLatin1String( "gml3" ), Qt::CaseInsensitive ) == 0 )
249  f = Format::GML3;
250  else if ( fStr.compare( QLatin1String( "geojson" ), Qt::CaseInsensitive ) == 0 )
251  f = Format::GeoJSON;
252 
253  if ( f == Format::NONE &&
254  request().compare( QLatin1String( "describefeaturetype" ), Qt::CaseInsensitive ) == 0 &&
255  fStr.compare( QLatin1String( "xmlschema" ), Qt::CaseInsensitive ) == 0 )
256  f = Format::GML2;
257 
258  return f;
259  }
