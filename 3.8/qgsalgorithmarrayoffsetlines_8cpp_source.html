104 bool QgsCreateArrayOffsetLinesAlgorithm::prepareAlgorithm( const QVariantMap &parameters, QgsProcessingContext &context, QgsProcessingFeedback * )
105 {
106  mCount = parameterAsInt( parameters, QStringLiteral( "COUNT" ), context );
107  mDynamicCount = QgsProcessingParameters::isDynamic( parameters, QStringLiteral( "COUNT" ) );
108  if ( mDynamicCount )
109  mCountProperty = parameters.value( QStringLiteral( "COUNT" ) ).value< QgsProperty >();
110 
111  mOffsetStep = parameterAsDouble( parameters, QStringLiteral( "OFFSET" ), context );
112  mDynamicOffset = QgsProcessingParameters::isDynamic( parameters, QStringLiteral( "OFFSET" ) );
113  if ( mDynamicOffset )
114  mOffsetStepProperty = parameters.value( QStringLiteral( "OFFSET" ) ).value< QgsProperty >();
115 
116  mSegments = parameterAsInt( parameters, QStringLiteral( "SEGMENTS" ), context );
117  mJoinStyle = static_cast< QgsGeometry::JoinStyle>( 1 + parameterAsInt( parameters, QStringLiteral( "JOIN_STYLE" ), context ) );
118  mMiterLimit = parameterAsDouble( parameters, QStringLiteral( "MITER_LIMIT" ), context );
119 
120  return true;

123 QgsFeatureList QgsCreateArrayOffsetLinesAlgorithm::processFeature( const QgsFeature &feature, QgsProcessingContext &context, QgsProcessingFeedback * )
124 {
125  QgsFeatureList result = QgsFeatureList();
126 
127  if ( feature.hasGeometry() )
128  {
129  const QgsGeometry geometry = feature.geometry();
130  const QgsAttributes attr = feature.attributes();
131 
132  int count = mCount;
133  if ( mDynamicCount )
134  count = mCountProperty.valueAsInt( context.expressionContext(), count );
135  result.reserve( count + 1 );
136 
137  double offsetStep = mOffsetStep;
138  if ( mDynamicOffset )
139  offsetStep = mOffsetStepProperty.valueAsDouble( context.expressionContext(), offsetStep );
140 
141  {
142  QgsFeature original = feature;
143  original.setGeometry( geometry );
144  QgsAttributes outAttrs = attr;
145  outAttrs << QVariant( 0 ) << QVariant( 0 );
146  original.setAttributes( outAttrs );
147  result << original;
148  }
149 
150  double offset = 0;
151  for ( int i = 0; i < count; ++i )
152  {
153  offset += offsetStep;
154  // FIXME: shouldn't we use QgsVectorLayerUtils::createFeature?
155  QgsFeature offsetFeature = feature;
156  const QgsGeometry offsetGeometry = geometry.offsetCurve( offset, mSegments, mJoinStyle, mMiterLimit );
157  offsetFeature.setGeometry( offsetGeometry );
158  QgsAttributes outAttrs = attr;
159  outAttrs << QVariant( i + 1 ) << QVariant( offset );
160  offsetFeature.setAttributes( outAttrs );
161  result << offsetFeature;
162  }
163  }
164  else
165  {
166  result << feature;
167  }
168 
169  return result;
170 }
