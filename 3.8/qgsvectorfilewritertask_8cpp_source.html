48 bool QgsVectorFileWriterTask::run()
49 {
50  if ( mError != QgsVectorFileWriter::NoError )
51  return false;
52 
53  connect( mOptions.feedback, &QgsFeedback::progressChanged, this, &QgsVectorFileWriterTask::setProgress );

55 
56  mError = QgsVectorFileWriter::writeAsVectorFormat(
