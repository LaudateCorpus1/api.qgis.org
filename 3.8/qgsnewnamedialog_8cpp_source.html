180 QStringList QgsNewNameDialog::matching( const QStringList &newNames, const QStringList &existingNames,
181  Qt::CaseSensitivity cs )
182 {
183  QStringList list;
184 
185  const auto constNewNames = newNames;
186  for ( const QString &newName : constNewNames )
187  {
188  const auto constExistingNames = existingNames;
189  for ( const QString &existingName : constExistingNames )
190  {
191  if ( existingName.compare( newName, cs ) == 0 )
192  {
193  list << existingName;
194  }
195  }
196  }
