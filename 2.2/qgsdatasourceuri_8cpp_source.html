406 QString QgsDataSourceURI::getValue( const QString &uri, int &i )
407 {
408  skipBlanks( uri, i );
409 
410  // Get the parameter value
411  QString pval;
412  if ( uri[i] == '\'' || uri[i] == '"' )
413  {
414  QChar delim = uri[i];
415 
416  i++;
417 
418  // value is quoted
419  for ( ;; )
420  {
421  if ( i == uri.length() )
422  {
423  QgsDebugMsg( "unterminated quoted string in connection info string" );
424  return pval;
425  }
426 
427  if ( uri[i] == '\\' )
428  {
429  i++;
430  if ( i == uri.length() )
431  continue;
432  if ( uri[i] != delim && uri[i] != '\\' )
433  i--;
434  }
435  else if ( uri[i] == delim )
436  {
437  i++;
438  break;
439  }
440 
441  pval += uri[i++];
442  }
443  }
444  else
445  {
446  // value is not quoted
447  while ( i < uri.length() )
448  {
449  if ( uri[i].isSpace() )
450  {
451  // end of value
452  break;
453  }
454 
455  if ( uri[i] == '\\' )
456  {
457  i++;
458  if ( i == uri.length() )
459  break;
460  if ( uri[i] != '\\' && uri[i] != '\'' )
461  i--;
462  }
463 
464  pval += uri[i++];
465  }
466  }
467 
468  skipBlanks( uri, i );
469 
470  return pval;
471 }
