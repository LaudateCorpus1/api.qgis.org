124  void readNextEntry()
125  {
126  QgsFeature f;
127  SpatialIndex::Region r;
128  QgsFeatureId id;
129  while ( mFi.nextFeature( f ) )
130  {
131  if ( QgsSpatialIndex::featureInfo( f, r, id ) )
132  {
133  mNextData = new RTree::Data( 0, nullptr, r, id );
134  return;
135  }
136  }
