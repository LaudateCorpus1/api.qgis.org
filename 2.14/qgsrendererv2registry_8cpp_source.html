78 bool QgsRendererV2Registry::addRenderer( QgsRendererV2AbstractMetadata* metadata )
79 {
80  if ( !metadata || mRenderers.contains( metadata->name() ) )
81  return false;
82 
83  mRenderers[metadata->name()] = metadata;
84  mRenderersOrder << metadata->name();

88 bool QgsRendererV2Registry::removeRenderer( const QString& rendererName )

90  if ( !mRenderers.contains( rendererName ) )

92
