32 QgsWKBTypes::Type QgsWKBTypes::parseType( const QString &wktStr )
33 {
34  QString typestr = wktStr.left( wktStr.indexOf( '(' ) ).simplified().remove( ' ' );
35 
36  QMap<QgsWKBTypes::Type, QgsWKBTypes::wkbEntry>* knownTypes = entries();
37  QMap<QgsWKBTypes::Type, QgsWKBTypes::wkbEntry>::const_iterator it = knownTypes->constBegin();
38  for ( ; it != knownTypes->constEnd(); ++it )
39  {
40  if ( it.value().mName.compare( typestr, Qt::CaseInsensitive ) == 0 )
41  {
42  return it.key();
43  }
44  }
45  return Unknown;
46 }
