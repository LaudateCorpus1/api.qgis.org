112 QgsRenderContext QgsRenderContext::fromMapSettings( const QgsMapSettings& mapSettings )
113 {
114  QgsRenderContext ctx;
115  ctx.setMapToPixel( mapSettings.mapToPixel() );
116  ctx.setExtent( mapSettings.visibleExtent() );
117  ctx.setFlag( DrawEditingInfo, mapSettings.testFlag( QgsMapSettings::DrawEditingInfo ) );
118  ctx.setFlag( ForceVectorOutput, mapSettings.testFlag( QgsMapSettings::ForceVectorOutput ) );
119  ctx.setFlag( UseAdvancedEffects, mapSettings.testFlag( QgsMapSettings::UseAdvancedEffects ) );
120  ctx.setFlag( UseRenderingOptimization, mapSettings.testFlag( QgsMapSettings::UseRenderingOptimization ) );
121  ctx.setCoordinateTransform( nullptr );
122  ctx.setSelectionColor( mapSettings.selectionColor() );
123  ctx.setFlag( DrawSelection, mapSettings.testFlag( QgsMapSettings::DrawSelection ) );
124  ctx.setFlag( DrawSymbolBounds, mapSettings.testFlag( QgsMapSettings::DrawSymbolBounds ) );
125  ctx.setFlag( RenderMapTile, mapSettings.testFlag( QgsMapSettings::RenderMapTile ) );
126  ctx.setRasterScaleFactor( 1.0 );
127  ctx.setScaleFactor( mapSettings.outputDpi() / 25.4 ); // = pixels per mm
128  ctx.setRendererScale( mapSettings.scale() );
129  ctx.setExpressionContext( mapSettings.expressionContext() );
130 
131  //this flag is only for stopping during the current rendering progress,
132  //so must be false at every new render operation
133  ctx.setRenderingStopped( false );
134 
135  return ctx;
136 }
