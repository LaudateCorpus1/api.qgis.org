419 inline double QgsRasterBlock::readValue( void *data, QGis::DataType type, qgssize index )
420 {
421  if ( !data )
422  {
423  return std::numeric_limits<double>::quiet_NaN();
424  }
425 
426  switch ( type )
427  {
428  case QGis::Byte:
429  return static_cast< double >(( static_cast< quint8 * >( data ) )[index] );
430  break;
431  case QGis::UInt16:
432  return static_cast< double >(( static_cast< quint16 * >( data ) )[index] );
433  break;
434  case QGis::Int16:
435  return static_cast< double >(( static_cast< qint16 * >( data ) )[index] );
436  break;
437  case QGis::UInt32:
438  return static_cast< double >(( static_cast< quint32 * >( data ) )[index] );
439  break;
440  case QGis::Int32:
441  return static_cast< double >(( static_cast< qint32 * >( data ) )[index] );
442  break;
443  case QGis::Float32:
444  return static_cast< double >(( static_cast< float * >( data ) )[index] );
445  break;
446  case QGis::Float64:
447  return static_cast< double >(( static_cast< double * >( data ) )[index] );
448  break;
449  default:
450  QgsDebugMsg( QString( "Data type %1 is not supported" ).arg( type ) );
451  break;
452  }
453 
454  return std::numeric_limits<double>::quiet_NaN();
455 }

457 inline void QgsRasterBlock::writeValue( void *data, QGis::DataType type, qgssize index, double value )
458 {
459  if ( !data ) return;
460 
461  switch ( type )

463  case QGis::Byte:

465  break;
466  case QGis::UInt16:

468  break;
469  case QGis::Int16:
470  ( static_cast< qint16 * >( data ) )[index] = static_cast< qint16 >( value );
471  break;
472  case QGis::UInt32:
473  ( static_cast< quint32 * >( data ) )[index] = static_cast< quint32 >( value );
474  break;
475  case QGis::Int32:
476  ( static_cast< qint32 * >( data ) )[index] = static_cast< qint32 >( value );

478  case QGis::Float32:
479  ( static_cast< float * >( data ) )[index] = static_cast< float >( value );
